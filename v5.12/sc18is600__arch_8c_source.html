271  if (sc18is600.status == Sc18Is600SendingRequest) {
272  sc18is600.status = Sc18Is600WaitingForI2C;
273  Sc18Is600Unselect();
274  } else if (sc18is600.status == Sc18Is600ReadingI2CStat) {
275  sc18is600.status = Sc18Is600ReadingBuffer;
276  sc18is600.priv_tx_buf[0] = Sc18Is600_Cmd_Read_Buffer;
277  // debug

279  Sc18Is600Select();
280  sc18is600_setup_SPI_DMA(sc18is600.rx_len + 1);
281  } else if (sc18is600.status == Sc18Is600ReadingBuffer) {
282  sc18is600.status = Sc18Is600TransactionComplete;
283  Sc18Is600Unselect();
284  }
285  break;
