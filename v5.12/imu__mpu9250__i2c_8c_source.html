119 void imu_mpu9250_event(void)
120 {
121  uint32_t now_ts = get_sys_time_usec();
122 
123  // If the MPU9250 I2C transaction has succeeded: convert the data
124  mpu9250_i2c_event(&imu_mpu9250.mpu);
125 
126  if (imu_mpu9250.mpu.data_available) {
127  // set channel order
128  struct Int32Vect3 accel = {
129  IMU_MPU9250_X_SIGN * (int32_t)(imu_mpu9250.mpu.data_accel.value[IMU_MPU9250_CHAN_X]),
130  IMU_MPU9250_Y_SIGN * (int32_t)(imu_mpu9250.mpu.data_accel.value[IMU_MPU9250_CHAN_Y]),
131  IMU_MPU9250_Z_SIGN * (int32_t)(imu_mpu9250.mpu.data_accel.value[IMU_MPU9250_CHAN_Z])
132  };
133  struct Int32Rates rates = {
134  IMU_MPU9250_X_SIGN * (int32_t)(imu_mpu9250.mpu.data_rates.value[IMU_MPU9250_CHAN_X]),
135  IMU_MPU9250_Y_SIGN * (int32_t)(imu_mpu9250.mpu.data_rates.value[IMU_MPU9250_CHAN_Y]),
136  IMU_MPU9250_Z_SIGN * (int32_t)(imu_mpu9250.mpu.data_rates.value[IMU_MPU9250_CHAN_Z])
137  };
138  // unscaled vector
139  VECT3_COPY(imu.accel_unscaled, accel);
140  RATES_COPY(imu.gyro_unscaled, rates);
141 
142  imu_mpu9250.mpu.data_available = false;
143 
144  imu_scale_gyro(&imu);
145  imu_scale_accel(&imu);
146  AbiSendMsgIMU_GYRO_INT32(IMU_MPU9250_ID, now_ts, &imu.gyro);
147  AbiSendMsgIMU_ACCEL_INT32(IMU_MPU9250_ID, now_ts, &imu.accel);
148  }
149 
150  // Test if mag data are updated
151  if (imu_mpu9250.mpu.akm.data_available) {
152  struct Int32Vect3 mag = {
153  IMU_MPU9250_X_SIGN * (int32_t)(imu_mpu9250.mpu.akm.data.value[IMU_MPU9250_CHAN_Y]),
154  IMU_MPU9250_Y_SIGN * (int32_t)(imu_mpu9250.mpu.akm.data.value[IMU_MPU9250_CHAN_X]),
155  -IMU_MPU9250_Z_SIGN * (int32_t)(imu_mpu9250.mpu.akm.data.value[IMU_MPU9250_CHAN_Z])
156  };
157  VECT3_COPY(imu.mag_unscaled, mag);
158  imu_mpu9250.mpu.akm.data_available = false;
159  imu_scale_mag(&imu);
160  AbiSendMsgIMU_MAG_INT32(IMU_MPU9250_ID, now_ts, &imu.mag);
161  }
162 
163 }
