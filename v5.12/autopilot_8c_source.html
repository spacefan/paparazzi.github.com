91 void autopilot_init(void)
92 {
93 #ifdef MODE_AUTO2
94  autopilot.mode_auto2 = MODE_AUTO2; // FIXME
95 #endif
96  autopilot.flight_time = 0;
97  autopilot.motors_on = false;
98  autopilot.kill_throttle = true;
99  autopilot.in_flight = false;
100  autopilot.ground_detected = false;
101  autopilot.detect_ground_once = false;
102  autopilot.use_rc = true;
103  autopilot.power_switch = false;
104 #ifdef POWER_SWITCH_GPIO
105  gpio_setup_output(POWER_SWITCH_GPIO);
106  gpio_clear(POWER_SWITCH_GPIO); // POWER OFF
107 #endif
108 
109  // call firmware specific init
110  autopilot_firmware_init();
111 
112  // static / generated AP init part is called later by main program
113  // and will set the correct initial mode
114 
115  // register messages
116  register_periodic_telemetry(DefaultPeriodic, PPRZ_MSG_ID_AUTOPILOT_VERSION, send_autopilot_version);
117  register_periodic_telemetry(DefaultPeriodic, PPRZ_MSG_ID_ALIVE, send_alive);
118  register_periodic_telemetry(DefaultPeriodic, PPRZ_MSG_ID_ATTITUDE, send_attitude);
119  register_periodic_telemetry(DefaultPeriodic, PPRZ_MSG_ID_DL_VALUE, send_dl_value);
120 #ifdef ACTUATORS
121  register_periodic_telemetry(DefaultPeriodic, PPRZ_MSG_ID_ACTUATORS, send_actuators);
122 #endif
123 #ifdef RADIO_CONTROL
124  register_periodic_telemetry(DefaultPeriodic, PPRZ_MSG_ID_RC, send_rc);
125 #endif
126 }
