92 void lidar_sf11_periodic(void)
93 {
94  switch (lidar_sf11.status) {
95  case LIDAR_SF11_REQ_READ:
96  // read two bytes
97  lidar_sf11.trans.buf[0] = 0; // set tx to zero
98  i2c_transceive(&LIDAR_SF11_I2C_DEV, &lidar_sf11.trans, lidar_sf11.addr, 1, 2);
99  break;
100  case LIDAR_SF11_READ_OK:
101  // process results
102  // filter data
103  lidar_sf11.distance_raw = update_median_filter_i(
104  &lidar_sf11_filter,
105  (uint32_t)((lidar_sf11.trans.buf[0] << 8) | lidar_sf11.trans.buf[1]));
106  lidar_sf11.distance = ((float)lidar_sf11.distance_raw)/100.0;
107 
108  // compensate AGL measurement for body rotation
109  if (lidar_sf11.compensate_rotation) {
110  float phi = stateGetNedToBodyEulers_f()->phi;
111  float theta = stateGetNedToBodyEulers_f()->theta;
112  float gain = (float)fabs( (double) (cosf(phi) * cosf(theta)));
113  lidar_sf11.distance = lidar_sf11.distance / gain;
114  }
115 
116  // send message (if requested)
117  if (lidar_sf11.update_agl) {
118  AbiSendMsgAGL(AGL_LIDAR_SF11_ID, lidar_sf11.distance);
119  }
120 
121  // reset status
122  lidar_sf11.status = LIDAR_SF11_REQ_READ;
123  break;
124  default:
125  break;
126  }
127 }
