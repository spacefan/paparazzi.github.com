136 /* Parse incomming InterMCU messages */
137 #pragma GCC diagnostic ignored "-Wcast-align"
138 static inline void intermcu_parse_msg(void (*rc_frame_handler)(void))
139 {
140  /* Parse the Inter MCU message */
141  uint8_t msg_id = imcu_msg_buf[1];
142  switch (msg_id) {
143  case DL_IMCU_RADIO_COMMANDS: {
144  uint8_t i;
145  uint8_t size = DL_IMCU_RADIO_COMMANDS_values_length(imcu_msg_buf);
146  intermcu.status = DL_IMCU_RADIO_COMMANDS_status(imcu_msg_buf);
147  for (i = 0; i < size; i++) {
148  radio_control.values[i] = DL_IMCU_RADIO_COMMANDS_values(imcu_msg_buf)[i];
149  }
150 
151  radio_control.frame_cpt++;
152  radio_control.time_since_last_frame = 0;
153  radio_control.status = RC_OK;
154  rc_frame_handler();
155  break;
156  }
157 
158  case DL_IMCU_FBW_STATUS: {
159  fbw_status.rc_status = DL_IMCU_FBW_STATUS_rc_status(imcu_msg_buf);
160  fbw_status.frame_rate = DL_IMCU_FBW_STATUS_frame_rate(imcu_msg_buf);
161  fbw_status.mode = DL_IMCU_FBW_STATUS_mode(imcu_msg_buf);
162  fbw_status.vsupply = DL_IMCU_FBW_STATUS_vsupply(imcu_msg_buf);
163  fbw_status.current = DL_IMCU_FBW_STATUS_current(imcu_msg_buf);
164  break;
165  }
166 
167 #if TELEMETRY_INTERMCU
168  case DL_IMCU_DATALINK: {
169  uint8_t size = DL_IMCU_DATALINK_msg_length(imcu_msg_buf);
170  uint8_t *msg = DL_IMCU_DATALINK_msg(imcu_msg_buf);
171  telemetry_intermcu_on_msg(0, msg, size);
172  break;
173  }
174 #endif
175 
176 #if IMCU_GPS
177  case DL_IMCU_REMOTE_GPS: {
178  uint32_t now_ts = get_sys_time_usec();
179  gps_imcu.ecef_pos.x = DL_IMCU_REMOTE_GPS_ecef_x(imcu_msg_buf);
180  gps_imcu.ecef_pos.y = DL_IMCU_REMOTE_GPS_ecef_y(imcu_msg_buf);
181  gps_imcu.ecef_pos.z = DL_IMCU_REMOTE_GPS_ecef_z(imcu_msg_buf);
182  SetBit(gps_imcu.valid_fields, GPS_VALID_POS_ECEF_BIT);
183 
184  gps_imcu.lla_pos.alt = DL_IMCU_REMOTE_GPS_alt(imcu_msg_buf);
185  gps_imcu.hmsl = DL_IMCU_REMOTE_GPS_hmsl(imcu_msg_buf);
186  SetBit(gps_imcu.valid_fields, GPS_VALID_HMSL_BIT);
187 
188  gps_imcu.ecef_vel.x = DL_IMCU_REMOTE_GPS_ecef_xd(imcu_msg_buf);
189  gps_imcu.ecef_vel.y = DL_IMCU_REMOTE_GPS_ecef_yd(imcu_msg_buf);
190  gps_imcu.ecef_vel.z = DL_IMCU_REMOTE_GPS_ecef_zd(imcu_msg_buf);
191  SetBit(gps_imcu.valid_fields, GPS_VALID_VEL_ECEF_BIT);
192 
193  gps_imcu.course = DL_IMCU_REMOTE_GPS_course(imcu_msg_buf);
194  gps_imcu.gspeed = DL_IMCU_REMOTE_GPS_gspeed(imcu_msg_buf);
195  SetBit(gps_imcu.valid_fields, GPS_VALID_COURSE_BIT);
196 
197  gps_imcu.pacc = DL_IMCU_REMOTE_GPS_pacc(imcu_msg_buf);
198  gps_imcu.sacc = DL_IMCU_REMOTE_GPS_sacc(imcu_msg_buf);
199  gps_imcu.num_sv = DL_IMCU_REMOTE_GPS_numsv(imcu_msg_buf);
200  gps_imcu.fix = DL_IMCU_REMOTE_GPS_fix(imcu_msg_buf);
201 
202  // set gps msg time
203  gps_imcu.last_msg_ticks = sys_time.nb_sec_rem;
204  gps_imcu.last_msg_time = sys_time.nb_sec;
205 
206  if (gps_imcu.fix >= GPS_FIX_3D) {
207  gps_imcu.last_3dfix_ticks = sys_time.nb_sec_rem;
208  gps_imcu.last_3dfix_time = sys_time.nb_sec;
209  }
210 
211  AbiSendMsgGPS(GPS_IMCU_ID, now_ts, &gps_imcu);
212  break;
213  }
214 
215 #endif
216 
217  default:
218  break;
219  }
