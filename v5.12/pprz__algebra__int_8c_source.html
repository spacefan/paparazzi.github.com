30 uint32_t int32_sqrt(uint32_t in)
31 {
32  if (in == 0) {
33  return 0;
34  } else {
35  uint32_t s1, s2;
36  uint8_t iter = 0;
37  s2 = in;
38  do {
39  s1 = s2;
40  s2 = in / s1;
41  s2 += s1;
42  s2 /= 2;
43  iter++;
44  } while (((s1 - s2) > 1) && (iter < INT32_SQRT_MAX_ITER));
45  return s2;
46  }
47 }

369  v_in->z) >> INT32_QUAT_FRAC;
