83 static inline void autopilot_arming_check_motors_on(void)
84 {
85  /* only allow switching motor if not in KILL mode */
86  if (autopilot_get_mode() != AP_MODE_KILL) {
87 
88  switch (autopilot_check_motor_status) {
89  case STATUS_INITIALISE_RC: // Wait until RC is initialised (it being centered is a good pointer to this)
90  if (THROTTLE_STICK_DOWN() && YAW_STICK_CENTERED() && PITCH_STICK_CENTERED() && ROLL_STICK_CENTERED()) {
91  autopilot_check_motor_status = STATUS_MOTORS_OFF;
92  }
93  break;
94  case STATUS_MOTORS_AUTOMATICALLY_OFF: // Motors were disarmed externally
95  //(possibly due to crash)
96  //wait extra delay before enabling the normal arming state machine
97  autopilot.motors_on = false;
98  autopilot_motors_on_counter = 0;
99  if (THROTTLE_STICK_DOWN() && YAW_STICK_CENTERED()) { // stick released
100  autopilot_check_motor_status = STATUS_MOTORS_AUTOMATICALLY_OFF_SAFETY_WAIT;
101  }
102  break;
103  case STATUS_MOTORS_AUTOMATICALLY_OFF_SAFETY_WAIT:
104  autopilot_motors_on_counter++;
105  if (autopilot_motors_on_counter >= MOTOR_ARMING_DELAY) {
106  autopilot_check_motor_status = STATUS_MOTORS_OFF;
107  }
108  break;
109  case STATUS_MOTORS_OFF:
110  autopilot.motors_on = false;
111  autopilot_motors_on_counter = 0;
112  if (THROTTLE_STICK_DOWN() && YAW_STICK_PUSHED()) { // stick pushed
113  autopilot_check_motor_status = STATUS_M_OFF_STICK_PUSHED;
114  }
115  break;
116  case STATUS_M_OFF_STICK_PUSHED:
117  autopilot.motors_on = false;
118  autopilot_motors_on_counter++;
119  if (autopilot_motors_on_counter >= MOTOR_ARMING_DELAY) {
120  autopilot_check_motor_status = STATUS_START_MOTORS;
121  } else if (!(THROTTLE_STICK_DOWN() && YAW_STICK_PUSHED())) { // stick released too soon
122  autopilot_check_motor_status = STATUS_MOTORS_OFF;
123  }
124  break;
125  case STATUS_START_MOTORS:
126  autopilot.motors_on = true;
127  autopilot_motors_on_counter = MOTOR_ARMING_DELAY;
128  if (!(THROTTLE_STICK_DOWN() && YAW_STICK_PUSHED())) { // wait until stick released
129  autopilot_check_motor_status = STATUS_MOTORS_ON;
130  }
131  break;
132  case STATUS_MOTORS_ON:
133  autopilot.motors_on = true;
134  autopilot_motors_on_counter = MOTOR_ARMING_DELAY;
135  if (THROTTLE_STICK_DOWN() && YAW_STICK_PUSHED()) { // stick pushed
136  autopilot_check_motor_status = STATUS_M_ON_STICK_PUSHED;
137  }
138  break;
139  case STATUS_M_ON_STICK_PUSHED:
140  autopilot.motors_on = true;
141  autopilot_motors_on_counter--;
142  if (autopilot_motors_on_counter == 0) {
143  autopilot_check_motor_status = STATUS_STOP_MOTORS;
144  } else if (!(THROTTLE_STICK_DOWN() && YAW_STICK_PUSHED())) { // stick released too soon
145  autopilot_check_motor_status = STATUS_MOTORS_ON;
146  }
147  break;
148  case STATUS_STOP_MOTORS:
149  autopilot.motors_on = false;
150  autopilot_motors_on_counter = 0;
151  if (!(THROTTLE_STICK_DOWN() && YAW_STICK_PUSHED())) { // wait until stick released
152  autopilot_check_motor_status = STATUS_MOTORS_OFF;
153  }
154  break;
155  default:
156  break;
157  }
158  }
159 }
