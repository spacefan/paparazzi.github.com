58 void baro_periodic(void)
59 {
60 
61  // Run some loops to get correct readings from the adc
62  if (startup_cnt > 0) {
63  --startup_cnt;
64 #ifdef BARO_LED
65  LED_TOGGLE(BARO_LED);
66  if (startup_cnt == 0) {

69 #endif
70  }
71  // Read the ADC

73 }
