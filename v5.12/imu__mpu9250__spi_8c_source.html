114 void imu_mpu9250_init(void)
115 {
116  /* MPU9250 */
117  mpu9250_spi_init(&imu_mpu9250.mpu, &(IMU_MPU9250_SPI_DEV), IMU_MPU9250_SPI_SLAVE_IDX);
118  // change the default configuration
119  imu_mpu9250.mpu.config.smplrt_div = IMU_MPU9250_SMPLRT_DIV;
120  imu_mpu9250.mpu.config.dlpf_gyro_cfg = IMU_MPU9250_GYRO_LOWPASS_FILTER;
121  imu_mpu9250.mpu.config.dlpf_accel_cfg = IMU_MPU9250_ACCEL_LOWPASS_FILTER;
122  imu_mpu9250.mpu.config.gyro_range = IMU_MPU9250_GYRO_RANGE;
123  imu_mpu9250.mpu.config.accel_range = IMU_MPU9250_ACCEL_RANGE;
124 
125 
126  /* "internal" ak8963 magnetometer as I2C slave */
127 #if IMU_MPU9250_READ_MAG
128  /* read 15 bytes for status, accel, gyro + 7 bytes for mag slave */
129  imu_mpu9250.mpu.config.nb_bytes = 22;
130  imu_mpu9250.mpu.config.nb_slaves = 1;
131 #endif
132  /* set callback function to configure mag */
133  imu_mpu9250.mpu.config.slaves[0].configure = &imu_mpu9250_configure_mag_slave;
134 
135  /* Set MPU I2C master clock */
136  imu_mpu9250.mpu.config.i2c_mst_clk = MPU9250_MST_CLK_400KHZ;
137  /* Enable I2C slave0 delayed sample rate */
138  imu_mpu9250.mpu.config.i2c_mst_delay = 1;
139 
140 
141  /* configure spi transaction for wait_slave4 */
142  imu_mpu9250.wait_slave4_trans.cpol = SPICpolIdleHigh;
143  imu_mpu9250.wait_slave4_trans.cpha = SPICphaEdge2;
144  imu_mpu9250.wait_slave4_trans.dss = SPIDss8bit;
145  imu_mpu9250.wait_slave4_trans.bitorder = SPIMSBFirst;
146  imu_mpu9250.wait_slave4_trans.cdiv = SPIDiv64;
147 
148  imu_mpu9250.wait_slave4_trans.select = SPISelectUnselect;
149  imu_mpu9250.wait_slave4_trans.slave_idx = IMU_MPU9250_SPI_SLAVE_IDX;

158  imu_mpu9250.slave4_ready = false;
159 }

166 #define Int16FromBuf(_buf,_idx) ((int16_t)(_buf[_idx] | (_buf[_idx+1] << 8)))
167 void imu_mpu9250_event(void)
168 {
169  uint32_t now_ts = get_sys_time_usec();
170 
171  // If the MPU9250 SPI transaction has succeeded: convert the data
172  mpu9250_spi_event(&imu_mpu9250.mpu);
173 
174  if (imu_mpu9250.mpu.data_available) {
175  // set channel order
176  struct Int32Vect3 accel = {
177  IMU_MPU9250_X_SIGN * (int32_t)(imu_mpu9250.mpu.data_accel.value[IMU_MPU9250_CHAN_X]),
178  IMU_MPU9250_Y_SIGN * (int32_t)(imu_mpu9250.mpu.data_accel.value[IMU_MPU9250_CHAN_Y]),
179  IMU_MPU9250_Z_SIGN * (int32_t)(imu_mpu9250.mpu.data_accel.value[IMU_MPU9250_CHAN_Z])
180  };
181  struct Int32Rates rates = {
182  IMU_MPU9250_X_SIGN * (int32_t)(imu_mpu9250.mpu.data_rates.value[IMU_MPU9250_CHAN_X]),
183  IMU_MPU9250_Y_SIGN * (int32_t)(imu_mpu9250.mpu.data_rates.value[IMU_MPU9250_CHAN_Y]),
184  IMU_MPU9250_Z_SIGN * (int32_t)(imu_mpu9250.mpu.data_rates.value[IMU_MPU9250_CHAN_Z])
185  };
186  // unscaled vector
187  VECT3_COPY(imu.accel_unscaled, accel);
188  RATES_COPY(imu.gyro_unscaled, rates);
189 
190 #if IMU_MPU9250_READ_MAG
191  if (!bit_is_set(imu_mpu9250.mpu.data_ext[6], 3)) { //mag valid just HOFL == 0
193  struct Int32Vect3 mag;
194  mag.x = (IMU_MPU9250_X_SIGN) * Int16FromBuf(imu_mpu9250.mpu.data_ext, 2 * IMU_MPU9250_CHAN_Y);
195  mag.y = (IMU_MPU9250_Y_SIGN) * Int16FromBuf(imu_mpu9250.mpu.data_ext, 2 * IMU_MPU9250_CHAN_X);
196  mag.z = -(IMU_MPU9250_Z_SIGN) * Int16FromBuf(imu_mpu9250.mpu.data_ext, 2 * IMU_MPU9250_CHAN_Z);
197  VECT3_COPY(imu.mag_unscaled, mag);
198  imu_scale_mag(&imu);
199  AbiSendMsgIMU_MAG_INT32(IMU_MPU9250_ID, now_ts, &imu.mag);
200  }
201 #endif
202 
203  imu_mpu9250.mpu.data_available = false;
204 
205  imu_scale_gyro(&imu);
206  imu_scale_accel(&imu);
207  AbiSendMsgIMU_GYRO_INT32(IMU_MPU9250_ID, now_ts, &imu.gyro);
208  AbiSendMsgIMU_ACCEL_INT32(IMU_MPU9250_ID, now_ts, &imu.accel);
209  }
210 
211 }

223 bool imu_mpu9250_configure_mag_slave(Mpu9250ConfigSet mpu_set, void *mpu)
224 {
225  // wait before starting the configuration of the mag
226  // doing to early may void the mode configuration
227  if (get_sys_time_float() < IMU_MPU9250_MAG_STARTUP_DELAY) {
228  return false;
229  }
230 
231  //config AK8963 soft reset
232  mpu_set_and_wait(mpu_set, mpu, MPU9250_REG_I2C_SLV4_ADDR, (MPU9250_MAG_ADDR >> 1));
233  mpu_set_and_wait(mpu_set, mpu, MPU9250_REG_I2C_SLV4_REG, AK8963_REG_CNTL2);
234  mpu_set_and_wait(mpu_set, mpu, MPU9250_REG_I2C_SLV4_DO, 1);
235  mpu_set_and_wait(mpu_set, mpu, MPU9250_REG_I2C_SLV4_CTRL, (1 << 7)); // Slave 4 enable
236 
237  mpu_wait_slave4_ready();
238 
239  // Set it to continious measuring mode 2
240  mpu_set_and_wait(mpu_set, mpu, MPU9250_REG_I2C_SLV4_ADDR, (MPU9250_MAG_ADDR >> 1));
241  mpu_set_and_wait(mpu_set, mpu, MPU9250_REG_I2C_SLV4_REG, AK8963_REG_CNTL1);
242  mpu_set_and_wait(mpu_set, mpu, MPU9250_REG_I2C_SLV4_DO, AK8963_CNTL1_CM_2);
243  mpu_set_and_wait(mpu_set, mpu, MPU9250_REG_I2C_SLV4_CTRL, (1 << 7)); // Slave 4 enable
244 
245  mpu_wait_slave4_ready();
246 
247  //Config SLV0 for continus Read
248  mpu_set_and_wait(mpu_set, mpu, MPU9250_REG_I2C_SLV0_ADDR, (MPU9250_MAG_ADDR >> 1) | MPU9250_SPI_READ);
249  mpu_set_and_wait(mpu_set, mpu, MPU9250_REG_I2C_SLV0_REG, AK8963_REG_HXL);
250  // Put the enable command as last.
251  mpu_set_and_wait(mpu_set, mpu, MPU9250_REG_I2C_SLV0_CTRL,
252  (1 << 7) | // Slave 0 enable
253  (7 << 0)); // Read 7 bytes (mag x,y,z + status)
254 
255  return true;
256 }

258 void mpu_wait_slave4_ready(void)
259 {
260  while (!imu_mpu9250.slave4_ready) {
261  if (imu_mpu9250.wait_slave4_trans.status == SPITransDone) {
262  imu_mpu9250.wait_slave4_tx_buf[0] = MPU9250_REG_I2C_MST_STATUS | MPU9250_SPI_READ;
263  spi_submit(imu_mpu9250.mpu.spi_p, &(imu_mpu9250.wait_slave4_trans));
264  }
265  }
266 }
