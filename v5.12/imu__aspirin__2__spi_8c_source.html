99 void imu_aspirin2_init(void)
100 {
101  mpu60x0_spi_init(&imu_aspirin2.mpu, &(ASPIRIN_2_SPI_DEV), ASPIRIN_2_SPI_SLAVE_IDX);
102  // change the default configuration
103  imu_aspirin2.mpu.config.smplrt_div = ASPIRIN_2_SMPLRT_DIV;
104  imu_aspirin2.mpu.config.dlpf_cfg = ASPIRIN_2_LOWPASS_FILTER;
105  imu_aspirin2.mpu.config.gyro_range = ASPIRIN_2_GYRO_RANGE;
106  imu_aspirin2.mpu.config.accel_range = ASPIRIN_2_ACCEL_RANGE;
107 
108  /* read 15 bytes for status, accel, gyro + 6 bytes for mag slave */
109  imu_aspirin2.mpu.config.nb_bytes = 21;
110 
111  /* use mag if not disabled */
112 #if !ASPIRIN_2_DISABLE_MAG
113  /* HMC5883 magnetometer as I2C slave */
114  imu_aspirin2.mpu.config.nb_slaves = 1;
115 
116  /* set function to configure mag */
117  imu_aspirin2.mpu.config.slaves[0].configure = &imu_aspirin2_configure_mag_slave;
118 
119  /* Set MPU I2C master clock */
120  imu_aspirin2.mpu.config.i2c_mst_clk = MPU60X0_MST_CLK_400KHZ;
121  /* Enable I2C slave0 delayed sample rate */
122  imu_aspirin2.mpu.config.i2c_mst_delay = 1;
123 
124 
125  /* configure spi transaction for wait_slave4 */
126  imu_aspirin2.wait_slave4_trans.cpol = SPICpolIdleHigh;
127  imu_aspirin2.wait_slave4_trans.cpha = SPICphaEdge2;
128  imu_aspirin2.wait_slave4_trans.dss = SPIDss8bit;
129  imu_aspirin2.wait_slave4_trans.bitorder = SPIMSBFirst;
130  imu_aspirin2.wait_slave4_trans.cdiv = SPIDiv64;
131 
132  imu_aspirin2.wait_slave4_trans.select = SPISelectUnselect;

154 void imu_aspirin2_event(void)
155 {
156  uint32_t now_ts = get_sys_time_usec();
157 
158  mpu60x0_spi_event(&imu_aspirin2.mpu);
159  if (imu_aspirin2.mpu.data_available) {
160 #if !ASPIRIN_2_DISABLE_MAG
161  /* HMC5883 has xzy order of axes in returned data */
162  struct Int32Vect3 mag;
163  mag.x = Int16FromBuf(imu_aspirin2.mpu.data_ext, 0);
164  mag.z = Int16FromBuf(imu_aspirin2.mpu.data_ext, 2);
165  mag.y = Int16FromBuf(imu_aspirin2.mpu.data_ext, 4);
166 #endif
167 
168  /* Handle axis assignement for Lisa/S integrated Aspirin like IMU. */
169 #ifdef LISA_S
170 #ifdef LISA_S_UPSIDE_DOWN
171  RATES_ASSIGN(imu.gyro_unscaled,
172  imu_aspirin2.mpu.data_rates.rates.p,
173  -imu_aspirin2.mpu.data_rates.rates.q,
174  -imu_aspirin2.mpu.data_rates.rates.r);
175  VECT3_ASSIGN(imu.accel_unscaled,
176  imu_aspirin2.mpu.data_accel.vect.x,
177  -imu_aspirin2.mpu.data_accel.vect.y,
178  -imu_aspirin2.mpu.data_accel.vect.z);
179 #if !ASPIRIN_2_DISABLE_MAG
180  VECT3_ASSIGN(imu.mag_unscaled, mag.x, -mag.y, -mag.z);
181 #endif
182 #else
183  RATES_COPY(imu.gyro_unscaled, imu_aspirin2.mpu.data_rates.rates);
184  VECT3_COPY(imu.accel_unscaled, imu_aspirin2.mpu.data_accel.vect);
185 #if !ASPIRIN_2_DISABLE_MAG
186  VECT3_COPY(imu.mag_unscaled, mag);
187 #endif
188 #endif
189 #else
190 
191  /* Handle axis assignement for Lisa/M or Lisa/MX V2.1 integrated Aspirin like
192  * IMU.
193  */
194 #ifdef LISA_M_OR_MX_21
195  RATES_ASSIGN(imu.gyro_unscaled,
196  -imu_aspirin2.mpu.data_rates.rates.q,
197  imu_aspirin2.mpu.data_rates.rates.p,
198  imu_aspirin2.mpu.data_rates.rates.r);
199  VECT3_ASSIGN(imu.accel_unscaled,
200  -imu_aspirin2.mpu.data_accel.vect.y,
201  imu_aspirin2.mpu.data_accel.vect.x,
202  imu_aspirin2.mpu.data_accel.vect.z);
203 #if !ASPIRIN_2_DISABLE_MAG
204  VECT3_ASSIGN(imu.mag_unscaled, -mag.y, mag.x, mag.z);
205 #endif
206 #else
207 
208  /* Handle real Aspirin IMU axis assignement. */
209 #ifdef LISA_M_LONGITUDINAL_X
210  RATES_ASSIGN(imu.gyro_unscaled,
211  imu_aspirin2.mpu.data_rates.rates.q,
212  -imu_aspirin2.mpu.data_rates.rates.p,
213  imu_aspirin2.mpu.data_rates.rates.r);
214  VECT3_ASSIGN(imu.accel_unscaled,
215  imu_aspirin2.mpu.data_accel.vect.y,
216  -imu_aspirin2.mpu.data_accel.vect.x,
217  imu_aspirin2.mpu.data_accel.vect.z);
218 #if !ASPIRIN_2_DISABLE_MAG
219  VECT3_ASSIGN(imu.mag_unscaled, -mag.x, -mag.y, mag.z);
220 #endif
221 #else
222  RATES_COPY(imu.gyro_unscaled, imu_aspirin2.mpu.data_rates.rates);
223  VECT3_COPY(imu.accel_unscaled, imu_aspirin2.mpu.data_accel.vect);
224 #if !ASPIRIN_2_DISABLE_MAG
225  VECT3_ASSIGN(imu.mag_unscaled, mag.y, -mag.x, mag.z)
226 #endif
227 #endif
228 #endif
229 #endif
230 
231  imu_aspirin2.mpu.data_available = false;
232 
233  imu_scale_gyro(&imu);
234  imu_scale_accel(&imu);
235  AbiSendMsgIMU_GYRO_INT32(IMU_ASPIRIN2_ID, now_ts, &imu.gyro);
236  AbiSendMsgIMU_ACCEL_INT32(IMU_ASPIRIN2_ID, now_ts, &imu.accel);
237 #if !ASPIRIN_2_DISABLE_MAG
238  imu_scale_mag(&imu);
239  AbiSendMsgIMU_MAG_INT32(IMU_ASPIRIN2_ID, now_ts, &imu.mag);
240 #endif
241  }
242 }

254 bool imu_aspirin2_configure_mag_slave(Mpu60x0ConfigSet mpu_set, void *mpu)
255 {
256  // wait before starting the configuration of the HMC58xx mag
257  // doing to early may void the mode configuration
258  if (get_sys_time_float() < ASPIRIN_2_MAG_STARTUP_DELAY) {
259  return false;
260  }
261 
262  mpu_set_and_wait(mpu_set, mpu, MPU60X0_REG_I2C_SLV4_ADDR, (HMC58XX_ADDR >> 1));
263  mpu_set_and_wait(mpu_set, mpu, MPU60X0_REG_I2C_SLV4_REG, HMC58XX_REG_CFGA);
264  mpu_set_and_wait(mpu_set, mpu, MPU60X0_REG_I2C_SLV4_DO, HMC58XX_CRA);
265  mpu_set_and_wait(mpu_set, mpu, MPU60X0_REG_I2C_SLV4_CTRL, (1 << 7)); // Slave 4 enable
266 
267  mpu_wait_slave4_ready();
268 
269  mpu_set_and_wait(mpu_set, mpu, MPU60X0_REG_I2C_SLV4_ADDR, (HMC58XX_ADDR >> 1));
270  mpu_set_and_wait(mpu_set, mpu, MPU60X0_REG_I2C_SLV4_REG, HMC58XX_REG_CFGB);
271  mpu_set_and_wait(mpu_set, mpu, MPU60X0_REG_I2C_SLV4_DO, HMC58XX_CRB);
272  mpu_set_and_wait(mpu_set, mpu, MPU60X0_REG_I2C_SLV4_CTRL, (1 << 7)); // Slave 4 enable
273 
274  mpu_wait_slave4_ready();
275 
276  mpu_set_and_wait(mpu_set, mpu, MPU60X0_REG_I2C_SLV4_ADDR, (HMC58XX_ADDR >> 1));
277  mpu_set_and_wait(mpu_set, mpu, MPU60X0_REG_I2C_SLV4_REG, HMC58XX_REG_MODE);
278  mpu_set_and_wait(mpu_set, mpu, MPU60X0_REG_I2C_SLV4_DO, HMC58XX_MD);
279  mpu_set_and_wait(mpu_set, mpu, MPU60X0_REG_I2C_SLV4_CTRL, (1 << 7)); // Slave 4 enable
280 
281  mpu_set_and_wait(mpu_set, mpu, MPU60X0_REG_I2C_SLV0_ADDR, (HMC58XX_ADDR >> 1) | MPU60X0_SPI_READ);
282  mpu_set_and_wait(mpu_set, mpu, MPU60X0_REG_I2C_SLV0_REG, HMC58XX_REG_DATXM);
283  // Put the enable command as last.
284  mpu_set_and_wait(mpu_set, mpu, MPU60X0_REG_I2C_SLV0_CTRL,
285  (1 << 7) | // Slave 0 enable
286  (6 << 0)); // Read 6 bytes
287 
288  return true;
289 }

291 void mpu_wait_slave4_ready(void)
292 {
293  while (!imu_aspirin2.slave4_ready) {
294  if (imu_aspirin2.wait_slave4_trans.status == SPITransDone) {
295  imu_aspirin2.wait_slave4_tx_buf[0] = MPU60X0_REG_I2C_MST_STATUS | MPU60X0_SPI_READ;
296  spi_submit(imu_aspirin2.mpu.spi_p, &(imu_aspirin2.wait_slave4_trans));
297  }
298  }
299 }
