126 static void sys_tick_handler(void)
127 {
128  chMtxLock(&sys_time_mtx);
129  /* current time in sys_ticks */
130  sys_time.nb_tick = chVTGetSystemTime();
131  /* max time is 2^32 / CH_CFG_ST_FREQUENCY, i.e. around 10 days at 10kHz */
132  uint32_t sec = sys_time.nb_tick / CH_CFG_ST_FREQUENCY;
133 #ifdef SYS_TIME_LED
134  if (sec > sys_time.nb_sec) {
135  LED_TOGGLE(SYS_TIME_LED);
136  }
137 #endif
138  sys_time.nb_sec = sec;
139  sys_time.nb_sec_rem = sys_time.nb_tick - sys_time_ticks_of_sec(sys_time.nb_sec);
140 
141  /* advance virtual timers */
142  for (unsigned int i = 0; i < SYS_TIME_NB_TIMER; i++) {
143  if (sys_time.timer[i].in_use &&
144  sys_time.nb_tick >= sys_time.timer[i].end_time) {
145  sys_time.timer[i].end_time += sys_time.timer[i].duration;
146  sys_time.timer[i].elapsed = TRUE;
147  /* call registered callbacks, WARNING: they will be executed in the sys_time thread! */
148  if (sys_time.timer[i].cb) {
149  sys_time.timer[i].cb(i);
150  }
151  }
152  }
153  chMtxUnlock(&sys_time_mtx);
154 }
