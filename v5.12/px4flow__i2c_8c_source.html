65 static inline void px4flow_i2c_frame_cb(void)
66 {
67  static float quality = 0;
68  static float noise = 0;
69  quality = ((float)px4flow.i2c_frame.qual) / 255.0;
70  noise = px4flow.stddev + (1 - quality) * px4flow.stddev * 10;
71  noise = noise * noise; // square the noise to get variance of the measurement
72 
73  static float timestamp = 0;
74  static uint32_t time_usec = 0;
75 
76  static float flow_comp_m_x = 0.0;
77  static float flow_comp_m_y = 0.0;
78 
79  if (quality > PX4FLOW_QUALITY_THRESHOLD) {
80  timestamp = get_sys_time_float();
81  time_usec = (uint32_t)(timestamp * 1e6);
82 
83  flow_comp_m_x = ((float)px4flow.i2c_frame.flow_comp_m_x) / 1000.0;
84  flow_comp_m_y = ((float)px4flow.i2c_frame.flow_comp_m_y) / 1000.0;
85 
86  // flip the axis (if the PX4FLOW is mounted as shown in
87  // https://pixhawk.org/modules/px4flow
88  AbiSendMsgVELOCITY_ESTIMATE(PX4FLOW_VELOCITY_ID,
89  time_usec,
90  flow_comp_m_y,
91  flow_comp_m_x,
92  0.0f,
93  noise);
94  }
95 
96  // distance is always positive - use median filter to remove outliers
97  static int32_t ground_distance = 0;
98  static float ground_distance_float = 0.0;
99 
100  // update filter
101  ground_distance = update_median_filter_i(&sonar_filter, (int32_t)px4flow.i2c_frame.ground_distance);
102  ground_distance_float = ((float)ground_distance) / 1000.0;
103 
104  // compensate AGL measurement for body rotation
105  if (px4flow.compensate_rotation) {
106  float phi = stateGetNedToBodyEulers_f()->phi;
107  float theta = stateGetNedToBodyEulers_f()->theta;
108  float gain = (float)fabs( (double) (cosf(phi) * cosf(theta)));
109  ground_distance_float = ground_distance_float / gain;
110  }
111 
112  if (px4flow.update_agl) {
113  AbiSendMsgAGL(AGL_SONAR_PX4FLOW_ID, ground_distance_float);
114  }
115 }

152 void px4flow_i2c_periodic(void)
153 {
154  switch (px4flow.status) {
155  case PX4FLOW_FRAME_REQ:
156  // ask for i2c frame
157  px4flow.trans.buf[0] = PX4FLOW_I2C_FRAME;
158  if (i2c_transceive(&PX4FLOW_I2C_DEV, &px4flow.trans, px4flow.addr, 1, PX4FLOW_I2C_FRAME_LENGTH)) {
159  // transaction OK, increment status
160  px4flow.status = PX4FLOW_FRAME_REC;
161  }
162  break;
163  case PX4FLOW_FRAME_REC:
164  // check if the transaction was successful
165  if (px4flow.trans.status == I2CTransSuccess) {
166  // retrieve data
167  uint8_t idx = 0;
168  px4flow.i2c_frame.frame_count = (px4flow.trans.buf[idx + 1] << 8 | px4flow.trans.buf[idx]);
169  idx += sizeof(uint16_t);
170  px4flow.i2c_frame.pixel_flow_x_sum = (px4flow.trans.buf[idx + 1] << 8 | px4flow.trans.buf[idx]);
171  idx += sizeof(int16_t);
172  px4flow.i2c_frame.pixel_flow_y_sum = (px4flow.trans.buf[idx + 1] << 8 | px4flow.trans.buf[idx]);
173  idx += sizeof(int16_t);
174  px4flow.i2c_frame.flow_comp_m_x = (px4flow.trans.buf[idx + 1] << 8 | px4flow.trans.buf[idx]);
175  idx += sizeof(int16_t);
176  px4flow.i2c_frame.flow_comp_m_y = (px4flow.trans.buf[idx + 1] << 8 | px4flow.trans.buf[idx]);
177  idx += sizeof(int16_t);
178  px4flow.i2c_frame.qual = (px4flow.trans.buf[idx + 1] << 8 | px4flow.trans.buf[idx]);
179  idx += sizeof(int16_t);
180  px4flow.i2c_frame.gyro_x_rate = (px4flow.trans.buf[idx + 1] << 8 | px4flow.trans.buf[idx]);
181  idx += sizeof(int16_t);
182  px4flow.i2c_frame.gyro_y_rate = (px4flow.trans.buf[idx + 1] << 8 | px4flow.trans.buf[idx]);
183  idx += sizeof(int16_t);
184  px4flow.i2c_frame.gyro_z_rate = (px4flow.trans.buf[idx + 1] << 8 | px4flow.trans.buf[idx]);
185  idx += sizeof(int16_t);
186  px4flow.i2c_frame.gyro_range = px4flow.trans.buf[idx];
187  idx += sizeof(uint8_t);
188  px4flow.i2c_frame.sonar_timestamp = px4flow.trans.buf[idx];
189  idx += sizeof(uint8_t);
190  px4flow.i2c_frame.ground_distance = (px4flow.trans.buf[idx + 1] << 8 | px4flow.trans.buf[idx]);
191 
192  // propagate measurements
193  px4flow_i2c_frame_cb();
194  }
195  // increment status
196  // ask for regular frame again
197  px4flow.status = PX4FLOW_FRAME_REQ;
198  break;
199  case PX4FLOW_INT_FRAME_REQ:
200  // Send the command to begin a measurement.
201  px4flow.trans.buf[0] = PX4FLOW_I2C_INTEGRAL_FRAME;
202  if (i2c_transmit(&PX4FLOW_I2C_DEV, &px4flow.trans, px4flow.addr, 1)) {
203  // transaction OK, increment status
204  px4flow.status = PX4FLOW_INT_FRAME_REC;
205  }
206  break;
207  case PX4FLOW_INT_FRAME_REC:
208  // ask for integral frame
209  px4flow.trans.buf[0] = PX4FLOW_I2C_INTEGRAL_FRAME;
210  if (i2c_transceive(&PX4FLOW_I2C_DEV, &px4flow.trans, px4flow.addr, 1, PX4FLOW_I2C_INTEGRAL_FRAME_LENGTH)) {
211  // transaction OK, increment status
212  px4flow.status = PX4FLOW_INT_FRAME_REC_OK;
213  }
214  break;
215  case PX4FLOW_INT_FRAME_REC_OK:
216  // check if the transaction was successful
217  if (px4flow.trans.status == I2CTransSuccess) {
218  // retrieve data
219  uint8_t idx = 0;
220  px4flow.i2c_int_frame.frame_count_since_last_readout = (px4flow.trans.buf[idx + 1] << 8 | px4flow.trans.buf[idx]);
221  idx += sizeof(uint16_t);
222  px4flow.i2c_int_frame.pixel_flow_x_integral = (px4flow.trans.buf[idx + 1] << 8 | px4flow.trans.buf[idx]);
223  idx += sizeof(int16_t);
224  px4flow.i2c_int_frame.pixel_flow_y_integral = (px4flow.trans.buf[idx + 1] << 8 | px4flow.trans.buf[idx]);
225  idx += sizeof(uint16_t);
226  px4flow.i2c_int_frame.gyro_x_rate_integral = (px4flow.trans.buf[idx + 1] << 8 | px4flow.trans.buf[idx]);
227  idx += sizeof(int16_t);
228  px4flow.i2c_int_frame.gyro_y_rate_integral = (px4flow.trans.buf[idx + 1] << 8 | px4flow.trans.buf[idx]);
229  idx += sizeof(int16_t);
230  px4flow.i2c_int_frame.gyro_z_rate_integral = (px4flow.trans.buf[idx + 1] << 8 | px4flow.trans.buf[idx]);
231  idx += sizeof(int16_t);
232  px4flow.i2c_int_frame.integration_timespan = (px4flow.trans.buf[idx + 3] << 24 | px4flow.trans.buf[idx + 2] << 16
233  | px4flow.trans.buf[idx + 1] << 8 | px4flow.trans.buf[idx]);
234  idx += sizeof(uint32_t);
235  px4flow.i2c_int_frame.sonar_timestamp = (px4flow.trans.buf[idx + 3] << 24 | px4flow.trans.buf[idx + 2] << 16
236  | px4flow.trans.buf[idx + 1] << 8 | px4flow.trans.buf[idx]);
237  idx += sizeof(uint32_t);
238  px4flow.i2c_int_frame.ground_distance = (px4flow.trans.buf[idx + 1] << 8 | px4flow.trans.buf[idx]);
239  idx += sizeof(int16_t);
240  px4flow.i2c_int_frame.gyro_temperature = (px4flow.trans.buf[idx + 1] << 8 | px4flow.trans.buf[idx]);
241  idx += sizeof(int16_t);
242  px4flow.i2c_int_frame.qual = px4flow.trans.buf[idx];
243 
244  // propagate measurements
245  px4flow_i2c_int_frame_cb();
246  }
247  // increment status
248  px4flow.status = PX4FLOW_INT_FRAME_REQ;
249  break;
250  default:
251  break;
252  }
