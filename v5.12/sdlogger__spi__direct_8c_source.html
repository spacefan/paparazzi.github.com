110 void sdlogger_spi_direct_periodic(void)
111 {
112  sdcard_spi_periodic(&sdcard1);
113 
114 #if SDLOGGER_ON_ARM
115  if(autopilot_get_motors_on()) {
116  sdlogger_spi.do_log = 1;
117  } else {
118  sdlogger_spi.do_log = 0;
119  }
120 #endif
121 
122  switch (sdlogger_spi.status) {
123  case SDLogger_Initializing:
124  if (sdcard1.status == SDCard_Idle) {
125  sdcard_spi_read_block(&sdcard1, 0x00002000, &sdlogger_spi_direct_index_received);
126  sdlogger_spi.status = SDLogger_RetreivingIndex;
127  }
128  break;
129 
130  case SDLogger_Ready:
131  if ((sdlogger_spi.do_log == 1) &&
132  sdcard1.status == SDCard_Idle) {
133  LOGGER_LED_ON;
134  sdcard_spi_multiwrite_start(&sdcard1, sdlogger_spi.next_available_address);
135  sdlogger_spi.status = SDLogger_Logging;
136  }
137  break;
138 
139  case SDLogger_Logging:
140  /* This line is NOT unit-tested because it is an inline function */
141  #if PERIODIC_TELEMETRY
142  periodic_telemetry_send_Logger(DefaultPeriodic,
143  &pprzlog_tp.trans_tx,
144  &sdlogger_spi.device);
145  #endif
146  /* Check if SD Card buffer is full and SD Card is ready for new data */
147  if (sdlogger_spi.sdcard_buf_idx > 512 &&
148  sdcard1.status == SDCard_MultiWriteIdle) {
149  sdcard_spi_multiwrite_next(&sdcard1, &sdlogger_spi_direct_multiwrite_written);
150  }
151  /* Check if switch is flipped to stop logging */
152  if (sdlogger_spi.do_log == 0) {
153  sdlogger_spi.status = SDLogger_LoggingFinalBlock;
154  }
155  break;
156 
157  case SDLogger_LoggingFinalBlock:
158  if (sdcard1.status == SDCard_MultiWriteIdle) {
159  if (sdlogger_spi.sdcard_buf_idx > 512) {
160  sdcard_spi_multiwrite_next(&sdcard1, &sdlogger_spi_direct_multiwrite_written);
161  }
162  else if (sdlogger_spi.sdcard_buf_idx > 1) {
163  /* Fill with trailing zero's */
164  for (uint16_t i = sdlogger_spi.sdcard_buf_idx; i < (SD_BLOCK_SIZE+1); i++) {
165  sdcard1.output_buf[i] = 0x00;
166  }
167  sdcard_spi_multiwrite_next(&sdcard1, &sdlogger_spi_direct_multiwrite_written);
168  }
169  else if (sdlogger_spi.sdcard_buf_idx == 1) {
170  sdcard_spi_multiwrite_stop(&sdcard1);
171  sdlogger_spi.status = SDLogger_StoppedLogging;
172  }
173  }
174  break;
175 
176  case SDLogger_StoppedLogging:
177  if (sdcard1.status == SDCard_Idle) {
178  sdcard_spi_read_block(&sdcard1, 0x00002000, &sdlogger_spi_direct_index_received);
179  sdlogger_spi.status = SDLogger_GettingIndexForUpdate;
180  }
181  break;
182 
183  case SDLogger_UpdatingIndex:
184  if (sdcard1.status == SDCard_Idle) {
185  LOGGER_LED_OFF;
186  sdlogger_spi.status = SDLogger_Ready;
187  }
188  break;
189 
190  case SDLogger_Downloading:
191  if (sdcard1.status == SDCard_Idle) {
192  /* Put bytes to the buffer until all is written or buffer is full */
193  for (uint16_t i = sdlogger_spi.sdcard_buf_idx; i < SD_BLOCK_SIZE; i++) {
194  long fd = 0;
195  if (uart_check_free_space(&(DOWNLINK_DEVICE), &fd, 1)) {
196  uart_put_byte(&(DOWNLINK_DEVICE), fd, sdcard1.input_buf[i]);
197  }
198  else {
199  /* No free space left, abort for-loop */
200  break;
201  }
202  sdlogger_spi.sdcard_buf_idx++;
203  }
204  /* Request next block if entire buffer was written to uart */
205  if (sdlogger_spi.sdcard_buf_idx >= SD_BLOCK_SIZE) {
206  if (sdlogger_spi.download_length > 0) {
207  sdcard_spi_read_block(&sdcard1, sdlogger_spi.download_address, NULL);
208  sdlogger_spi.download_address++;
209  sdlogger_spi.download_length--;
210  }
211  else {
212  LOGGER_LED_OFF;
213  sdlogger_spi.status = SDLogger_Ready;
214  }
215  sdlogger_spi.sdcard_buf_idx = 0;
216  }
217  }
218  break;
219 
220  default:
221  break;
222  }
223 }

232 void sdlogger_spi_direct_index_received(void)
233 {
234 
235  switch (sdlogger_spi.status) {
236  case SDLogger_RetreivingIndex:
237  sdlogger_spi.next_available_address = 0x00004000;
238  sdlogger_spi.last_completed = 0;
239  // Save data for later use
240  sdlogger_spi.next_available_address = (sdcard1.input_buf[0] << 24) |
241  (sdcard1.input_buf[1] << 16) |
242  (sdcard1.input_buf[2] << 8) |
243  (sdcard1.input_buf[3]);
244  sdlogger_spi.last_completed = sdcard1.input_buf[4];
245 
246  if(sdlogger_spi.next_available_address < 0x00004000) {
247  sdlogger_spi.next_available_address = 0x00004000;
248  }
249 
250  /* Ready to start logging */
251  sdlogger_spi.status = SDLogger_Ready;
252  break;
253 
254  case SDLogger_GettingIndexForUpdate:
255  /* Copy input buffer to output buffer */
256  for (uint16_t i = 0; i < SD_BLOCK_SIZE; i++) {
257  sdcard1.output_buf[i+6] = sdcard1.input_buf[i];
258  }
259 
260  /* Increment last completed log */
261  sdcard1.output_buf[4+6] = ++sdlogger_spi.last_completed;
262  /* Write log info at dedicated location */
263  {
264  uint16_t log_idx_start = 5 + 6 + (sdlogger_spi.last_completed - 1) * 12;
265 
266  /* Set start address and length at location that belongs to the log nr */
267  sdcard1.output_buf[log_idx_start+0] = sdlogger_spi.next_available_address >> 24;
268  sdcard1.output_buf[log_idx_start+1] = sdlogger_spi.next_available_address >> 16;
269  sdcard1.output_buf[log_idx_start+2] = sdlogger_spi.next_available_address >> 8;
270  sdcard1.output_buf[log_idx_start+3] = sdlogger_spi.next_available_address >> 0;
271  sdcard1.output_buf[log_idx_start+4] = sdlogger_spi.log_len >> 24;
272  sdcard1.output_buf[log_idx_start+5] = sdlogger_spi.log_len >> 16;
273  sdcard1.output_buf[log_idx_start+6] = sdlogger_spi.log_len >> 8;
274  sdcard1.output_buf[log_idx_start+7] = sdlogger_spi.log_len >> 0;
275  }
276 
277  /* Increment and update the next available address */
278  sdlogger_spi.next_available_address += sdlogger_spi.log_len;
279  sdcard1.output_buf[0+6] = sdlogger_spi.next_available_address >> 24;
280  sdcard1.output_buf[1+6] = sdlogger_spi.next_available_address >> 16;
281  sdcard1.output_buf[2+6] = sdlogger_spi.next_available_address >> 8;
282  sdcard1.output_buf[3+6] = sdlogger_spi.next_available_address >> 0;
283 
284  sdcard_spi_write_block(&sdcard1, 0x00002000);
285  /* Reset log length */
286  sdlogger_spi.log_len = 0;
287  sdlogger_spi.status = SDLogger_UpdatingIndex;
288  break;
289 
290  case SDLogger_GettingIndexForDownload:
291  {
292  uint16_t info_idx = 5 + (sdlogger_spi.download_id - 1) * 12;
293  sdlogger_spi.download_address = (sdcard1.input_buf[info_idx+0] << 24) |
294  (sdcard1.input_buf[info_idx+1] << 16) |
295  (sdcard1.input_buf[info_idx+2] << 8) |
296  (sdcard1.input_buf[info_idx+3] << 0);
297  sdlogger_spi.download_length = (sdcard1.input_buf[info_idx+4] << 24) |
298  (sdcard1.input_buf[info_idx+5] << 16) |
299  (sdcard1.input_buf[info_idx+6] << 8) |
300  (sdcard1.input_buf[info_idx+7] << 0);
301  if (sdlogger_spi.download_length > 0) {
302  /* Request the first block */
303  sdcard_spi_read_block(&sdcard1, sdlogger_spi.download_address, NULL);
304  /* After each read block, incr address, decr length */
305  sdlogger_spi.download_address++;
306  sdlogger_spi.download_length--;
307  sdlogger_spi.status = SDLogger_Downloading;
308  }
309  else {
310  LOGGER_LED_OFF;
311  sdlogger_spi.status = SDLogger_Ready;
312  }
313  sdlogger_spi.sdcard_buf_idx = 0;
314  }
315  break;
316 
317  default:
318  break;
319  }

328 void sdlogger_spi_direct_multiwrite_written(void)
329 {
330  /* Increment log length */

332 
333  /* Copy data from logger buffer to SD Card buffer */

339  /* And reset the logger buffer index */
340  sdlogger_spi.idx = 0;

343 void sdlogger_spi_direct_command(void)
344 {
345  if (sdcard1.status == SDCard_Idle && sdlogger_spi.command > 0 &&
346  sdlogger_spi.command < 43) {
347  LOGGER_LED_ON;
348  sdcard_spi_read_block(&sdcard1, 0x00002000,
349  &sdlogger_spi_direct_index_received);
350  sdlogger_spi.download_id = sdlogger_spi.command;
351  sdlogger_spi.status = SDLogger_GettingIndexForDownload;
352  }
353  else if (sdcard1.status == SDCard_Idle && sdlogger_spi.command == 255) {
354  telemetry_mode_Main = TELEMETRY_MODE_Main_empty;

356  sdcard_spi_read_block(&sdcard1, 0x00002000, NULL);
357  sdlogger_spi.download_length = 0;
358  sdlogger_spi.sdcard_buf_idx = 0;
359  sdlogger_spi.status = SDLogger_Downloading;
360  }
361  /* Always reset command value back to zero */
362  sdlogger_spi.command = 0;

376 void sdlogger_spi_direct_put_byte(struct sdlogger_spi_periph *p, long fd __attribute__((unused)), uint8_t data)
377 {
378  /* SD Buffer full, write in logger buffer */
379  if (p->sdcard_buf_idx > 512) {
380  if (p->idx < SDLOGGER_BUFFER_SIZE) {
381  p->buffer[p->idx++] = data;
382  }
383  /* else: data lost */
384  }
385  /* Writing directly to SD Card buffer */
386  else {
387  sdcard1.output_buf[p->sdcard_buf_idx++] = data;
388 
389  /* Flush buffer */
390  if (p->sdcard_buf_idx > 512 && sdcard1.status == SDCard_MultiWriteIdle) {
391  sdcard_spi_multiwrite_next(&sdcard1, &sdlogger_spi_direct_multiwrite_written);
392  }
393  }
394 }
