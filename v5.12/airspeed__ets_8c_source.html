158 void airspeed_ets_read_event(void)
159 {
160  int n;
161  float airspeed_tmp = 0.0;
162 
163  // Get raw airspeed from buffer
164  airspeed_ets_raw = ((uint16_t)(airspeed_ets_i2c_trans.buf[1]) << 8) | (uint16_t)(airspeed_ets_i2c_trans.buf[0]);
165  // Check if this is valid airspeed
166  if (airspeed_ets_raw == 0) {
167  airspeed_ets_valid = false;
168  } else {
169  airspeed_ets_valid = true;
170  }
171 
172  // Continue only if a new airspeed value was received
173  if (airspeed_ets_valid) {
174 #if !AIRSPEED_ETS_3RD_PARTY_MODE
175  // Calculate offset average if not done already
176  if (!airspeed_ets_offset_init) {
177  --airspeed_ets_cnt;
178  // Check if averaging completed
179  if (airspeed_ets_cnt == 0) {
180  // Calculate average
181  airspeed_ets_offset = (uint16_t)(airspeed_ets_offset_tmp / AIRSPEED_ETS_OFFSET_NBSAMPLES_AVRG);
182  // Limit offset
183  if (airspeed_ets_offset < AIRSPEED_ETS_OFFSET_MIN) {
184  airspeed_ets_offset = AIRSPEED_ETS_OFFSET_MIN;
185  }
186  if (airspeed_ets_offset > AIRSPEED_ETS_OFFSET_MAX) {
187  airspeed_ets_offset = AIRSPEED_ETS_OFFSET_MAX;
188  }
189  airspeed_ets_offset_init = true;
190  }
191  // Check if averaging needs to continue
192  else if (airspeed_ets_cnt <= AIRSPEED_ETS_OFFSET_NBSAMPLES_AVRG) {
193  airspeed_ets_offset_tmp += airspeed_ets_raw;
194  }
195  }
196  // Convert raw to m/s
197 #ifdef AIRSPEED_ETS_REVERSE
198  if (airspeed_ets_offset_init && airspeed_ets_raw < airspeed_ets_offset) {
199  airspeed_tmp = AIRSPEED_ETS_SCALE * sqrtf((float)(airspeed_ets_offset - airspeed_ets_raw)) - AIRSPEED_ETS_OFFSET;
200  }
201 #else
202  if (airspeed_ets_offset_init && airspeed_ets_raw > airspeed_ets_offset) {
203  airspeed_tmp = AIRSPEED_ETS_SCALE * sqrtf((float)(airspeed_ets_raw - airspeed_ets_offset)) - AIRSPEED_ETS_OFFSET;
204  }
205 #endif
206  else {
207  airspeed_tmp = 0.0;
208  }
209 //use raw value for sensor set to third-party mode
210 #else
211  airspeed_tmp = airspeed_ets_raw;
212 #endif //AIRSPEED_ETS_3RD_PARTY_MODE
213 
214  // Airspeed should always be positive
215  if (airspeed_tmp < 0.0) {
216  airspeed_tmp = 0.0;
217  }
218  // Moving average
219  airspeed_ets_buffer[airspeed_ets_buffer_idx++] = airspeed_tmp;
220  if (airspeed_ets_buffer_idx >= AIRSPEED_ETS_NBSAMPLES_AVRG) {
221  airspeed_ets_buffer_idx = 0;
222  }
223  airspeed_ets = 0.0;
224  for (n = 0; n < AIRSPEED_ETS_NBSAMPLES_AVRG; ++n) {
225  airspeed_ets += airspeed_ets_buffer[n];
226  }
227  airspeed_ets = airspeed_ets / (float)AIRSPEED_ETS_NBSAMPLES_AVRG;
228 #if USE_AIRSPEED_ETS
229  stateSetAirspeed_f(airspeed_ets);
230 #endif
231 #if AIRSPEED_ETS_SYNC_SEND
232  DOWNLINK_SEND_AIRSPEED_ETS(DefaultChannel, DefaultDevice, &airspeed_ets_raw, &airspeed_ets_offset, &airspeed_ets);
233 #endif
234  } else {
235  airspeed_ets = 0.0;
236  }
237

239 #if AIRSPEED_ETS_SDLOG
240 #ifndef SITL
241  if (pprzLogFile != -1) {
242  if (!log_airspeed_ets_started) {
243  sdLogWriteLog(pprzLogFile, "AIRSPEED_ETS: raw offset airspeed(m/s) GPS_fix TOW(ms) Week Lat(1e7deg) Lon(1e7deg) HMSL(mm) gpseed(cm/s) course(1e7rad) climb(cm/s)\n");

247  airspeed_ets_raw, airspeed_ets_offset, airspeed_ets,
248  gps.fix, gps.tow, gps.week,
