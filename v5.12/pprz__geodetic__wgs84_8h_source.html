81 static inline int32_t wgs84_ellipsoid_to_geoid_i(int32_t lat, int32_t lon)
82 {
83  float x = (180.0f + (float)lon / 1e7) / 10.0f;
84  Bound(x, 0.0f, 35.99999f);
85  float y = (90.0f - (float)lat / 1e7) / 10.0f;
86  Bound(y, 0.0f, 17.99999f);
87  uint8_t ex1 = (uint8_t) x;
88  uint8_t ex2 = ex1 + 1;
89  if (ex2 >= 36) { ex2 = 0; }
90  uint8_t ey1 = (uint8_t) y;
91  uint8_t ey2 = ey1 + 1;
92  float lin_x = x - ((float) ex1);
93  float lin_y = y - ((float) ey1);
94  float h11 = (1.0f - lin_x) * (1.0f - lin_y) * WGS84_H(ex1, ey1);
95  float h12 = lin_x * (1.0f - lin_y) * WGS84_H(ex2, ey1);
96  float h21 = (1.0f - lin_x) * lin_y * WGS84_H(ex1, ey2);
97  float h22 = lin_x * lin_y * WGS84_H(ex2, ey2);
98  return (uint32_t)((h11 + h12 + h21 + h22) * 1000.);
99 }
