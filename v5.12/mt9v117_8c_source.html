196 static void write_reg(struct mt9v117_t *mt, uint16_t addr, uint32_t val, uint16_t len)
197 {
198  mt->i2c_trans.buf[0] = addr >> 8;
199  mt->i2c_trans.buf[1] = addr & 0xFF;
200 
201  // Fix sigdness based on length
202  if(len == 1) {
203  mt->i2c_trans.buf[2] = val & 0xFF;
204  }
205  else if(len == 2) {
206  mt->i2c_trans.buf[2] = (val >> 8) & 0xFF;
207  mt->i2c_trans.buf[3] = val & 0xFF;
208  }
209  else if(len == 4) {
210  mt->i2c_trans.buf[2] = (val >> 24) & 0xFF;
211  mt->i2c_trans.buf[3] = (val >> 16) & 0xFF;
212  mt->i2c_trans.buf[4] = (val >> 8) & 0xFF;
213  mt->i2c_trans.buf[5] = val & 0xFF;
214  }
215  else {
216  printf("[MT9V117] write_reg with incorrect length %d\r\n", len);
217  }
218 
219  // Transmit the buffer
220  i2c_transmit(mt->i2c_periph, &mt->i2c_trans, MT9V117_ADDRESS, len + 2);
221 }

226 static uint32_t read_reg(struct mt9v117_t *mt, uint16_t addr, uint16_t len)
227 {
228  uint32_t ret = 0;
229  mt->i2c_trans.buf[0] = addr >> 8;
230  mt->i2c_trans.buf[1] = addr & 0xFF;
231 
232  // Transmit the buffer and receive back
233  i2c_transceive(mt->i2c_periph, &mt->i2c_trans, MT9V117_ADDRESS, 2, len);
234 
235  /* Fix sigdness */
236  for(uint8_t i =0; i < len; i++) {
237  ret |= mt->i2c_trans.buf[len-i-1] << (8*i);
238  }
239  return ret;
240 }

275  write_var(mt, MT9V117_LOW_LIGHT_VAR, 0x003a, 0x007f, 2);
276  write_var(mt, MT9V117_LOW_LIGHT_VAR, 0x003c, 0x007f, 2);
277  write_var(mt, MT9V117_LOW_LIGHT_VAR, 0x0004, 0x00f4, 2);
278 
279  /* Patch 0403; Critical; Sensor optimization */
280  write_reg(mt, MT9V117_ACCESS_CTL_STAT, 0x0001, 2);
281  write_reg(mt, MT9V117_PHYSICAL_ADDRESS_ACCESS, 0x7000, 2);
282 
283  /* Write patch */
284  for (uint8_t i = 0; i < MT9V117_PATCH_LINE_NUM; ++i) {
285  // Copy buffer
286  for(uint8_t j = 0; j < mt9v117_patch_lines[i].len; ++j) {
287  mt->i2c_trans.buf[j] = mt9v117_patch_lines[i].data[j];
288  }
289 
290  // Transmit the buffer
291  i2c_transmit(mt->i2c_periph, &mt->i2c_trans, mt->i2c_trans.slave_addr, mt9v117_patch_lines[i].len);
292  }
293 
294  write_reg(mt, MT9V117_LOGICAL_ADDRESS_ACCESS, 0x0000, 2);
295  write_var(mt, MT9V117_PATCHLDR_VAR, MT9V117_PATCHLDR_LOADER_ADDRESS_OFFSET, 0x05d8, 2);
296  write_var(mt, MT9V117_PATCHLDR_VAR, MT9V117_PATCHLDR_PATCH_ID_OFFSET, 0x0403, 2);
297  write_var(mt, MT9V117_PATCHLDR_VAR, MT9V117_PATCHLDR_FIRMWARE_ID_OFFSET, 0x00430104, 4);
298  write_reg(mt, MT9V117_COMMAND, MT9V117_COMMAND_OK | MT9V117_COMMAND_APPLY_PATCH, 2);
299 
300  /* Wait for command OK */
301  for(uint8_t retries = 100; retries > 0; retries--) {
302  /* Wait 10ms */
303  usleep(10000);
304 
305  /* Check the command */
306  uint16_t cmd = read_reg(mt, MT9V117_COMMAND, 2);
307  if((cmd & MT9V117_COMMAND_APPLY_PATCH) == 0) {
308  if((cmd & MT9V117_COMMAND_OK) == 0) {
309  printf("[MT9V117] Applying patch failed (No OK)\r\n");
310  }
311  return;
312  }
313  }
314 
315  printf("[MT9V117] Applying patch failed after 10 retries\r\n");
316 }

364 {
365  /* bytes written to gpios/pwm */
366  int wc = 0;
367  /* Reset the device */

369  wc += write(gpio129, "0", 1);
370  wc += write(gpio129, "1", 1);

372

375  }

377  /* Start PWM 9 (Which probably is the clock of the MT9V117) */
378  //#define BEBOP_CAMV_PWM_FREQ 43333333
379  int pwm9 = open("/sys/class/pwm/pwm_9/run", O_WRONLY | O_CREAT | O_TRUNC, 0666);
380  wc = 0;
381  wc += write(pwm9, "0", 1);
382  wc += write(pwm9, "1", 1);
383  close(pwm9);
384 
385  if (wc != 2) {

388 
389  //TODO: Make PWM and GPIO generic

391  /* Wait 50ms */
392  usleep(50000);
393 
394  /* Setup i2c transaction */
395  mt->i2c_trans.status = I2CTransDone;

397  /* See if the device is there and correct */
