78 static void decode_optical_flow_msg(struct mavlink_message *msg __attribute__((unused)))
79 {
80  static float quality = 0;
81  static float noise = 0;
82  quality = ((float)optical_flow.quality) / 255.0;
83  noise = px4flow_stddev + (1 - quality) * px4flow_stddev * 10;
84  noise = noise * noise; // square the noise to get variance of the measurement
85 
86  if (quality > PX4FLOW_QUALITY_THRESHOLD) {
87  // flip the axis (if the PX4FLOW is mounted as shown in
88  // https://pixhawk.org/modules/px4flow
89  AbiSendMsgVELOCITY_ESTIMATE(PX4FLOW_VELOCITY_ID,
90  optical_flow.time_usec,
91  optical_flow.flow_comp_m_y,
92  optical_flow.flow_comp_m_x,
93  0.0f,
94  noise);
95  }
96 
97  // compensate AGL measurement for body rotation
98  if (px4flow_compensate_rotation) {
99  float phi = stateGetNedToBodyEulers_f()->phi;
100  float theta = stateGetNedToBodyEulers_f()->theta;
101  float gain = (float)fabs( (double) (cosf(phi) * cosf(theta)));
102  optical_flow.ground_distance = optical_flow.ground_distance / gain;
103  }
104 
105  if (px4flow_update_agl) {
106  // positive distance means it's known/valid
107  if (optical_flow.ground_distance > 0) {
108  AbiSendMsgAGL(AGL_SONAR_PX4FLOW_ID, optical_flow.ground_distance);
109  }
110  }
111 }

132 void px4flow_init(void)
133 {
134  // register mavlink messages
135  req_flow.msg_id = MAVLINK_OPTICAL_FLOW_MSG_ID;
136  req_flow.callback = decode_optical_flow_msg;
137  req_flow.msg.payload = (uint8_t *)(&optical_flow);
138  mavlink_register_msg(&mavlink_tp, &req_flow);
139 
140  req_flow_rad.msg_id = MAVLINK_OPTICAL_FLOW_RAD_MSG_ID;
141  req_flow_rad.callback = decode_optical_flow_rad_msg;
142  req_flow_rad.msg.payload = (uint8_t *)(&optical_flow_rad);
143  mavlink_register_msg(&mavlink_tp, &req_flow_rad);
144 
145  req_heartbeat.msg_id = MAVLINK_OPTICAL_HEARTBEAT_MSG_ID;
146  req_heartbeat.callback = decode_heartbeat_msg;
147  req_heartbeat.msg.payload = (uint8_t *)(&heartbeat);
148  mavlink_register_msg(&mavlink_tp, &req_heartbeat);
