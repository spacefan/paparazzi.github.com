80 void baro_periodic(void)
81 {
82 
83  baro_board.absolute = baro_board.buf.sum / baro_board.buf.av_nb_sample;
84  baro_board.value_filtered = (3 * baro_board.value_filtered + baro_board.absolute) / 4;
85  if (baro_board.status == BB_UNINITIALIZED) {

97  if (baro_board.value_filtered == 0 && baro_board.offset > 15) {
98  baro_board.offset -= 15;
99  } else {
100  baro_board.offset--;

106  } else {
