74 static void write_reg(struct mt9f002_t *mt, uint16_t addr, uint32_t val, uint8_t len)
75 {
76  mt->i2c_trans.buf[0] = addr >> 8;
77  mt->i2c_trans.buf[1] = addr & 0xFF;
78 
79  // Fix sigdness based on length
80  if (len == 1) {
81  mt->i2c_trans.buf[2] = val & 0xFF;
82  } else if (len == 2) {
83  mt->i2c_trans.buf[2] = (val >> 8) & 0xFF;
84  mt->i2c_trans.buf[3] = val & 0xFF;
85  } else if (len == 4) {
86  mt->i2c_trans.buf[2] = (val >> 24) & 0xFF;
87  mt->i2c_trans.buf[3] = (val >> 16) & 0xFF;
88  mt->i2c_trans.buf[4] = (val >> 8) & 0xFF;
89  mt->i2c_trans.buf[5] = val & 0xFF;
90  } else {
91  printf("[MT9F002] write_reg with incorrect length %d\r\n", len);
92  }
93 
94  // Transmit the buffer
95  i2c_transmit(mt->i2c_periph, &mt->i2c_trans, MT9F002_ADDRESS, len + 2);
96 }

101 static uint32_t read_reg(struct mt9f002_t *mt, uint16_t addr, uint8_t len)
102 {
103  uint32_t ret = 0;
104  mt->i2c_trans.buf[0] = addr >> 8;
105  mt->i2c_trans.buf[1] = addr & 0xFF;
106 
107  // Transmit the buffer and receive back
108  i2c_transceive(mt->i2c_periph, &mt->i2c_trans, MT9F002_ADDRESS, 2, len);
109 
110  /* Fix sigdness */
111  for (uint8_t i = 0; i < len; i++) {
112  ret |= mt->i2c_trans.buf[len - i - 1] << (8 * i);
113  }
114  return ret;
115 }

563 static void mt9f002_set_blanking(struct mt9f002_t *mt)
564 {
565  /* Read some config values in order to calculate blanking configuration */
566  uint16_t min_line_blanking_pck = read_reg(mt, MT9F002_MIN_LINE_BLANKING_PCK, 2);
567  uint16_t x_odd_inc = read_reg(mt, MT9F002_X_ODD_INC, 2);
568  uint16_t min_frame_blanking_lines = read_reg(mt, MT9F002_MIN_FRAME_BLANKING_LINES, 2);
569  uint16_t min_line_length_pck = read_reg(mt, MT9F002_MIN_LINE_LENGTH_PCK, 2);
570 
571  /* Calculate minimum line length */
572  float subsampling_factor = (float)(1 + x_odd_inc) / 2.0f; // See page 52
573  uint16_t min_line_length = Max(min_line_length_pck,
574  mt->scaled_width / subsampling_factor + min_line_blanking_pck); // EQ 9
575  min_line_length = Max(min_line_length,
576  (mt->scaled_width - 1 + x_odd_inc) / subsampling_factor / 2 + min_line_blanking_pck);
577  if (mt->interface == MT9F002_MIPI ||
578  mt->interface == MT9F002_HiSPi) {
579  min_line_length = Max(min_line_length,
580  ((uint16_t)((float)mt->scaled_width * mt->vt_pix_clk / mt->op_pix_clk) / 2) + 0x005E); // 2 lanes, pll clocks
581  } else {
582  min_line_length = Max(min_line_length,
583  ((uint16_t)((float)mt->scaled_width * mt->vt_pix_clk / mt->op_pix_clk)) + 0x005E); // pll clocks
584  }
585 
586  /* Do some magic to get it to work with P7 ISP (with horizontal blanking) */
587  uint32_t clkRatio_num = mt->op_sys_clk_div * mt->op_pix_clk_div * mt->row_speed_10_8 * (1 + mt->shift_vt_pix_clk_div);
588  uint32_t clkRatio_den = mt->vt_sys_clk_div * mt->vt_pix_clk_div;
589 
590  /* Divide by the GCD to find smallest ratio */
591  uint32_t clkRatio_gcd = int32_gcd(clkRatio_num, clkRatio_den);
592  clkRatio_num = clkRatio_num / clkRatio_gcd;
593  clkRatio_den = clkRatio_den / clkRatio_gcd;
594 
595  /* Calculate minimum horizontal blanking, since fpga line_length must be divisible by 2 */
596  uint32_t min_horizontal_blanking = clkRatio_num;
597  if ((clkRatio_den % 2) != 0) {
598  min_horizontal_blanking = 2 * clkRatio_num;
599  }
600 
601  /* Fix fpga correction based on min horizontal blanking */
602  if ((min_line_length % min_horizontal_blanking) != 0) {
603  min_line_length += min_horizontal_blanking - (min_line_length % min_horizontal_blanking);
604  }
605 
606  /* Calculate minimum frame length lines */
607  uint16_t min_frame_length = (mt->scaled_height) / subsampling_factor + min_frame_blanking_lines; // (EQ 10)
608 
609  /* Calculate FPS we get using these minimums (Maximum FPS) */
610  mt->line_length = min_line_length;
611  mt->frame_length = min_frame_length;
612  mt->real_fps = mt->vt_pix_clk * 1000000 / (float)(mt->line_length * mt->frame_length);
613 
614  /* Check if we need to downscale the FPS and bruteforce better solution */
615  if (mt->target_fps < mt->real_fps) {
616  float min_fps_err = fabs(mt->target_fps - mt->real_fps);
617  float new_fps = mt->real_fps;
618 
619  // Go through all possible line lengths
620  for (uint16_t ll = min_line_length; ll <= MT9F002_LINE_LENGTH_MAX; ll += min_horizontal_blanking) {
621  // Go through all possible frame lengths
622  for (uint16_t fl = min_frame_length; fl < MT9F002_FRAME_LENGTH_MAX; ++fl) {
623  new_fps = mt->vt_pix_clk * 1000000 / (float)(ll * fl);
624 
625  // Calculate FPS error and save if it is better
626  float fps_err = fabs(mt->target_fps - new_fps);
627  if (fps_err < min_fps_err) {
628  min_fps_err = fps_err;
629  mt->line_length = ll;
630  mt->frame_length = fl;
631  mt->real_fps = new_fps;
632  }
633 
634  // Stop searching if FPS is lower or equal
635  if (mt->target_fps > new_fps) {
636  break;
637  }
638  }
639 
640  // Calculate if next step is still needed (since we only need to go one step below target_fps)
641  new_fps = mt->vt_pix_clk * 1000000 / (float)(ll * min_frame_length);
642 
643  // Stop searching if FPS is lower or equal
644  if (mt->target_fps > new_fps) {
645  break;
646  }
647  }
648  }
649 
650  /* Actually set the calculated values */
651  write_reg(mt, MT9F002_LINE_LENGTH_PCK, mt->line_length, 2);
652  write_reg(mt, MT9F002_FRAME_LENGTH_LINES, mt->frame_length, 2);
653 }

659 void mt9f002_set_exposure(struct mt9f002_t *mt)
660 {
661  /* Fetch minimum and maximum integration times */
662  uint16_t coarse_integration_min = read_reg(mt, MT9F002_COARSE_INTEGRATION_TIME_MIN, 2);
663  uint16_t coarse_integration_max = mt->frame_length - read_reg(mt, MT9F002_COARSE_INTEGRATION_TIME_MAX_MARGIN, 2);
664  uint16_t fine_integration_min = read_reg(mt, MT9F002_FINE_INTEGRATION_TIME_MIN, 2);
665  uint16_t fine_integration_max = mt->line_length - read_reg(mt, MT9F002_FINE_INTEGRATION_TIME_MAX_MARGIN, 2);
666 
667  /* Compute fine and coarse integration time */
668  uint32_t integration = mt->target_exposure * mt->vt_pix_clk * 1000;
669  uint16_t coarse_integration = integration / mt->line_length;
670  uint16_t fine_integration = integration % mt->line_length;
671 
672  /* Make sure fine integration is inside bounds */
673  if (fine_integration_min > fine_integration || fine_integration > fine_integration_max) {
674  int32_t upper_coarse_integration = coarse_integration + 1;
675  int32_t upper_fine_integration = fine_integration_min;
676 
677  int32_t lower_coarse_integration = coarse_integration - 1;
678  int32_t lower_fine_integration = fine_integration_max;
679 
680  // Check if lower case is invalid (take upper coarse)
681  if (lower_coarse_integration < coarse_integration_min) {
682  coarse_integration = upper_coarse_integration;
683  fine_integration = upper_fine_integration;
684  }
685  // Check if upper case is invalid (take lower coarse)
686  else if (upper_coarse_integration > coarse_integration_max) {
687  coarse_integration = lower_coarse_integration;
688  fine_integration = lower_fine_integration;
689  }
690  // Both are good
691  else {
692  // Calculate error to decide which is better
693  int32_t upper_error = abs((mt->line_length * upper_coarse_integration + upper_fine_integration) - integration);
694  int32_t lower_error = abs((mt->line_length * lower_coarse_integration + lower_fine_integration) - integration);
695 
696  if (upper_error < lower_error) {
697  coarse_integration = upper_coarse_integration;
698  fine_integration = upper_fine_integration;
699  } else {
700  coarse_integration = lower_coarse_integration;
701  fine_integration = lower_fine_integration;
702  }
703  }
704  }
705 
706  /* Fix saturations */
707  Bound(fine_integration, fine_integration_min, fine_integration_max);
708  Bound(coarse_integration, coarse_integration_min, coarse_integration_max);
709 
710  /* Set the registers */
711  mt->real_exposure = (float)(coarse_integration * mt->line_length + fine_integration) / (mt->vt_pix_clk * 1000);
712  write_reg(mt, MT9F002_COARSE_INTEGRATION_TIME, coarse_integration, 2);
713  write_reg(mt, MT9F002_FINE_INTEGRATION_TIME_, fine_integration, 2);
714 }

774 {

778 
779  /* Set scaling */

781  mt->output_scaler = 0.125;

784  mt->output_scaler = 1.;

786  }
787  uint16_t scaleFactor = ceil((float)MT9F002_SCALER_N / mt->output_scaler);
788  mt->output_scaler = (float)MT9F002_SCALER_N / scaleFactor;
789  int x_skip_factor = (mt->x_odd_inc + 1) / 2;
790  int y_skip_factor = (mt->y_odd_inc + 1) / 2;
791  mt->scaled_width = mt->output_width * x_skip_factor / mt->output_scaler + mt->x_odd_inc - 1;
792  mt->scaled_height = mt->output_height * y_skip_factor / mt->output_scaler + mt->y_odd_inc - 1;
793

797  }
798  if (mt->scaled_height % (y_skip_factor * 8) != 0) {
799  mt->scaled_height = floor(mt->scaled_height / (y_skip_factor * 8)) * (y_skip_factor * 8);
800  printf("[MT9F002] Warning, scaled_height not a multiple of %i, changing to %i\n", 8 * y_skip_factor, mt->scaled_height);
801  }
802  if (mt->output_scaler < 1.0) {

804  write_reg(mt, MT9F002_SCALE_M, scaleFactor, 2);

808  mt->scaled_height);

811  if (-mt->offset_x * MT9F002_MAX_WIDTH > mt->scaled_width) {mt->offset_x = -(float)mt->scaled_width / 2 / MT9F002_MAX_WIDTH;}

814  if (-mt->offset_y * MT9F002_MAX_HEIGHT > mt->scaled_height / 2) {mt->offset_y = -(float)mt->scaled_height / 2 / MT9F002_MAX_HEIGHT;}
815 
816  /* Set position (based on offset and subsample increment) */
817  uint16_t start_addr_x = (uint16_t)((MT9F002_MAX_WIDTH - mt->scaled_width) / 2 + mt->offset_x * MT9F002_MAX_WIDTH);
818  uint16_t start_addr_y = (uint16_t)((MT9F002_MAX_HEIGHT - mt->scaled_height) / 2 + mt->offset_y * MT9F002_MAX_HEIGHT);
819 
820  if (start_addr_x < 24) {

826  start_addr_x = round(start_addr_x / (x_skip_factor * 8)) * (x_skip_factor * 8);

831  printf("[MT9F002] Warning, offset_y not a multiple of %i, changing to %i\n", 8 * y_skip_factor, start_addr_y);
832  }
833  write_reg(mt, MT9F002_X_ADDR_START, start_addr_x, 2);
834  write_reg(mt, MT9F002_Y_ADDR_START, start_addr_y , 2);
835  uint16_t end_addr_x = start_addr_x + mt->scaled_width - 1;
836  uint16_t end_addr_y = start_addr_y + mt->scaled_height - 1;
837  write_reg(mt, MT9F002_X_ADDR_END, end_addr_x, 2);

843  isp_config.statistics_bayer.window_x.x_width = mt->scaled_width / BAYERSTATS_STATX;

845 
846  isp_config.statistics_yuv.window_pos_x.window_x_end = mt->scaled_width - 1;
847  isp_config.statistics_yuv.window_pos_y.window_y_end = mt->scaled_height - 1;
