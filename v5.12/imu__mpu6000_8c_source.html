81 void imu_mpu_spi_event(void)
82 {
83  mpu60x0_spi_event(&imu_mpu_spi.mpu);
84  if (imu_mpu_spi.mpu.data_available) {
85  uint32_t now_ts = get_sys_time_usec();
86  RATES_COPY(imu.gyro_unscaled, imu_mpu_spi.mpu.data_rates.rates);
87  VECT3_COPY(imu.accel_unscaled, imu_mpu_spi.mpu.data_accel.vect);
88  imu_mpu_spi.mpu.data_available = false;
89  imu_scale_gyro(&imu);
90  imu_scale_accel(&imu);
91  AbiSendMsgIMU_GYRO_INT32(IMU_MPU6000_ID, now_ts, &imu.gyro);
92  AbiSendMsgIMU_ACCEL_INT32(IMU_MPU6000_ID, now_ts, &imu.accel);
93  }
