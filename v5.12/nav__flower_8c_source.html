90 bool nav_flower_run(void)
91 {
92  TransCurrentX = stateGetPositionEnu_f()->x - WaypointX(Center);
93  TransCurrentY = stateGetPositionEnu_f()->y - WaypointY(Center);
94  DistanceFromCenter = sqrtf(TransCurrentX * TransCurrentX + TransCurrentY * TransCurrentY);
95 
96  bool InCircle = true;
97  float CircleTheta;
98 
99  if (DistanceFromCenter > Flowerradius) {
100  InCircle = false;
101  }
102 
103  NavVerticalAutoThrottleMode(0); /* No pitch */
104  NavVerticalAltitudeMode(waypoints[Center].a, 0.);
105 
106  switch (CFlowerStatus) {
107  case Outside:
108  nav_route_xy(FlyFromX, FlyFromY, Fly2X, Fly2Y);
109  if (InCircle) {
110  CFlowerStatus = FlowerLine;
111  FlowerTheta = atan2f(TransCurrentY, TransCurrentX);
112  Fly2X = Flowerradius * cosf(FlowerTheta + 3.14) + WaypointX(Center);
113  Fly2Y = Flowerradius * sinf(FlowerTheta + 3.14) + WaypointY(Center);
114  FlyFromX = stateGetPositionEnu_f()->x;
115  FlyFromY = stateGetPositionEnu_f()->y;
116  nav_init_stage();
117  }
118  break;
119  case FlowerLine:
120  nav_route_xy(FlyFromX, FlyFromY, Fly2X, Fly2Y);
121  if (!InCircle) {
122  CFlowerStatus = Circle;
123  CircleTheta = nav_radius / Flowerradius;
124  CircleX = Flowerradius * cosf(FlowerTheta + 3.14 - CircleTheta) + WaypointX(Center);
125  CircleY = Flowerradius * sinf(FlowerTheta + 3.14 - CircleTheta) + WaypointY(Center);
126  nav_init_stage();
127  }
128  break;
129  case Circle:
130  nav_circle_XY(CircleX, CircleY, nav_radius);
131  if (InCircle) {
132  EdgeCurrentX = WaypointX(Edge) - WaypointX(Center);
133  EdgeCurrentY = WaypointY(Edge) - WaypointY(Center);
134  Flowerradius = sqrtf(EdgeCurrentX * EdgeCurrentX + EdgeCurrentY * EdgeCurrentY);
135  if (DistanceFromCenter > Flowerradius) {
136  CFlowerStatus = Outside;
137  } else {
138  CFlowerStatus = FlowerLine;
139  }
140  FlowerTheta = atan2f(TransCurrentY, TransCurrentX);
141  Fly2X = Flowerradius * cosf(FlowerTheta + 3.14) + WaypointX(Center);
142  Fly2Y = Flowerradius * sinf(FlowerTheta + 3.14) + WaypointY(Center);
143  FlyFromX = stateGetPositionEnu_f()->x;
144  FlyFromY = stateGetPositionEnu_f()->y;
145  nav_init_stage();
146  }
147  break;
148 
149  default:
150  break;
151  }
152  return true;
153 }
