125 bool nav_launcher_run(void)
126 {
127  //Find distance from laucher
128  float dist_x = stateGetPositionEnu_f()->x - launch_x;
129  float dist_y = stateGetPositionEnu_f()->y - launch_y;
130  float launch_dist = sqrtf(dist_x * dist_x + dist_y * dist_y);
131  if (launch_dist <= 0.01) {
132  launch_dist = 0.01;
133  }
134 
135  switch (CLaunch_Status) {
136  case L_Pitch_Nav:
137  //Follow Launch Line
138  NavVerticalAltitudeMode(launch_alt, 0);
139  NavVerticalAutoThrottleMode(LAUNCHER_TAKEOFF_PITCH);
140  NavVerticalThrottleMode(MAX_PPRZ * (1));
141  NavAttitude(0);
142 
143 
144  //If the plane has been launched and has traveled for more than a specified distance, switch to line nav
145  if (stateGetHorizontalSpeedNorm_f() > LAUNCHER_TAKEOFF_MIN_SPEED_LINE) {
146  if (launch_dist > LAUNCHER_TAKEOFF_DISTANCE) {
147  launch_line_x = stateGetPositionEnu_f()->x;
148  launch_line_y = stateGetPositionEnu_f()->y;
149  CLaunch_Status = L_Line_Nav;
150  }
151  }
152 
153  break;
154  case L_Line_Nav:
155  //Follow Launch Line
156  NavVerticalAltitudeMode(launch_alt, 0);
157  NavVerticalAutoThrottleMode(LAUNCHER_TAKEOFF_PITCH);
158  NavVerticalThrottleMode(MAX_PPRZ * (1));
159  nav_route_xy(launch_x, launch_y, launch_line_x, launch_line_y);
160 
161  //If the aircraft is above a specific alt, greater than a specific speed or too far away, circle up
162  if (((stateGetPositionUtm_f()->alt
163  > (launch_alt - LAUNCHER_TAKEOFF_HEIGHT_THRESHOLD))
164  && (stateGetHorizontalSpeedNorm_f()
165  > LAUNCHER_TAKEOFF_MIN_SPEED_CIRCLE))
166  || (launch_dist > LAUNCHER_TAKEOFF_MAX_CIRCLE_DISTANCE)) {
167  CLaunch_Status = L_CircleUp;
168 
169  //Find position of circle
170  float x_1 = dist_x / launch_dist;
171  float y_1 = dist_y / launch_dist;
172 
173  launch_circle.x = stateGetPositionEnu_f()->x
174  + y_1 * LAUNCHER_TAKEOFF_CIRCLE_RADIUS;
175  launch_circle.y = stateGetPositionEnu_f()->y
176  - x_1 * LAUNCHER_TAKEOFF_CIRCLE_RADIUS;
177  }
178  break;
179  case L_CircleUp:
180  NavVerticalAutoThrottleMode(0);
181  NavVerticalAltitudeMode(launch_circle_alt, 0);
182  nav_circle_XY(launch_circle.x, launch_circle.y,
183  LAUNCHER_TAKEOFF_CIRCLE_RADIUS);
184 
185  if (stateGetPositionUtm_f()->alt
186  > (launch_circle_alt - LAUNCHER_TAKEOFF_HEIGHT_THRESHOLD)) {
187  CLaunch_Status = L_Finished;
188  return FALSE;
189  }
190  break;
191  default:
192  break;
193  }
194  return TRUE;
195 }
