84 void baro_periodic(void)
85 {
86  // check i2c_done
87  if (!i2c_idle(&i2c2)) { return; }
88 
89  switch (baro_board.status) {
90  case LBS_UNINITIALIZED:
91  baro_board_send_reset();
92  baro_board.status = LBS_RESETED;
93  break;
94  case LBS_RESETED:
95  baro_board_send_config_abs();
96  baro_board.status = LBS_INITIALIZING_ABS;
97  break;
98  case LBS_INITIALIZING_ABS:
99  baro_board_set_current_register(BARO_ABS_ADDR, 0x00);
100  baro_board.status = LBS_INITIALIZING_ABS_1;
101  break;
102  case LBS_INITIALIZING_ABS_1:
103  baro_board_send_config_diff();
104  baro_board.status = LBS_INITIALIZING_DIFF;
105  break;
106  case LBS_INITIALIZING_DIFF:
107  baro_board_set_current_register(BARO_DIFF_ADDR, 0x00);
108  baro_board.status = LBS_INITIALIZING_DIFF_1;
109  // baro_board.status = LBS_UNINITIALIZED;
110  break;
111  case LBS_INITIALIZING_DIFF_1:
112  baro_board.running = true;
113  case LBS_READ_DIFF:
114  baro_board_read_from_current_register(BARO_ABS_ADDR);
115  baro_board.status = LBS_READING_ABS;
116  break;
117  case LBS_READ_ABS:
118  baro_board_read_from_current_register(BARO_DIFF_ADDR);
119  baro_board.status = LBS_READING_DIFF;
120  break;
121  default:
122  break;
123  }
124 
125 #ifdef BARO_LED
126  if (baro_board.running == TRUE) {
127  LED_ON(BARO_LED);
128  } else {
129  LED_TOGGLE(BARO_LED);
130  }
131 #endif
132 }

134 void lisa_l_baro_event(void)
135 {
136  if (baro_board.status == LBS_READING_ABS &&
137  baro_trans.status != I2CTransPending) {
138  baro_board.status = LBS_READ_ABS;
139  if (baro_trans.status == I2CTransSuccess) {
140  int16_t tmp = baro_trans.buf[0] << 8 | baro_trans.buf[1];
141  float pressure = LISA_L_BARO_SENS * (float)tmp;
142  AbiSendMsgBARO_ABS(BARO_BOARD_SENDER_ID, pressure);
143  }
144  } else if (baro_board.status == LBS_READING_DIFF &&
145  baro_trans.status != I2CTransPending) {
146  baro_board.status = LBS_READ_DIFF;
147  if (baro_trans.status == I2CTransSuccess) {
148  int16_t tmp = baro_trans.buf[0] << 8 | baro_trans.buf[1];
149  float diff = LISA_L_DIFF_SENS * (float)tmp;
150  AbiSendMsgBARO_DIFF(BARO_BOARD_SENDER_ID, diff);
151  }
152  }
153 }
