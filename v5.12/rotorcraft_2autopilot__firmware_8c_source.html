109 static void send_energy(struct transport_tx *trans, struct link_device *dev)
110 {
111  uint16_t e = electrical.energy;
112  if (fabs(electrical.energy) >= INT16_MAX) {
113  e = INT16_MAX;
114  }
115  float vsup = ((float)electrical.vsupply) / 10.0f;
116  float curs = ((float)electrical.current) / 1000.0f;
117  float power = vsup * curs;
118  pprz_msg_send_ENERGY(trans, dev, AC_ID, &vsup, &curs, &e, &power);
119 }

207 void autopilot_event(void)
208 {
209  if (autopilot.detect_ground_once
210 #ifdef AP_MODE_FAILSAFE
211  || autopilot.mode == AP_MODE_FAILSAFE
212 #endif
213  ) {
214  struct NedCoor_f *accel = stateGetAccelNed_f();
215  if (accel->z < -THRESHOLD_GROUND_DETECT ||
216  accel->z > THRESHOLD_GROUND_DETECT) {
217  autopilot.ground_detected = true;
218  autopilot.detect_ground_once = false;
219  }
220  }

232 void autopilot_check_in_flight(bool motors_on)
233 {
234  if (autopilot.in_flight) {
235  if (autopilot_in_flight_counter > 0) {
236  /* probably in_flight if thrust, speed and accel above IN_FLIGHT_MIN thresholds */
237  if ((stabilization_cmd[COMMAND_THRUST] <= AUTOPILOT_IN_FLIGHT_MIN_THRUST) &&
238  (fabsf(stateGetSpeedNed_f()->z) < AUTOPILOT_IN_FLIGHT_MIN_SPEED) &&
239  (fabsf(stateGetAccelNed_f()->z) < AUTOPILOT_IN_FLIGHT_MIN_ACCEL)) {
240  autopilot_in_flight_counter--;
241  if (autopilot_in_flight_counter == 0) {
242  autopilot.in_flight = false;
243  }
244  } else { /* thrust, speed or accel not above min threshold, reset counter */
245  autopilot_in_flight_counter = AUTOPILOT_IN_FLIGHT_TIME;
246  }
247  }
248  } else { /* currently not in flight */
249  if (autopilot_in_flight_counter < AUTOPILOT_IN_FLIGHT_TIME &&
250  motors_on) {
251  /* if thrust above min threshold, assume in_flight.
252  * Don't check for velocity and acceleration above threshold here...
253  */
254  if (stabilization_cmd[COMMAND_THRUST] > AUTOPILOT_IN_FLIGHT_MIN_THRUST) {
255  autopilot_in_flight_counter++;
256  if (autopilot_in_flight_counter == AUTOPILOT_IN_FLIGHT_TIME) {
257  autopilot.in_flight = true;
258  }
259  } else { /* currently not in_flight and thrust below threshold, reset counter */
260  autopilot_in_flight_counter = 0;
261  }
262  }
263  }
