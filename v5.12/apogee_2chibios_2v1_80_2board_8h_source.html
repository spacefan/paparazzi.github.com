130 /*
131  * I/O ports initial setup, this configuration is established soon after reset
132  * in the initialization code.
133  * Please refer to the STM32 Reference Manual for details.
134  */
135 #define PIN_MODE_INPUT(n) (0U << ((n) * 2U))
136 #define PIN_MODE_OUTPUT(n) (1U << ((n) * 2U))
137 #define PIN_MODE_ALTERNATE(n) (2U << ((n) * 2U))
138 #define PIN_MODE_ANALOG(n) (3U << ((n) * 2U))
139 #define PIN_ODR_LOW(n) (0U << (n))
140 #define PIN_ODR_HIGH(n) (1U << (n))
141 #define PIN_OTYPE_PUSHPULL(n) (0U << (n))
142 #define PIN_OTYPE_OPENDRAIN(n) (1U << (n))
143 #define PIN_OSPEED_2M(n) (0U << ((n) * 2U))
144 #define PIN_OSPEED_25M(n) (1U << ((n) * 2U))
145 #define PIN_OSPEED_50M(n) (2U << ((n) * 2U))
146 #define PIN_OSPEED_100M(n) (3U << ((n) * 2U))
147 #define PIN_PUPDR_FLOATING(n) (0U << ((n) * 2U))
148 #define PIN_PUPDR_PULLUP(n) (1U << ((n) * 2U))
149 #define PIN_PUPDR_PULLDOWN(n) (2U << ((n) * 2U))
150 #define PIN_AFIO_AF(n, v) ((v) << (((n) % 8U) * 4U))
