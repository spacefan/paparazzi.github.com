114 {
115  uint8_t temp = (spi_slave_hs_tx_insert_idx + 1) % SPI_SLAVE_HS_TX_BUFFER_SIZE;
116  if (temp != spi_slave_hs_tx_extract_idx) /* there is room left */
117  {
118  spi_slave_hs_tx_buffer[spi_slave_hs_tx_insert_idx] = byte;
119  spi_slave_hs_tx_insert_idx = temp;
