143 void airspeed_amsys_read_event(void)
144 {
145 
146  // Get raw airspeed from buffer
147  airspeed_amsys_raw = 0;
148  airspeed_amsys_raw = (airspeed_amsys_i2c_trans.buf[0] << 8) | airspeed_amsys_i2c_trans.buf[1];
149 #ifdef MEASURE_AMSYS_TEMPERATURE
150  tempAS_amsys_raw = (airspeed_amsys_i2c_trans.buf[2] << 8) | airspeed_amsys_i2c_trans.buf[3];
151  const float temp_off_scale = (float)(TEMPERATURE_AMSYS_MAX - TEMPERATURE_AMSYS_MIN) /
152  (TEMPERATURE_AMSYS_OFFSET_MAX - TEMPERATURE_AMSYS_OFFSET_MIN);
153  // Tmin=-25, Tmax=85
154  airspeed_temperature = temp_off_scale * (tempAS_amsys_raw - TEMPERATURE_AMSYS_OFFSET_MIN) +
155  TEMPERATURE_AMSYS_MIN;
156 #endif
157 
158  // Check if this is valid airspeed
159  if (airspeed_amsys_raw == 0) {
160  airspeed_amsys_valid = false;
161  } else {
162  airspeed_amsys_valid = true;
163  }
164 
165  // Continue only if a new airspeed value was received
166  if (airspeed_amsys_valid) {
167 
168  // raw not under offest min
169  if (airspeed_amsys_raw < AIRSPEED_AMSYS_OFFSET_MIN) {
170  airspeed_amsys_raw = AIRSPEED_AMSYS_OFFSET_MIN;
171  }
172  // raw not over offest max
173  if (airspeed_amsys_raw > AIRSPEED_AMSYS_OFFSET_MAX) {
174  airspeed_amsys_raw = AIRSPEED_AMSYS_OFFSET_MAX;
175  }
176 
177  // calculate raw to pressure
178  const float p_off_scale = (float)(AIRSPEED_AMSYS_MAXPRESURE) /
179  (AIRSPEED_AMSYS_OFFSET_MAX - AIRSPEED_AMSYS_OFFSET_MIN);
180  airspeed_amsys_p = p_off_scale * (airspeed_amsys_raw - AIRSPEED_AMSYS_OFFSET_MIN);
181 
182  if (!airspeed_amsys_offset_init) {
183  --airspeed_amsys_cnt;
184  // Check if averaging completed
185  if (airspeed_amsys_cnt == 0) {
186  // Calculate average
187  airspeed_amsys_offset = airspeed_amsys_offset_tmp / AIRSPEED_AMSYS_OFFSET_NBSAMPLES_AVRG;
188  airspeed_amsys_offset_init = true;
189  }
190  // Check if averaging needs to continue
191  else if (airspeed_amsys_cnt <= AIRSPEED_AMSYS_OFFSET_NBSAMPLES_AVRG) {
192  airspeed_amsys_offset_tmp += airspeed_amsys_p;
193  }
194 
195  airspeed_amsys = 0.;
196 
197  } else {
198  airspeed_amsys_p = airspeed_amsys_p - airspeed_amsys_offset;
199  if (airspeed_amsys_p <= 0) {
200  airspeed_amsys_p = 0.000000001;
201  }
202  // convert pressure to airspeed
203  airspeed_amsys_tmp = sqrtf(2 * airspeed_amsys_p * airspeed_scale / 1.2041);
204  // Lowpassfiltering
205  airspeed_amsys = airspeed_filter * airspeed_old +
206  (1.0 - airspeed_filter) * airspeed_amsys_tmp;
207  airspeed_old = airspeed_amsys;
208 
209  //New value available
210 #if USE_AIRSPEED
211  stateSetAirspeed_f(airspeed_amsys);
212 #endif
213 #ifdef AIRSPEED_AMSYS_SYNC_SEND
214  airspeed_amsys_downlink();
215 #endif
216  }
217 
218  }
219  /*else {
220  airspeed_amsys = 0.0;
221  }*/
222 
223 
224  // Transaction has been read
225  airspeed_amsys_i2c_trans.status = I2CTransDone;
226 }
