106 void lidar_lite_periodic(void)
107 {
108  switch (lidar_lite.status) {
109  case LIDAR_LITE_INIT_RANGING:
110  if (lidar_lite.trans.status == I2CTransDone) {
111  // ask for i2c frame
112  lidar_lite.trans.buf[0] = LIDAR_LITE_REG_ADDR; // sets register pointer to (0x00)
113  lidar_lite.trans.buf[1] = LIDAR_LITE_REG_VAL; // take acquisition & correlation processing with DC correction
114  if (i2c_transmit(&LIDAR_LITE_I2C_DEV, &lidar_lite.trans, lidar_lite.addr, 2)){
115  // transaction OK, increment status
116  lidar_lite.status = LIDAR_LITE_REQ_READ;
117  }
118  }
119  break;
120  case LIDAR_LITE_REQ_READ:
121  if (lidar_lite.trans.status == I2CTransDone) {
122  lidar_lite.trans.buf[0] = LIDAR_LITE_READ_ADDR; // sets register pointer to results register
123  if (i2c_transmit(&LIDAR_LITE_I2C_DEV, &lidar_lite.trans, lidar_lite.addr, 1)){
124  // transaction OK, increment status
125  lidar_lite.status = LIDAR_LITE_READ_DISTANCE;
126  }
127  }
128  break;
129  case LIDAR_LITE_READ_DISTANCE:
130  if (lidar_lite.trans.status == I2CTransDone) {
131  // clear buffer
132  lidar_lite.trans.buf[0] = 0;
133  lidar_lite.trans.buf[1] = 0;
134  if (i2c_receive(&LIDAR_LITE_I2C_DEV, &lidar_lite.trans, lidar_lite.addr, 2)){
135  // transaction OK, increment status
136  lidar_lite.status = LIDAR_LITE_PARSE;
137  }
138  }
139  break;
140  case LIDAR_LITE_PARSE:
141  // filter data
142  lidar_lite.distance_raw = update_median_filter_i(
143  &lidar_lite_filter,
144  (uint32_t)((lidar_lite.trans.buf[0] << 8) | lidar_lite.trans.buf[1]));
145  lidar_lite.distance = ((float)lidar_lite.distance_raw)/100.0;
146 
147  // compensate AGL measurement for body rotation
148  if (lidar_lite.compensate_rotation) {
149  float phi = stateGetNedToBodyEulers_f()->phi;
150  float theta = stateGetNedToBodyEulers_f()->theta;
151  float gain = (float)fabs( (double) (cosf(phi) * cosf(theta)));
152  lidar_lite.distance = lidar_lite.distance / gain;
153  }
154 
155  // send message (if requested)
156  if (lidar_lite.update_agl) {
157  AbiSendMsgAGL(AGL_LIDAR_LITE_ID, lidar_lite.distance);
158  }
159 
160  // increment status
161  lidar_lite.status = LIDAR_LITE_INIT_RANGING;
162  break;
163  default:
164  break;
165  }
166 }
