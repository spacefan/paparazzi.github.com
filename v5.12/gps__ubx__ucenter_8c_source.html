100 void gps_ubx_ucenter_periodic(void)
101 {
102  switch (gps_ubx_ucenter.status) {
103  // Save processing time inflight
104  case GPS_UBX_UCENTER_STATUS_STOPPED:
105  return;
106  break;
107  // Automatically Determine Current Baudrate
108  case GPS_UBX_UCENTER_STATUS_AUTOBAUD:
109 #ifdef GPS_I2C
110  gps_ubx_ucenter.cnt = 0;
111  gps_ubx_ucenter.status = GPS_UBX_UCENTER_STATUS_CONFIG;
112 #else
113  if (gps_ubx_ucenter_autobaud(gps_ubx_ucenter.cnt) == FALSE) {
114  gps_ubx_ucenter.status = GPS_UBX_UCENTER_STATUS_CONFIG;
115  gps_ubx_ucenter.cnt = 0;
116 #if PRINT_DEBUG_GPS_UBX_UCENTER
117  if (gps_ubx_ucenter.baud_init > 0) {
118  DEBUG_PRINT("Initial ublox baudrate found: %u\n", gps_ubx_ucenter.baud_init);
119  } else {
120  DEBUG_PRINT("WARNING: Unable to determine the ublox baudrate. Autoconfiguration is unlikely to work.\n");
121  }
122 #endif
123  } else {
124  gps_ubx_ucenter.cnt++;
125  }
126 #endif /* GPS_I2C */
127  break;
128  // Send Configuration
129  case GPS_UBX_UCENTER_STATUS_CONFIG:
130  if (gps_ubx_ucenter_configure(gps_ubx_ucenter.cnt) == FALSE) {
131  gps_ubx_ucenter.status = GPS_UBX_UCENTER_STATUS_STOPPED;
132 #ifdef GPS_I2C
133  gps_i2c_begin();
134 #endif
135  gps_ubx_ucenter.cnt = 0;
136  } else {
137  gps_ubx_ucenter.cnt++;
138 #ifdef GPS_I2C
139  gps_ubx_ucenter.status = GPS_UBX_UCENTER_STATUS_WAITING;
140  }
141  break;
142  case GPS_UBX_UCENTER_STATUS_WAITING:
143  if (gps_i2c_tx_is_ready())
144  {
145  gps_ubx_ucenter.status = GPS_UBX_UCENTER_STATUS_CONFIG;
146 #endif /*GPS_I2C*/
147  }
148  break;
149  default:
150  // stop this module now...
151  // todo
152  break;
153  }
154 }

160 void gps_ubx_ucenter_event(void)
161 {
162  // Save processing time inflight
163  if (gps_ubx_ucenter.status == GPS_UBX_UCENTER_STATUS_STOPPED) {
164  return;
165  }
166 
167  // Read Configuration Reply's
168  switch (gps_ubx.msg_class) {
169  case UBX_ACK_ID:
170  if (gps_ubx.msg_id == UBX_ACK_ACK_ID) {
171  gps_ubx_ucenter.reply = GPS_UBX_UCENTER_REPLY_ACK;
172  DEBUG_PRINT("ACK\n");
173  } else if (gps_ubx.msg_id == UBX_ACK_NAK_ID) {
174  gps_ubx_ucenter.reply = GPS_UBX_UCENTER_REPLY_NACK;
175  DEBUG_PRINT("NACK\n");
176  }
177  break;
178  case UBX_MON_ID:
179  if (gps_ubx.msg_id == UBX_MON_VER_ID) {
180  gps_ubx_ucenter.reply = GPS_UBX_UCENTER_REPLY_VERSION;
181  gps_ubx_ucenter.sw_ver_h = UBX_MON_VER_c(gps_ubx.msg_buf, 0) - '0';
182  gps_ubx_ucenter.sw_ver_l = 10 * (UBX_MON_VER_c(gps_ubx.msg_buf, 2) - '0');
183  gps_ubx_ucenter.sw_ver_l += UBX_MON_VER_c(gps_ubx.msg_buf, 3) - '0';
184  gps_ubx_ucenter.hw_ver_h = UBX_MON_VER_c(gps_ubx.msg_buf, 33) - '0';
185  gps_ubx_ucenter.hw_ver_h += 10 * (UBX_MON_VER_c(gps_ubx.msg_buf, 32) - '0');
186  gps_ubx_ucenter.hw_ver_l = UBX_MON_VER_c(gps_ubx.msg_buf, 37) - '0';
187  gps_ubx_ucenter.hw_ver_l += 10 * (UBX_MON_VER_c(gps_ubx.msg_buf, 36) - '0');
188 
189  DEBUG_PRINT("ublox sw_ver: %u.%u\n", gps_ubx_ucenter.sw_ver_h, gps_ubx_ucenter.sw_ver_l);
190  DEBUG_PRINT("ublox hw_ver: %u.%u\n", gps_ubx_ucenter.hw_ver_h, gps_ubx_ucenter.hw_ver_l);
191  }
192  break;
193  case UBX_CFG_ID:
194  if (gps_ubx.msg_id == UBX_CFG_PRT_ID) {
195  gps_ubx_ucenter.reply = GPS_UBX_UCENTER_REPLY_CFG_PRT;
196  gps_ubx_ucenter.port_id = UBX_CFG_PRT_PortId(gps_ubx.msg_buf, 0);
197  gps_ubx_ucenter.baud_run = UBX_CFG_PRT_Baudrate(gps_ubx.msg_buf, 0);
198 
199  DEBUG_PRINT("gps_ubx_ucenter.baud_run: %u\n", gps_ubx_ucenter.baud_run);
200  DEBUG_PRINT("gps_ubx_ucenter.port_id: %u\n", gps_ubx_ucenter.port_id);
201  }
202  break;
203  default:
204  break;
205  }
206 }

250 static bool gps_ubx_ucenter_autobaud(uint8_t nr)
251 {
252  switch (nr) {
253  case 0:
254  case 1:
255  // Very important for some modules:
256  // Give the GPS some time to boot (up to 0.75 second)
257  break;
258  case 2:
259  gps_ubx_ucenter.reply = GPS_UBX_UCENTER_REPLY_NONE;
260  uart_periph_set_baudrate(&(UBX_GPS_LINK), B38400); // Try the most common first?
261  gps_ubx_ucenter_config_port_poll();
262  break;
263  case 3:
264  if (gps_ubx_ucenter.reply == GPS_UBX_UCENTER_REPLY_ACK) {
265  gps_ubx_ucenter.baud_init = gps_ubx_ucenter.baud_run;
266  return false;
267  }
268  gps_ubx_ucenter.reply = GPS_UBX_UCENTER_REPLY_NONE;
269  uart_periph_set_baudrate(&(UBX_GPS_LINK), B9600); // Maybe the factory default?
270  gps_ubx_ucenter_config_port_poll();
271  break;
272  case 4:
273  if (gps_ubx_ucenter.reply == GPS_UBX_UCENTER_REPLY_ACK) {
274  gps_ubx_ucenter.baud_init = gps_ubx_ucenter.baud_run;
275  return false;
276  }
277  gps_ubx_ucenter.reply = GPS_UBX_UCENTER_REPLY_NONE;
278  uart_periph_set_baudrate(&(UBX_GPS_LINK), B57600); // The high-rate default?
279  gps_ubx_ucenter_config_port_poll();
280  break;
281  case 5:
282  if (gps_ubx_ucenter.reply == GPS_UBX_UCENTER_REPLY_ACK) {
283  gps_ubx_ucenter.baud_init = gps_ubx_ucenter.baud_run;
284  return false;
285  }
286  gps_ubx_ucenter.reply = GPS_UBX_UCENTER_REPLY_NONE;
287  uart_periph_set_baudrate(&(UBX_GPS_LINK), B4800); // Default NMEA baudrate?
288  gps_ubx_ucenter_config_port_poll();
289  break;
290  case 6:
291  if (gps_ubx_ucenter.reply == GPS_UBX_UCENTER_REPLY_ACK) {
292  gps_ubx_ucenter.baud_init = gps_ubx_ucenter.baud_run;
293  return false;
294  }
295  gps_ubx_ucenter.reply = GPS_UBX_UCENTER_REPLY_NONE;
296  uart_periph_set_baudrate(&(UBX_GPS_LINK), B115200); // Last possible option for ublox
297  gps_ubx_ucenter_config_port_poll();
298  break;
299  case 7:
300  if (gps_ubx_ucenter.reply == GPS_UBX_UCENTER_REPLY_ACK) {
301  gps_ubx_ucenter.baud_init = gps_ubx_ucenter.baud_run;
302  return false;
303  }
304  gps_ubx_ucenter.reply = GPS_UBX_UCENTER_REPLY_NONE;
305  uart_periph_set_baudrate(&(UBX_GPS_LINK), B230400); // Last possible option for ublox
306  gps_ubx_ucenter_config_port_poll();
307  break;
308  case 8:
309  if (gps_ubx_ucenter.reply == GPS_UBX_UCENTER_REPLY_ACK) {
310  gps_ubx_ucenter.baud_init = gps_ubx_ucenter.baud_run;
311  return false;
312  }
313 
314  // Autoconfig Failed... let's setup the failsafe baudrate
315  // Should we try even a different baudrate?
316  gps_ubx_ucenter.baud_init = 0; // Set as zero to indicate that we couldn't verify the baudrate
317  uart_periph_set_baudrate(&(UBX_GPS_LINK), B9600);
318  return false;
319  default:
320  break;
321  }
322  return true;
323 }

424 static inline void gps_ubx_ucenter_config_port(void)
425 {
426  switch (gps_ubx_ucenter.port_id) {
427  // I2C Interface
428  case GPS_PORT_DDC:
429 #ifdef GPS_I2C
430  UbxSend_CFG_PRT(gps_ubx_ucenter.dev, gps_ubx_ucenter.port_id, 0x0, 0x0, (0x42<<1), 0x0, UBX_PROTO_MASK, UBX_PROTO_MASK, 0x0, 0x0);
431 #else
432  DEBUG_PRINT("WARNING: Please include the gps_i2c module.\n");
433 #endif
434  break;
435  // UART Interface
436  case GPS_PORT_UART1:
437  case GPS_PORT_UART2:
438  UbxSend_CFG_PRT(gps_ubx_ucenter.dev,
439  gps_ubx_ucenter.port_id, RESERVED, RESERVED,
440  UBX_UART_MODE_MASK, UART_SPEED(gps_ubx_ucenter.baud_target), UBX_PROTO_MASK | NMEA_PROTO_MASK | RTCM3_PROTO_MASK,
441  UBX_PROTO_MASK | (NMEA_PROTO_MASK & GPS_UBX_ENABLE_NMEA_DATA_MASK), 0x0, 0x0);
442  break;
443  // USB Interface
444  case GPS_PORT_USB:
445  UbxSend_CFG_PRT(gps_ubx_ucenter.dev,
446  gps_ubx_ucenter.port_id, 0x0, 0x0, 0x0, 0x0,
447  UBX_PROTO_MASK | NMEA_PROTO_MASK, UBX_PROTO_MASK| (NMEA_PROTO_MASK & GPS_UBX_ENABLE_NMEA_DATA_MASK), 0x0, 0x0);
448  break;
449  case GPS_PORT_SPI:
450  DEBUG_PRINT("WARNING: ublox SPI port is currently not supported.\n");
451  break;
452  default:
453  DEBUG_PRINT("WARNING: Unknown ublox port id: %u\n", gps_ubx_ucenter.port_id);
454  break;
455  }

479 static bool gps_ubx_ucenter_configure(uint8_t nr)
480 {
481  DEBUG_PRINT("gps_ubx_ucenter_configure nr: %u\n", nr);
482 
483  // Store the reply of the last configuration step and reset
484  if (nr < GPS_UBX_UCENTER_CONFIG_STEPS) {
485  gps_ubx_ucenter.replies[nr] = gps_ubx_ucenter.reply;
486  }
487 
488  switch (nr) {
489  case 0:
490  // Use old baudrate to issue a baudrate change command
491  gps_ubx_ucenter_config_port();
492  break;
493  case 1:
494 #ifndef GPS_I2C
495 #if PRINT_DEBUG_GPS_UBX_UCENTER
496  if (gps_ubx_ucenter.reply != GPS_UBX_UCENTER_REPLY_ACK) {
497  DEBUG_PRINT("ublox did not acknowledge port configuration.\n");
498  } else {
499  DEBUG_PRINT("Changed ublox baudrate to: %u\n", UART_SPEED(gps_ubx_ucenter.baud_target));
500  }
501 #endif
502  // Now the GPS baudrate should have changed
503  uart_periph_set_baudrate(&(UBX_GPS_LINK), gps_ubx_ucenter.baud_target);
504  gps_ubx_ucenter.baud_run = UART_SPEED(gps_ubx_ucenter.baud_target);
505 #endif /*GPS_I2C*/
506  UbxSend_MON_GET_VER(gps_ubx_ucenter.dev);
507  break;
508  case 2:
509  case 3:
510  case 4:
511  case 5:
512  // UBX_G5010 takes 0.7 seconds to answer a firmware request
513  // Version info is important for proper configuration as different firmwares have different settings
514  break;
515  case 6:
516  // Send some debugging info: detected baudrate, software version etc...
517  gps_ubx_ucenter.replies[0] = (gps_ubx_ucenter.baud_init / 1000);
518  gps_ubx_ucenter.replies[1] = (gps_ubx_ucenter.baud_init - 1000 * gps_ubx_ucenter.replies[0]) / 100;
519  gps_ubx_ucenter.replies[2] = gps_ubx_ucenter.sw_ver_h;
520  gps_ubx_ucenter.replies[3] = gps_ubx_ucenter.sw_ver_l;
521  gps_ubx_ucenter.replies[4] = gps_ubx_ucenter.hw_ver_h;
522  gps_ubx_ucenter.replies[5] = gps_ubx_ucenter.hw_ver_l;
523 #if DEBUG_GPS_UBX_UCENTER
524  DOWNLINK_SEND_DEBUG(DefaultChannel, DefaultDevice, 6, gps_ubx_ucenter.replies);
525 #endif
526  // Configure CFG-NAV(5) message
