40 static void superbitrf_rc_normalize(int16_t *in, int16_t *out, uint8_t count)
41 {
42  uint8_t i;
43  for (i = 0; i < count; i++) {
44  if (i == RADIO_THROTTLE) {
45  out[i] = (in[i] + MAX_PPRZ) / 2;
46  Bound(out[i], 0, MAX_PPRZ);
47  } else {
48  out[i] = in[i];
49  Bound(out[i], MIN_PPRZ, MAX_PPRZ);
50  }
51  }
52 }
