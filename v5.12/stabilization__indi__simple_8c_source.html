165 void indi_init_filters(void) {
166  // tau = 1/(2*pi*Fc)
167  float tau = 1.0/(2.0*M_PI*STABILIZATION_INDI_FILT_CUTOFF);
168  float tau_r = 1.0/(2.0*M_PI*STABILIZATION_INDI_FILT_CUTOFF_R);
169  float tau_axis[3] = {tau, tau, tau_r};
170  float tau_est = 1.0/(2.0*M_PI*STABILIZATION_INDI_ESTIMATION_FILT_CUTOFF);
171  float sample_time = 1.0/PERIODIC_FREQUENCY;
172  // Filtering of gyroscope and actuators
173  for(int8_t i=0; i<3; i++) {
174  init_butterworth_2_low_pass(&indi.u[i], tau_axis[i], sample_time, 0.0);
175  init_butterworth_2_low_pass(&indi.rate[i], tau_axis[i], sample_time, 0.0);
176  init_butterworth_2_low_pass(&indi.est.u[i], tau_est, sample_time, 0.0);
177  init_butterworth_2_low_pass(&indi.est.rate[i], tau_est, sample_time, 0.0);
178  }
179 }

285 static inline void stabilization_indi_calc_cmd(int32_t indi_commands[], struct Int32Quat *att_err, bool rate_control)
286 {
287  // Propagate the filter on the gyroscopes and actuators
288  struct FloatRates *body_rates = stateGetBodyRates_f();
289  filter_pqr(indi.u, &indi.u_act_dyn);
290  filter_pqr(indi.rate, body_rates);
291 
292  // Calculate the derivative of the rates
293  finite_difference_from_filter(indi.rate_d, indi.rate);
294 
295  //The rates used for feedback are by default the measured rates. If needed they can be filtered (see below)
296  struct FloatRates rates_for_feedback;
297  RATES_COPY(rates_for_feedback, (*body_rates));
298 
299  //If there is a lot of noise on the gyroscope, it might be good to use the filtered value for feedback.
300  //Note that due to the delay, the PD controller can not be as aggressive.
301 #if STABILIZATION_INDI_FILTER_ROLL_RATE
302  rates_for_feedback.p = indi.rate[0].o[0];
303 #endif
304 #if STABILIZATION_INDI_FILTER_PITCH_RATE
305  rates_for_feedback.q = indi.rate[1].o[0];
306 #endif
307 #if STABILIZATION_INDI_FILTER_YAW_RATE
308  rates_for_feedback.r = indi.rate[2].o[0];
309 #endif
310 
311  indi.angular_accel_ref.p = indi.reference_acceleration.err_p * QUAT1_FLOAT_OF_BFP(att_err->qx)
312  - indi.reference_acceleration.rate_p * rates_for_feedback.p;
313 
314  indi.angular_accel_ref.q = indi.reference_acceleration.err_q * QUAT1_FLOAT_OF_BFP(att_err->qy)
315  - indi.reference_acceleration.rate_q * rates_for_feedback.q;
316 
317  //This separates the P and D controller and lets you impose a maximum yaw rate.
318  float rate_ref_r = indi.reference_acceleration.err_r * QUAT1_FLOAT_OF_BFP(att_err->qz)/indi.reference_acceleration.rate_r;
319  BoundAbs(rate_ref_r, indi.attitude_max_yaw_rate);
320  indi.angular_accel_ref.r = indi.reference_acceleration.rate_r * (rate_ref_r - rates_for_feedback.r);
321 
322  /* Check if we are running the rate controller and overwrite */
323  if(rate_control) {
324  indi.angular_accel_ref.p = indi.reference_acceleration.rate_p * ((float)radio_control.values[RADIO_ROLL] / MAX_PPRZ * indi.max_rate - body_rates->p);
325  indi.angular_accel_ref.q = indi.reference_acceleration.rate_q * ((float)radio_control.values[RADIO_PITCH] / MAX_PPRZ * indi.max_rate - body_rates->q);
326  indi.angular_accel_ref.r = indi.reference_acceleration.rate_r * ((float)radio_control.values[RADIO_YAW] / MAX_PPRZ * indi.max_rate - body_rates->r);
327  }
328 
329  //Increment in angular acceleration requires increment in control input
330  //G1 is the control effectiveness. In the yaw axis, we need something additional: G2.
331  //It takes care of the angular acceleration caused by the change in rotation rate of the propellers
332  //(they have significant inertia, see the paper mentioned in the header for more explanation)
333  indi.du.p = 1.0 / indi.g1.p * (indi.angular_accel_ref.p - indi.rate_d[0]);
334  indi.du.q = 1.0 / indi.g1.q * (indi.angular_accel_ref.q - indi.rate_d[1]);
335  indi.du.r = 1.0 / (indi.g1.r + indi.g2) * (indi.angular_accel_ref.r - indi.rate_d[2] + indi.g2 * indi.du.r);
336 
337  //add the increment to the total control input
338  indi.u_in.p = indi.u[0].o[0] + indi.du.p;
339  indi.u_in.q = indi.u[1].o[0] + indi.du.q;
340  indi.u_in.r = indi.u[2].o[0] + indi.du.r;
341 
342  //bound the total control input
343  Bound(indi.u_in.p, -4500, 4500);
344  Bound(indi.u_in.q, -4500, 4500);
345  Bound(indi.u_in.r, -4500, 4500);
346 
347  //Propagate input filters
348  //first order actuator dynamics
349  indi.u_act_dyn.p = indi.u_act_dyn.p + STABILIZATION_INDI_ACT_DYN_P * (indi.u_in.p - indi.u_act_dyn.p);
350  indi.u_act_dyn.q = indi.u_act_dyn.q + STABILIZATION_INDI_ACT_DYN_Q * (indi.u_in.q - indi.u_act_dyn.q);
351  indi.u_act_dyn.r = indi.u_act_dyn.r + STABILIZATION_INDI_ACT_DYN_R * (indi.u_in.r - indi.u_act_dyn.r);
352 
353  //Don't increment if thrust is off
354  //TODO: this should be something more elegant, but without this the inputs
355  //will increment to the maximum before even getting in the air.
356  if (stabilization_cmd[COMMAND_THRUST] < 300) {
357  FLOAT_RATES_ZERO(indi.du);
358  FLOAT_RATES_ZERO(indi.u_act_dyn);
359  FLOAT_RATES_ZERO(indi.u_in);
360  } else {
361  // only run the estimation if the commands are not zero.
362  lms_estimation();
363  }
364 
365  /* INDI feedback */
366  indi_commands[COMMAND_ROLL] = indi.u_in.p;
367  indi_commands[COMMAND_PITCH] = indi.u_in.q;
368  indi_commands[COMMAND_YAW] = indi.u_in.r;
369 }

377 void stabilization_indi_run(bool in_flight __attribute__((unused)), bool rate_control)
378 {
379  /* attitude error */
380  struct Int32Quat att_err;
381  struct Int32Quat *att_quat = stateGetNedToBodyQuat_i();
382  int32_quat_inv_comp(&att_err, att_quat, &stab_att_sp_quat);
383  /* wrap it in the shortest direction */
384  int32_quat_wrap_shortest(&att_err);
385  int32_quat_normalize(&att_err);
386 
387  /* compute the INDI command */
388  stabilization_indi_calc_cmd(stabilization_att_indi_cmd, &att_err, rate_control);
389 
390  /* copy the INDI command */
391  stabilization_cmd[COMMAND_ROLL] = stabilization_att_indi_cmd[COMMAND_ROLL];
392  stabilization_cmd[COMMAND_PITCH] = stabilization_att_indi_cmd[COMMAND_PITCH];
393  stabilization_cmd[COMMAND_YAW] = stabilization_att_indi_cmd[COMMAND_YAW];
394 
395  /* bound the result */
396  BoundAbs(stabilization_cmd[COMMAND_ROLL], MAX_PPRZ);
397  BoundAbs(stabilization_cmd[COMMAND_PITCH], MAX_PPRZ);
398  BoundAbs(stabilization_cmd[COMMAND_YAW], MAX_PPRZ);
399 }

423 static inline void lms_estimation(void)
424 {
425  static struct IndiEstimation *est = &indi.est;
426  // Only pass really low frequencies so you don't adapt to noise
427  struct FloatRates *body_rates = stateGetBodyRates_f();
428  filter_pqr(est->u, &indi.u_act_dyn);
429  filter_pqr(est->rate, body_rates);
430 
431  // Calculate the first and second derivatives of the rates and actuators
432  float rate_d_prev[3];
433  float u_d_prev[3];
434  float_vect_copy(rate_d_prev, est->rate_d, 3);
435  float_vect_copy(u_d_prev, est->u_d, 3);
436  finite_difference_from_filter(est->rate_d, est->rate);
437  finite_difference_from_filter(est->u_d, est->u);
438  finite_difference(est->rate_dd, est->rate_d, rate_d_prev);
439  finite_difference(est->u_dd, est->u_d, u_d_prev);
440 
441  // The inputs are scaled in order to avoid overflows
442  float du[3];
443  float_vect_copy(du, est->u_d, 3);
444  float_vect_scale(du, INDI_EST_SCALE, 3);
445  est->g1.p = est->g1.p - (est->g1.p * du[0] - est->rate_dd[0]) * du[0] * est->mu;
446  est->g1.q = est->g1.q - (est->g1.q * du[1] - est->rate_dd[1]) * du[1] * est->mu;
447  float ddu = est->u_dd[2] * INDI_EST_SCALE / PERIODIC_FREQUENCY;
448  float error = (est->g1.r * du[2] + est->g2 * ddu - est->rate_dd[2]);
449  est->g1.r = est->g1.r - error * du[2] * est->mu / 3;
450  est->g2 = est->g2 - error * 1000 * ddu * est->mu / 3;
451 
452  //the g values should be larger than zero, otherwise there is positive feedback, the command will go to max and there is nothing to learn anymore...
453  if (est->g1.p < 0.01) { est->g1.p = 0.01; }
454  if (est->g1.q < 0.01) { est->g1.q = 0.01; }
455  if (est->g1.r < 0.01) { est->g1.r = 0.01; }
456  if (est->g2 < 0.01) { est->g2 = 0.01; }
457 
458  if (indi.adaptive) {
459  //Commit the estimated G values and apply the scaling
460  indi.g1.p = est->g1.p * INDI_EST_SCALE;
461  indi.g1.q = est->g1.q * INDI_EST_SCALE;
462  indi.g1.r = est->g1.r * INDI_EST_SCALE;
463  indi.g2 = est->g2 * INDI_EST_SCALE;
464  }
