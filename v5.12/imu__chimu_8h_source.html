96 {

98  float f;
99  unsigned char b[4];
100  } dat1, dat2;
101 
102  dat1.f = f;
103  dat2.b[0] = dat1.b[3];
104  dat2.b[1] = dat1.b[2];
105  dat2.b[2] = dat1.b[1];
106  dat2.b[3] = dat1.b[0];
107  return dat2.f;
