70 
71  for (i = furuno_cfg_cnt; i < GPS_FURUNO_SETTINGS_NB; i++) {
72  len = strlen(gps_furuno_settings[i]);

74  long fd = 0;
75  if (GpsLinkDevice->check_free_space(GpsLinkDevice->periph, &fd, len + 6)) {
