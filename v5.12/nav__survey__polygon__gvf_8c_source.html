57 static bool gvf_intercept_two_lines(struct FloatVect2 *p, struct FloatVect2 x, struct FloatVect2 y, float a1, float a2,
58  float b1, float b2)
59 {
60  float divider, fac;
61 
62  divider = (((b2 - a2) * (y.x - x.x)) + ((x.y - y.y) * (b1 - a1)));
63  if (divider == 0) { return false; }
64  fac = ((y.x * (x.y - a2)) + (x.x * (a2 - y.y)) + (a1 * (y.y - x.y))) / divider;
65  if (fac > 1.0) { return false; }
66  if (fac < 0.0) { return false; }
67 
68  p->x = a1 + fac * (b1 - a1);
69  p->y = a2 + fac * (b2 - a2);
70 
71  return true;
72 }

80 static bool gvf_get_two_intersects(struct FloatVect2 *x, struct FloatVect2 *y, struct FloatVect2 a, struct FloatVect2 b)
81 {
82  int i, count = 0;
83  struct FloatVect2 tmp;
84 
85  for (i = 0; i < gvf_survey.poly_count - 1; i++)
86  if (gvf_intercept_two_lines(&tmp, a, b, waypoints[gvf_survey.poly_first + i].x, waypoints[gvf_survey.poly_first + i].y,
87  waypoints[gvf_survey.poly_first + i + 1].x, waypoints[gvf_survey.poly_first + i + 1].y)) {
88  if (count == 0) {
89  *x = tmp;
90  count++;
91  } else {
92  *y = tmp;
93  count++;
94  break;
95  }
96  }
97 
98  //wrapover first,last polygon waypoint
99  if (count == 1
100  && gvf_intercept_two_lines(&tmp, a, b, waypoints[gvf_survey.poly_first + gvf_survey.poly_count - 1].x,
101  waypoints[gvf_survey.poly_first + gvf_survey.poly_count - 1].y, waypoints[gvf_survey.poly_first].x,
102  waypoints[gvf_survey.poly_first].y)) {
103  *y = tmp;
104  count++;
105  }
106 
107  if (count != 2) {
108  return false;
109  }
110 
111  //change points
112  if (fabs(gvf_survey.dir_vec.x) > fabs(gvf_survey.dir_vec.y)) {
113  if ((y->x - x->x) / gvf_survey.dir_vec.x < 0.0) {
114  tmp = *x;
115  *x = *y;
116  *y = tmp;
117  }
118  } else if ((y->y - x->y) / gvf_survey.dir_vec.y < 0.0) {
119  tmp = *x;
120  *x = *y;
121  *y = tmp;
122  }
123 
124  return true;
125 }

137 void gvf_nav_survey_polygon_setup(uint8_t first_wp, uint8_t size, float angle, float sweep_width, float shot_dist,
138  float min_rad, float altitude)
139 {
140  int i;
141  struct FloatVect2 small, sweep;
142  float divider, angle_rad = angle / 180.0 * M_PI;
143 
144  if (angle < 0.0) { angle += 360.0; }
145  if (angle >= 360.0) { angle -= 360.0; }
146 
147  gvf_survey.poly_first = first_wp;
148  gvf_survey.poly_count = size;
149 
150  gvf_survey.psa_sweep_width = sweep_width;
151  gvf_survey.psa_min_rad = min_rad;
152  gvf_survey.psa_shot_dist = shot_dist;
153  gvf_survey.psa_altitude = altitude;
154 
155  gvf_survey.segment_angle = angle;
156  gvf_survey.return_angle = angle + 180;
157  if (gvf_survey.return_angle > 359) { gvf_survey.return_angle -= 360; }
158 
159  if (angle <= 45.0 || angle >= 315.0) {
160  //north
161  gvf_survey.dir_vec.y = 1.0;
162  gvf_survey.dir_vec.x = 1.0 * tanf(angle_rad);
163  sweep.x = 1.0;
164  sweep.y = - gvf_survey.dir_vec.x / gvf_survey.dir_vec.y;
165  } else if (angle <= 135.0) {
166  //east
167  gvf_survey.dir_vec.x = 1.0;
168  gvf_survey.dir_vec.y = 1.0 / tanf(angle_rad);
169  sweep.y = - 1.0;
170  sweep.x = gvf_survey.dir_vec.y / gvf_survey.dir_vec.x;
171  } else if (angle <= 225.0) {
172  //south
173  gvf_survey.dir_vec.y = -1.0;
174  gvf_survey.dir_vec.x = -1.0 * tanf(angle_rad);
175  sweep.x = -1.0;
176  sweep.y = gvf_survey.dir_vec.x / gvf_survey.dir_vec.y;
177  } else {
178  //west
179  gvf_survey.dir_vec.x = -1.0;
180  gvf_survey.dir_vec.y = -1.0 / tanf(angle_rad);
181  sweep.y = 1.0;
182  sweep.x = - gvf_survey.dir_vec.y / gvf_survey.dir_vec.x;
183  }
184 
185  //normalize
186  FLOAT_VECT2_NORMALIZE(sweep);
187 
188  VECT2_SMUL(gvf_survey.rad_vec, sweep, gvf_survey.psa_min_rad);
189  VECT2_SMUL(gvf_survey.sweep_vec, sweep, gvf_survey.psa_sweep_width);
190 
191  //begin at leftmost position (relative to gvf_survey.dir_vec)
192  VECT2_COPY(small, waypoints[gvf_survey.poly_first]);
193 
194  divider = (gvf_survey.sweep_vec.y * gvf_survey.dir_vec.x) - (gvf_survey.sweep_vec.x * gvf_survey.dir_vec.y);
195 
196  //calculate the leftmost point if one sees the dir vec as going "up" and the sweep vec as going right
197  if (divider < 0.0) {
198  for (i = 1; i < gvf_survey.poly_count; i++) {
199  if ((gvf_survey.dir_vec.x * (waypoints[gvf_survey.poly_first + i].y - small.y)) + (gvf_survey.dir_vec.y *
200  (small.x - waypoints[gvf_survey.poly_first + i].x)) > 0.0) {
201  VECT2_COPY(small, waypoints[gvf_survey.poly_first + i]);
202  }
203  }
204  } else {
205  for (i = 1; i < gvf_survey.poly_count; i++) {
206  if ((gvf_survey.dir_vec.x * (waypoints[gvf_survey.poly_first + i].y - small.y)) + (gvf_survey.dir_vec.y *
207  (small.x - waypoints[gvf_survey.poly_first + i].x)) > 0.0) {
208  VECT2_COPY(small, waypoints[gvf_survey.poly_first + i]);
209  }
210  }
211  }
212 
213  //calculate the line the defines the first flyover
214  gvf_survey.seg_start.x = small.x + 0.5 * gvf_survey.sweep_vec.x;
215  gvf_survey.seg_start.y = small.y + 0.5 * gvf_survey.sweep_vec.y;
216  VECT2_SUM(gvf_survey.seg_end, gvf_survey.seg_start, gvf_survey.dir_vec);
217 
218  if (!gvf_get_two_intersects(&gvf_survey.seg_start, &gvf_survey.seg_end, gvf_survey.seg_start, gvf_survey.seg_end)) {
219  gvf_survey.stage = gERR;
220  return;
221  }
222 
223  //center of the entry circle
224  VECT2_DIFF(gvf_survey.entry_center, gvf_survey.seg_start, gvf_survey.rad_vec);
225 
226  //fast climbing to desired altitude
227  NavVerticalAutoThrottleMode(0.0);
228  NavVerticalAltitudeMode(gvf_survey.psa_altitude, 0.0);
229 
230  gvf_survey.stage = gENTRY;
231 }

247 bool gvf_nav_survey_polygon_run(void)
248 {
249  NavVerticalAutoThrottleMode(0.0);
250  NavVerticalAltitudeMode(gvf_survey.psa_altitude, 0.0);
251 
252  //entry circle around entry-center until the desired altitude is reached
253  if (gvf_survey.stage == gENTRY) {
254  gvf_nav_direction_circle(gvf_survey.psa_min_rad);
255  gvf_ellipse_XY(gvf_survey.entry_center.x, gvf_survey.entry_center.y, gvf_survey.psa_min_rad, gvf_survey.psa_min_rad, 0);
256  if (NavCourseCloseTo(gvf_survey.segment_angle)
257  && nav_approaching_xy(gvf_survey.seg_start.x, gvf_survey.seg_start.y, last_x, last_y, CARROT)
258  && fabs(stateGetPositionUtm_f()->alt - gvf_survey.psa_altitude) <= 20) {
259  gvf_survey.stage = gSEG;
260  nav_init_stage();
261 #ifdef DIGITAL_CAM
262  dc_survey(gvf_survey.psa_shot_dist, gvf_survey.seg_start.x - gvf_survey.dir_vec.x * gvf_survey.psa_shot_dist * 0.5,
263  gvf_survey.seg_start.y - gvf_survey.dir_vec.y * gvf_survey.psa_shot_dist * 0.5);
264 #endif
265  }
266  }
267  //fly the segment until seg_end is reached
268  if (gvf_survey.stage == gSEG) {
269  gvf_nav_points(gvf_survey.seg_start, gvf_survey.seg_end);
270  //calculate all needed points for the next flyover
271  if (nav_approaching_xy(gvf_survey.seg_end.x, gvf_survey.seg_end.y, gvf_survey.seg_start.x, gvf_survey.seg_start.y, 0)) {
272 #ifdef DIGITAL_CAM
273  dc_stop();
274 #endif
275  VECT2_DIFF(gvf_survey.seg_center1, gvf_survey.seg_end, gvf_survey.rad_vec);
276  gvf_survey.ret_start.x = gvf_survey.seg_end.x - 2 * gvf_survey.rad_vec.x;
277  gvf_survey.ret_start.y = gvf_survey.seg_end.y - 2 * gvf_survey.rad_vec.y;
278 
279  //if we get no intersection the survey is finished
280  static struct FloatVect2 sum_start_sweep;
281  static struct FloatVect2 sum_end_sweep;
282  VECT2_SUM(sum_start_sweep, gvf_survey.seg_start, gvf_survey.sweep_vec);
283  VECT2_SUM(sum_end_sweep, gvf_survey.seg_end, gvf_survey.sweep_vec);
284  if (!gvf_get_two_intersects(&gvf_survey.seg_start, &gvf_survey.seg_end, sum_start_sweep, sum_end_sweep)) {
285  return false;
286  }
287 
288  gvf_survey.ret_end.x = gvf_survey.seg_start.x - gvf_survey.sweep_vec.x - 2 * gvf_survey.rad_vec.x;
289  gvf_survey.ret_end.y = gvf_survey.seg_start.y - gvf_survey.sweep_vec.y - 2 * gvf_survey.rad_vec.y;
290 
291  gvf_survey.seg_center2.x = gvf_survey.seg_start.x - 0.5 * (2.0 * gvf_survey.rad_vec.x + gvf_survey.sweep_vec.x);
292  gvf_survey.seg_center2.y = gvf_survey.seg_start.y - 0.5 * (2.0 * gvf_survey.rad_vec.y + gvf_survey.sweep_vec.y);
293 
294  gvf_survey.stage = gTURN1;
295  nav_init_stage();
296  }
297  }
298  //turn from stage to return
299  else if (gvf_survey.stage == gTURN1) {
300  gvf_nav_direction_circle(gvf_survey.psa_min_rad);
301  gvf_ellipse_XY(gvf_survey.seg_center1.x, gvf_survey.seg_center1.y, gvf_survey.psa_min_rad, gvf_survey.psa_min_rad, 0);
302  if (NavCourseCloseTo(gvf_survey.return_angle)) {
303  gvf_survey.stage = gRET;
304  nav_init_stage();
305  }
306  //return
307  } else if (gvf_survey.stage == gRET) {
308  gvf_nav_points(gvf_survey.ret_start, gvf_survey.ret_end);
309  if (nav_approaching_xy(gvf_survey.ret_end.x, gvf_survey.ret_end.y, gvf_survey.ret_start.x, gvf_survey.ret_start.y, 0)) {
310  gvf_survey.stage = gTURN2;
311  nav_init_stage();
312  }
313  //turn from return to stage
314  } else if (gvf_survey.stage == gTURN2) {
315  float rad_sur = (2 * gvf_survey.psa_min_rad + gvf_survey.psa_sweep_width) * 0.5;
316  gvf_nav_direction_circle(rad_sur);
317  gvf_ellipse_XY(gvf_survey.seg_center2.x, gvf_survey.seg_center2.y, rad_sur, rad_sur, 0);
318  if (NavCourseCloseTo(gvf_survey.segment_angle)) {
319  gvf_survey.stage = gSEG;
320  nav_init_stage();
321 #ifdef DIGITAL_CAM
322  dc_survey(gvf_survey.psa_shot_dist, gvf_survey.seg_start.x - gvf_survey.dir_vec.x * gvf_survey.psa_shot_dist * 0.5,
323  gvf_survey.seg_start.y - gvf_survey.dir_vec.y * gvf_survey.psa_shot_dist * 0.5);
324 #endif
325  }
326  }
327 
328  return true;
329 }
