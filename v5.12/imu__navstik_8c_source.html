102 void imu_navstik_event(void)
103 {
104  uint32_t now_ts = get_sys_time_usec();
105 
106  /* MPU-60x0 event taks */
107  mpu60x0_i2c_event(&imu_navstik.mpu);
108 
109  if (imu_navstik.mpu.data_available) {
110  /* default orientation as should be printed on the pcb, z-down, ICs down */
111  RATES_COPY(imu.gyro_unscaled, imu_navstik.mpu.data_rates.rates);
112  VECT3_COPY(imu.accel_unscaled, imu_navstik.mpu.data_accel.vect);
113 
114  imu_navstik.mpu.data_available = false;
115  imu_scale_gyro(&imu);
116  imu_scale_accel(&imu);
117  AbiSendMsgIMU_GYRO_INT32(IMU_BOARD_ID, now_ts, &imu.gyro);
118  AbiSendMsgIMU_ACCEL_INT32(IMU_BOARD_ID, now_ts, &imu.accel);
119  }
120 
121  /* HMC58XX event task */
122  hmc58xx_event(&imu_navstik.hmc);
123  if (imu_navstik.hmc.data_available) {
124  imu.mag_unscaled.x = imu_navstik.hmc.data.vect.y;
125  imu.mag_unscaled.y = -imu_navstik.hmc.data.vect.x;
126  imu.mag_unscaled.z = imu_navstik.hmc.data.vect.z;
127  imu_navstik.hmc.data_available = false;
128  imu_scale_mag(&imu);
129  AbiSendMsgIMU_MAG_INT32(IMU_BOARD_ID, now_ts, &imu.mag);
130  }
131 }
