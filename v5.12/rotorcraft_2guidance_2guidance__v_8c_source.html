205 void guidance_v_read_rc(void)
206 {
207 
208  /* used in RC_DIRECT directly and as saturation in CLIMB and HOVER */
209  guidance_v_rc_delta_t = (int32_t)radio_control.values[RADIO_THROTTLE];
210 
211  /* used in RC_CLIMB */
212  guidance_v_rc_zd_sp = (MAX_PPRZ / 2) - (int32_t)radio_control.values[RADIO_THROTTLE];
213  DeadBand(guidance_v_rc_zd_sp, GUIDANCE_V_CLIMB_RC_DEADBAND);
214 
215  static const int32_t climb_scale = ABS(SPEED_BFP_OF_REAL(GUIDANCE_V_MAX_RC_CLIMB_SPEED) /
216  (MAX_PPRZ / 2 - GUIDANCE_V_CLIMB_RC_DEADBAND));
217  static const int32_t descent_scale = ABS(SPEED_BFP_OF_REAL(GUIDANCE_V_MAX_RC_DESCENT_SPEED) /
218  (MAX_PPRZ / 2 - GUIDANCE_V_CLIMB_RC_DEADBAND));
219 
220  if (guidance_v_rc_zd_sp > 0) {
221  guidance_v_rc_zd_sp *= descent_scale;
222  } else {
223  guidance_v_rc_zd_sp *= climb_scale;
224  }
225 }

273 void guidance_v_thrust_adapt(bool in_flight)
274 {
275  guidance_v_thrust_coeff = get_vertical_thrust_coeff();
276 
277  if (in_flight) {
278  /* Only run adaptive throttle estimation if we are in flight and
279  * the desired vertical velocity (zd) was updated (i.e. we ran hover_loop before).
280  * This means that the estimation is not updated when using direct throttle commands.
281  *
282  * FIXME... SATURATIONS NOT TAKEN INTO ACCOUNT, AKA SUPERVISION and co
283  */
284  if (desired_zd_updated) {
285  int32_t vertical_thrust = (stabilization_cmd[COMMAND_THRUST] * guidance_v_thrust_coeff) >> INT32_TRIG_FRAC;
286  gv_adapt_run(stateGetAccelNed_i()->z, vertical_thrust, guidance_v_zd_ref);
287  }
288  } else {
289  /* reset estimate while not in_flight */
290  gv_adapt_init();
291  }

294 void guidance_v_run(bool in_flight)
295 {
296  guidance_v_thrust_adapt(in_flight);
297 
298  /* reset flag indicating if desired zd was updated */
299  desired_zd_updated = false;
300 
301  switch (guidance_v_mode) {
302 
303  case GUIDANCE_V_MODE_RC_DIRECT:
304  guidance_v_z_sp = stateGetPositionNed_i()->z; // for display only
305  stabilization_cmd[COMMAND_THRUST] = guidance_v_rc_delta_t;
306  break;
307 
308  case GUIDANCE_V_MODE_RC_CLIMB:
309  guidance_v_zd_sp = guidance_v_rc_zd_sp;
310  gv_update_ref_from_zd_sp(guidance_v_zd_sp, stateGetPositionNed_i()->z);
311  run_hover_loop(in_flight);
312  stabilization_cmd[COMMAND_THRUST] = guidance_v_delta_t;
313  break;
314 
315  case GUIDANCE_V_MODE_CLIMB:
316  gv_update_ref_from_zd_sp(guidance_v_zd_sp, stateGetPositionNed_i()->z);
317  run_hover_loop(in_flight);
318 #if !NO_RC_THRUST_LIMIT
319  /* use rc limitation if available */
320  if (radio_control.status == RC_OK) {
321  stabilization_cmd[COMMAND_THRUST] = Min(guidance_v_rc_delta_t, guidance_v_delta_t);
322  } else
323 #endif
324  stabilization_cmd[COMMAND_THRUST] = guidance_v_delta_t;
325  break;
326 
327  case GUIDANCE_V_MODE_HOVER:
328  guidance_v_guided_mode = GUIDANCE_V_GUIDED_MODE_ZHOLD;
329  case GUIDANCE_V_MODE_GUIDED:
330  guidance_v_guided_run(in_flight);
331  break;
332 
333 #if GUIDANCE_V_MODE_MODULE_SETTING == GUIDANCE_V_MODE_MODULE
334  case GUIDANCE_V_MODE_MODULE:
335  guidance_v_module_run(in_flight);
336  break;
337 #endif
338 
339  case GUIDANCE_V_MODE_NAV: {
340  guidance_v_from_nav(in_flight);
341  break;
342  }
343 
344  case GUIDANCE_V_MODE_FLIP:
345  break;
346 
347  default:
348  break;
349  }
350 }

376 static int32_t get_vertical_thrust_coeff(void)
377 {
378  // cos(30) = 0.8660254
379  static const int32_t max_bank_coef = BFP_OF_REAL(0.8660254f, INT32_TRIG_FRAC);
380 
381  struct Int32RMat *att = stateGetNedToBodyRMat_i();
382  /* thrust vector:
383  * int32_rmat_vmult(&thrust_vect, &att, &zaxis)
384  * same as last colum of rmat with INT32_TRIG_FRAC
385  * struct Int32Vect thrust_vect = {att.m[2], att.m[5], att.m[8]};
386  *
387  * Angle between two vectors v1 and v2:
388  * angle = acos(dot(v1, v2) / (norm(v1) * norm(v2)))
389  * since here both are already of unit length:
390  * angle = acos(dot(v1, v2))
391  * since we we want the cosine of the angle we simply need
392  * thrust_coeff = dot(v1, v2)
393  * also can be simplified considering: v1 is zaxis with (0,0,1)
394  * dot(v1, v2) = v1.z * v2.z = v2.z
395  */
396  int32_t coef = att->m[8];
397  if (coef < max_bank_coef) {
398  coef = max_bank_coef;
399  }
400  return coef;
401 }

406 void run_hover_loop(bool in_flight)
407 {
408 
409  /* convert our reference to generic representation */
410  int64_t tmp = gv_z_ref >> (GV_Z_REF_FRAC - INT32_POS_FRAC);
411  guidance_v_z_ref = (int32_t)tmp;
412  guidance_v_zd_ref = gv_zd_ref << (INT32_SPEED_FRAC - GV_ZD_REF_FRAC);
413  guidance_v_zdd_ref = gv_zdd_ref << (INT32_ACCEL_FRAC - GV_ZDD_REF_FRAC);
414  /* set flag to indicate that desired zd was updated */
415  desired_zd_updated = true;
416  /* compute the error to our reference */
417  int32_t err_z = guidance_v_z_ref - stateGetPositionNed_i()->z;
418  Bound(err_z, GUIDANCE_V_MIN_ERR_Z, GUIDANCE_V_MAX_ERR_Z);
419  int32_t err_zd = guidance_v_zd_ref - stateGetSpeedNed_i()->z;
420  Bound(err_zd, GUIDANCE_V_MIN_ERR_ZD, GUIDANCE_V_MAX_ERR_ZD);
421 
422  if (in_flight) {
423  guidance_v_z_sum_err += err_z;
424  Bound(guidance_v_z_sum_err, -GUIDANCE_V_MAX_SUM_ERR, GUIDANCE_V_MAX_SUM_ERR);
425  } else {
426  guidance_v_z_sum_err = 0;
427  }
428 
429  /* our nominal command : (g + zdd)*m */
430  int32_t inv_m;
431  if (guidance_v_adapt_throttle_enabled) {
432  inv_m = gv_adapt_X >> (GV_ADAPT_X_FRAC - FF_CMD_FRAC);
433  } else {
434  /* use the fixed nominal throttle */
435  inv_m = BFP_OF_REAL(9.81 / (guidance_v_nominal_throttle * MAX_PPRZ), FF_CMD_FRAC);
436  }
437 
438  const int32_t g_m_zdd = (int32_t)BFP_OF_REAL(9.81, FF_CMD_FRAC) -
439  (guidance_v_zdd_ref << (FF_CMD_FRAC - INT32_ACCEL_FRAC));
440 
441  guidance_v_ff_cmd = g_m_zdd / inv_m;
442  /* feed forward command */
443  guidance_v_ff_cmd = (guidance_v_ff_cmd << INT32_TRIG_FRAC) / guidance_v_thrust_coeff;
444 
445 #if HYBRID_NAVIGATION
446  //FIXME: NOT USING FEEDFORWARD COMMAND BECAUSE OF QUADSHOT NAVIGATION
447  guidance_v_ff_cmd = guidance_v_nominal_throttle * MAX_PPRZ;
448 #endif
449 
450  /* bound the nominal command to 0.9*MAX_PPRZ */
451  Bound(guidance_v_ff_cmd, 0, 8640);
452 
453 
454  /* our error feed back command */
455  /* z-axis pointing down -> positive error means we need less thrust */
456  guidance_v_fb_cmd = ((-guidance_v_kp * err_z) >> 7) +
457  ((-guidance_v_kd * err_zd) >> 16) +
458  ((-guidance_v_ki * guidance_v_z_sum_err) >> 16);
459 
460  guidance_v_delta_t = guidance_v_ff_cmd + guidance_v_fb_cmd;
461 
462  /* bound the result */
463  Bound(guidance_v_delta_t, 0, MAX_PPRZ);
464 
465 }

468 {
469  if (vertical_mode == VERTICAL_MODE_ALT) {
