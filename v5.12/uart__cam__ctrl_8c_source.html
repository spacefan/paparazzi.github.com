65 void digital_cam_uart_event(void)
66 {
67  while (CameraLinkChAvailable()) {
68  parse_mora(&mora_protocol, CameraLinkGetch());
69  if (mora_protocol.msg_received) {
70  switch (mora_protocol.msg_id) {
71  case MORA_STATUS:
72  for (int i = 0; i < MORA_STATUS_MSG_SIZE; i++) {
73  mora_status_msg.bin[i] = mora_protocol.payload[i];
74  }
75  digital_cam_uart_status = mora_status_msg.data.shots;
76  break;
77  case MORA_PAYLOAD:
78  for (int i = 0; i < MORA_PAYLOAD_MSG_SIZE; i++) {
79  thumbs[thumb_pointer][i] = mora_protocol.payload[i];
80  }
81  break;
82  default:
83  break;
84  }
85  mora_protocol.msg_received = 0;
86  }
87  }
88 }
