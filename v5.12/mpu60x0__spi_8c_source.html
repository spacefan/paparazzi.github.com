77 void mpu60x0_spi_start_configure(struct Mpu60x0_Spi *mpu)
78 {
79  if (mpu->config.init_status == MPU60X0_CONF_UNINIT) {
80 
81  // First check if we found the chip (succesfull WHO_AM_I response)
82  if(mpu->spi_trans.status == SPITransSuccess && mpu->rx_buf[1] == MPU60X0_WHOAMI_REPLY) {
83  mpu->config.init_status++;
84  mpu->spi_trans.status = SPITransDone;
85  mpu60x0_send_config(mpu60x0_spi_write_to_reg, (void *)mpu, &(mpu->config));
86  }
87  // Send WHO_AM_I to check if chip is there
88  else if(mpu->spi_trans.status != SPITransRunning && mpu->spi_trans.status != SPITransPending) {
89  mpu->spi_trans.output_length = 1;
90  mpu->spi_trans.input_length = 2;
91  mpu->tx_buf[0] = MPU60X0_REG_WHO_AM_I | MPU60X0_SPI_READ;
92  spi_submit(mpu->spi_p, &(mpu->spi_trans));
93  }
94  }
95 }

97 void mpu60x0_spi_read(struct Mpu60x0_Spi *mpu)
98 {
99  if (mpu->config.initialized && mpu->spi_trans.status == SPITransDone) {
100  mpu->spi_trans.output_length = 1;
101  mpu->spi_trans.input_length = 1 + mpu->config.nb_bytes;
102  /* set read bit and multiple byte bit, then address */
103  mpu->tx_buf[0] = MPU60X0_REG_INT_STATUS | MPU60X0_SPI_READ;
104  spi_submit(mpu->spi_p, &(mpu->spi_trans));
105  }

110 void mpu60x0_spi_event(struct Mpu60x0_Spi *mpu)
111 {
112  if (mpu->config.initialized) {
113  if (mpu->spi_trans.status == SPITransFailed) {
114  mpu->spi_trans.status = SPITransDone;
115  } else if (mpu->spi_trans.status == SPITransSuccess) {
116  // Successfull reading
117  if (bit_is_set(mpu->rx_buf[1], 0)) {
118  // new data
119  mpu->data_accel.vect.x = Int16FromBuf(mpu->rx_buf, 2);
120  mpu->data_accel.vect.y = Int16FromBuf(mpu->rx_buf, 4);
121  mpu->data_accel.vect.z = Int16FromBuf(mpu->rx_buf, 6);
122  mpu->data_rates.rates.p = Int16FromBuf(mpu->rx_buf, 10);
123  mpu->data_rates.rates.q = Int16FromBuf(mpu->rx_buf, 12);
124  mpu->data_rates.rates.r = Int16FromBuf(mpu->rx_buf, 14);
125 
126  int16_t temp_raw = Int16FromBuf(mpu->rx_buf, 8);
127 #if ICM20608
128  mpu->temp = (float)temp_raw / 326.8f + 25.0f;
129 #else
130  mpu->temp = (float)temp_raw / 340.0f + 36.53f;
131 #endif
132 
133  // if we are reading slaves, copy the ext_sens_data
134  if (mpu->config.nb_slaves > 0) {
135  /* the buffer is volatile, since filled from ISR
136  * but we know it's ok to use it here so we silence the warning
137  */
138 #pragma GCC diagnostic push
139 #pragma GCC diagnostic ignored "-Wcast-qual"
140  memcpy(mpu->data_ext, (uint8_t *) & (mpu->rx_buf[16]), mpu->config.nb_bytes - 15);
141 #pragma GCC diagnostic pop
142  }
143 
144  mpu->data_available = true;
145  }
146  mpu->spi_trans.status = SPITransDone;
147  }
148  } else if (mpu->config.init_status != MPU60X0_CONF_UNINIT) { // Configuring but not yet initialized
149  switch (mpu->spi_trans.status) {
150  case SPITransFailed:
151  mpu->config.init_status--; // Retry config (TODO max retry)
152  case SPITransSuccess:
153  case SPITransDone:
154  mpu60x0_send_config(mpu60x0_spi_write_to_reg, (void *)mpu, &(mpu->config));
155  if (mpu->config.initialized) {
156  mpu->spi_trans.status = SPITransDone;
157  }
158  break;
159  default:
160  break;
161  }
162  }
163 }

166 bool mpu60x0_configure_i2c_slaves(Mpu60x0ConfigSet mpu_set, void *mpu)
167 {
168  struct Mpu60x0_Spi *mpu_spi = (struct Mpu60x0_Spi *)(mpu);
169 
170  if (mpu_spi->slave_init_status == MPU60X0_SPI_CONF_UNINIT) {
171  mpu_spi->slave_init_status++;
172  }
173 
174  switch (mpu_spi->slave_init_status) {
175  case MPU60X0_SPI_CONF_I2C_MST_CLK:
176  /* configure MPU I2C master clock and stop/start between slave reads */
177  mpu_set(mpu, MPU60X0_REG_I2C_MST_CTRL, ((1 << 4) | mpu_spi->config.i2c_mst_clk));
178  mpu_spi->slave_init_status++;
179  break;
180  case MPU60X0_SPI_CONF_I2C_MST_DELAY:
181  /* Set I2C slaves delayed sample rate */
182  mpu_set(mpu, MPU60X0_REG_I2C_MST_DELAY, mpu_spi->config.i2c_mst_delay);
183  mpu_spi->slave_init_status++;
184  break;
185  case MPU60X0_SPI_CONF_I2C_MST_EN:
186  /* enable internal I2C master and disable primary I2C interface */
187  mpu_set(mpu, MPU60X0_REG_USER_CTRL, ((1 << MPU60X0_I2C_IF_DIS) |
188  (1 << MPU60X0_I2C_MST_EN)));
189  mpu_spi->slave_init_status++;
190  break;
191  case MPU60X0_SPI_CONF_SLAVES_CONFIGURE:
192  /* configure each slave until all nb_slaves are done */
193  if (mpu_spi->config.nb_slave_init < mpu_spi->config.nb_slaves && mpu_spi->config.nb_slave_init < MPU60X0_I2C_NB_SLAVES) {
194  // proceed to next slave if configure for current one returns true
195  if (mpu_spi->config.slaves[mpu_spi->config.nb_slave_init].configure(mpu_set, mpu)) {
196  mpu_spi->config.nb_slave_init++;
197  }
198  }
199  else {
200  /* all slave devies configured, continue MPU side configuration of I2C slave stuff */
201  mpu_spi->slave_init_status++;
202  }
203  break;
204  case MPU60X0_SPI_CONF_DONE:
205  return true;
206  default:
207  break;
208  }
209  return false;
210 }
