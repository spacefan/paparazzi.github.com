43 inline void init_median_filter_i(struct MedianFilterInt *filter, uint8_t size)
44 {
45  uint8_t i;
46  if (size > MAX_MEDIAN_DATASIZE){
47  filter->size = MAX_MEDIAN_DATASIZE;
48  } else if ((size % 2) == 0) {
49  // force filter to have odd number of entries so that
50  // returned median is always an entry and not an average
51  filter->size = size + 1;
52  } else {
53  filter->size = size;
54  }
55  for (i = 0; i < filter->size; i++) {
56  filter->data[i] = 0;
57  filter->sortData[i] = 0;
58  }
59  filter->dataIndex = 0;
60 }

62 inline int32_t update_median_filter_i(struct MedianFilterInt *filter, int32_t new_data)
63 {
64  int temp, i, j; // used to sort array
65 
66  // Insert new data into raw data array round robin style
67  filter->data[filter->dataIndex] = new_data;
68  filter->dataIndex = (filter->dataIndex + 1) % filter->size;
69 
70  // Copy raw data to sort data array
71  memcpy(filter->sortData, filter->data, sizeof(int32_t) * filter->size);
72 
73  // Insertion Sort
74  for (i = 1; i < filter->size; i++) {
75  temp = filter->sortData[i];
76  j = i - 1;
77  while (j >= 0 && temp < filter->sortData[j]) {
78  filter->sortData[j + 1] = filter->sortData[j];
79  j = j - 1;
80  }
81  filter->sortData[j + 1] = temp;
82  }
83  // return data value in middle of sorted array
84  return get_median_filter_i(filter);
85 }

87 inline int32_t get_median_filter_i(struct MedianFilterInt *filter)
88 {
89  if (filter->size % 2){
90  return filter->sortData[filter->size >> 1];
91  } else {
92  // this should not be used if init_median_filter was used
93  return (filter->sortData[filter->size / 2] + filter->sortData[filter->size / 2 - 1]) / 2;
94  }
95 }
