106 void nps_autopilot_run_step(double time)
107 {
108 
109  nps_electrical_run_step(time);
110 
111 #if RADIO_CONTROL && !RADIO_CONTROL_TYPE_DATALINK
112  if (nps_radio_control_available(time)) {
113  radio_control_feed();
114  Fbw(event_task);
115  }
116 #endif
117 
118  if (nps_sensors_gyro_available()) {
119  imu_feed_gyro_accel();
120  Fbw(event_task);
121  Ap(event_task);
122  }
123 
124  if (nps_sensors_mag_available()) {
125  imu_feed_mag();
126  Fbw(event_task);
127  Ap(event_task);
128  }
129 
130  if (nps_sensors_baro_available()) {
131  float pressure = (float) sensors.baro.value;
132  AbiSendMsgBARO_ABS(BARO_SIM_SENDER_ID, pressure);
133  Fbw(event_task);
134  Ap(event_task);
135  }
136 
137  if (nps_sensors_temperature_available()) {
138  AbiSendMsgTEMPERATURE(BARO_SIM_SENDER_ID, (float)sensors.temp.value);
139  }
140 
141 #if USE_AIRSPEED || USE_NPS_AIRSPEED
142  if (nps_sensors_airspeed_available()) {
143  stateSetAirspeed_f((float)sensors.airspeed.value);
144  Fbw(event_task);
145  Ap(event_task);
146  }
147 #endif
148 
149  if (nps_sensors_gps_available()) {
150  gps_feed_value();
151  Fbw(event_task);
152  Ap(event_task);
153  }
154 
155 #if USE_SONAR
156  if (nps_sensors_sonar_available()) {
157  float dist = (float) sensors.sonar.value;
158  AbiSendMsgAGL(AGL_SONAR_NPS_ID, dist);
159 
160  uint16_t foo = 0;
161  DOWNLINK_SEND_SONAR(DefaultChannel, DefaultDevice, &foo, &dist);
162 
163  Fbw(event_task);
164  Ap(event_task);
165  }
166 #endif
167 
168 #if USE_NPS_AOA
169  if (nps_sensors_aoa_available()) {
170  stateSetAngleOfAttack_f((float)sensors.aoa.value);
171  Fbw(event_task);
172  Ap(event_task);
173  }
174 #endif
175 
176 #if USE_NPS_SIDESLIP
177  if (nps_sensors_sideslip_available()) {
178  stateSetSideslip_f((float)sensors.sideslip.value);
179  Fbw(event_task);
180  Ap(event_task);
181  }
182 #endif
183 
184  if (nps_bypass_ahrs) {
185  sim_overwrite_ahrs();
186  }
187 
188  if (nps_bypass_ins) {
189  sim_overwrite_ins();
190  }
191 
192  Fbw(handle_periodic_tasks);
193  Ap(handle_periodic_tasks);
194 
195  /* scale final motor commands to 0-1 for feeding the fdm */
196 #ifdef NPS_ACTUATOR_NAMES
197  PRINT_CONFIG_MSG("actuators for JSBSim explicitly set.")
198  PRINT_CONFIG_VAR(NPS_COMMANDS_NB)
199  //PRINT_CONFIG_VAR(NPS_ACTUATOR_NAMES)
200 
201  for (uint8_t i = 0; i < NPS_COMMANDS_NB; i++) {
202  nps_autopilot.commands[i] = (double)commands[i] / MAX_PPRZ;
203  }
204  // hack: invert pitch to fit most JSBSim models
205  nps_autopilot.commands[COMMAND_PITCH] = -(double)commands[COMMAND_PITCH] / MAX_PPRZ;
206 #else
207  PRINT_CONFIG_MSG("Using throttle, roll, pitch, yaw commands instead of explicit actuators.")
208  PRINT_CONFIG_VAR(COMMAND_THROTTLE)
209  PRINT_CONFIG_VAR(COMMAND_ROLL)
210  PRINT_CONFIG_VAR(COMMAND_PITCH)
211 
212  nps_autopilot.commands[COMMAND_THROTTLE] = (double)commands[COMMAND_THROTTLE] / MAX_PPRZ;
213  nps_autopilot.commands[COMMAND_ROLL] = (double)commands[COMMAND_ROLL] / MAX_PPRZ;
214  // hack: invert pitch to fit most JSBSim models
215  nps_autopilot.commands[COMMAND_PITCH] = -(double)commands[COMMAND_PITCH] / MAX_PPRZ;
216 #ifdef COMMAND_YAW
217  PRINT_CONFIG_VAR(COMMAND_YAW)
218  nps_autopilot.commands[COMMAND_YAW] = (double)commands[COMMAND_YAW] / MAX_PPRZ;
219 #endif /* COMMAND_YAW */
220 #endif /* NPS_ACTUATOR_NAMES */
221 
222  // do the launch when clicking launch in GCS
223  nps_autopilot.launch = autopilot.launch && !autopilot.kill_throttle;
224  if (!autopilot.launch) {
225  nps_autopilot.commands[COMMAND_THROTTLE] = 0;
226  }
227 }
