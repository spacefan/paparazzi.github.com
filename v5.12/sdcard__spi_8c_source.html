99 void sdcard_spi_periodic(struct SDCard *sdcard)
100 {
101 
102  /* Do nothing if spi transaction is in progress */
103  if (sdcard->spi_t.status == SPITransPending || sdcard->spi_t.status == SPITransRunning) {
104  return;
105  }
106 
107  switch (sdcard->status) {
108 
109  /* Send dummy clock to set SD card in SPI mode */
110  case SDCard_BeforeDummyClock:
111  sdcard->spi_t.select = SPINoSelect;
112  sdcard->spi_t.output_length = 10;
113  sdcard->spi_t.input_length = 0;
114  for (uint8_t i = 0; i < 10; i++) {
115  sdcard->output_buf[i] = 0xFF;
116  }
117  sdcard->spi_t.after_cb = &sdcard_spi_spicallback;
118  spi_submit(sdcard->spi_p, &sdcard->spi_t);
119  sdcard->status = SDCard_SendingDummyClock;
120  break;
121 
122  /* Sending ACMD41 */
123  case SDCard_SendingACMD41v2:
124  sdcard_spi_send_app_cmd(sdcard, 41, 0x40000000);
125  sdcard->timeout_counter++;
126  break;
127 
128  /* While busy, keep checking if it is still busy */
129  case SDCard_Busy:
130  sdcard_spi_request_bytes(sdcard, 1);
131  break;
132 
133  /* While waiting for data token, keep polling */
134  case SDCard_WaitingForDataToken:
135  sdcard_spi_request_bytes(sdcard, 1);
136  sdcard->timeout_counter++;
137  break;
138 
139  /* While multiwrite busy, keep checking if it is still busy */
140  case SDCard_MultiWriteBusy:
141  sdcard_spi_request_bytes(sdcard, 1);
142  break;
143 
144  default:
145  break;
146  }

156 void sdcard_spi_spicallback(struct spi_transaction *t)
157 {
158  (void) t; // ignore unused warning
159 
160  switch (sdcard1.status) {
161 
162  /* Ready with dummy clock, proceed with CMD0 */
163  case SDCard_SendingDummyClock:
164  sdcard1.spi_t.select = SPISelectUnselect;
165  sdcard_spi_send_cmd(&sdcard1, 0, 0x00000000);
166  sdcard1.status = SDCard_SendingCMD0;
167  break;
168 
169  /* Ready sending CMD0, start polling result */
170  case SDCard_SendingCMD0:
171  sdcard1.response_counter = 0;
172  sdcard_spi_request_bytes(&sdcard1, 1);
173  sdcard1.status = SDCard_ReadingCMD0Resp;
174  break;
175 
176  /* Continue polling bytes until there is a response or not */
177  case SDCard_ReadingCMD0Resp:
178  if (t->input_buf[0] == 0x01) {
179  sdcard_spi_send_cmd(&sdcard1, 8, 0x000001AA);
180  sdcard1.status = SDCard_SendingCMD8;
181  } else if (sdcard1.response_counter >= 9) {
182  sdcard1.status = SDCard_Error;
183  sdcard1.error_status = SDCardError_InitializationNoResponse;
184  } else {
185  sdcard_spi_request_bytes(&sdcard1, 1);
186  }
187  break;
188 
189  /* Ready sending CMD8, start polling result */
190  case SDCard_SendingCMD8:
191  sdcard1.response_counter = 0;
192  sdcard_spi_request_bytes(&sdcard1, 1);
193  sdcard1.status = SDCard_ReadingCMD8Resp;
194  break;
195 
196  /* Continue polling bytes until there is a response or not */
197  case SDCard_ReadingCMD8Resp:
198  if (t->input_buf[0] == 0x01) {
199  sdcard_spi_request_bytes(&sdcard1, 4);
200  sdcard1.status = SDCard_ReadingCMD8Parameter;
201  } else if (sdcard1.response_counter >= 9) {
202  sdcard1.status = SDCard_Error;
203  sdcard1.error_status = SDCardError_CardInfoNoResponse;
204  } else {
205  sdcard_spi_request_bytes(&sdcard1, 1);
206  }
207  break;
208 
209  /* Process parameter from CMD8 response (after 0x01 was received) */
210  case SDCard_ReadingCMD8Parameter:
211  if (sdcard1.input_buf[0] == 0x00 && sdcard1.input_buf[1] == 0x00 && sdcard1.input_buf[2] == 0x01
212  && sdcard1.input_buf[3] == 0xAA) {
213  sdcard1.status = SDCard_SendingACMD41v2;
214  sdcard1.timeout_counter = 0;
215  } else {
216  sdcard1.status = SDCard_Error;
217  sdcard1.error_status = SDCardError_InvalidCardInfo;
218  }
219  break;
220 
221  /* Ready sending the ACMDv2 command, start polling bytes for response */
222  case SDCard_SendingACMD41v2:
223  sdcard1.response_counter = 0;
224  sdcard_spi_request_bytes(&sdcard1, 1);
225  sdcard1.status = SDCard_ReadingACMD41v2Resp;
226  break;
227 
228  /* Grabbing bytes in response to the ACMD41v2 command */
229  case SDCard_ReadingACMD41v2Resp:
230  if (t->input_buf[0] == 0x01) {
231  if (sdcard1.timeout_counter < 500 - 1) { /* Wait 500 cycles until timeout */
232  sdcard1.status = SDCard_SendingACMD41v2;
233  } else {
234  sdcard1.status = SDCard_Error;
235  sdcard1.error_status = SDCardError_ACMD41Timeout;
236  }
237  } else if (t->input_buf[0] == 0x00) {
238  sdcard_spi_send_cmd(&sdcard1, 58, 0x00000000);
239  sdcard1.status = SDCard_SendingCMD58;
240  } else if (sdcard1.response_counter >= 9) {
241  sdcard1.status = SDCard_Error;
242  sdcard1.error_status = SDCardError_ACMD41NoResponse;
243  } else {
244  sdcard_spi_request_bytes(&sdcard1, 1);
245  }
246  break;
247 
248  /* Ready sending CMD58, request the first byte of R3 response */
249  case SDCard_SendingCMD58:
250  sdcard1.response_counter = 0;
251  sdcard_spi_request_bytes(&sdcard1, 1);
252  sdcard1.status = SDCard_ReadingCMD58Resp;
253  break;
254 
255  /* Continue polling bytes until there is a response to CMD58 */
256  case SDCard_ReadingCMD58Resp:
257  if (sdcard1.response_counter >= 9) {
258  sdcard1.status = SDCard_Error;
259  sdcard1.error_status = SDCardError_CMD58NoResponse;
260  } else if (sdcard1.input_buf[0] == 0x00) {
261  sdcard_spi_request_bytes(&sdcard1, 4);
262  sdcard1.status = SDCard_ReadingCMD58Parameter;
263  } else {
264  sdcard_spi_request_bytes(&sdcard1, 1);
265  }
266  break;
267 
268  /* Parameter of CMD58 ready, processing it */
269  case SDCard_ReadingCMD58Parameter:
270  if (sdcard1.input_buf[0] & 0x80) { // bit 31 set, CCS bit is valid
271  if (sdcard1.input_buf[0] & 0x40) { // bit 30 set
272  sdcard1.card_type = SDCardType_SdV2block;
273  sdcard1.status = SDCard_Idle;
274  } else { // bit 30 not set
275  sdcard1.card_type = SDCardType_SdV1;
276  sdcard_spi_send_cmd(&sdcard1, 16, SD_BLOCK_SIZE);
277  sdcard1.status = SDCard_SendingCMD16;
278  }
279  } else { // bit 31 not set, CCS bit is unvalid
280  sdcard1.status = SDCard_Error;
281  sdcard1.error_status = SDCardError_CCSBitInvalid;
282  }
283  break;
284 
285  /* Ready sending CMD16, request first byte */
286  case SDCard_SendingCMD16:
287  sdcard1.response_counter = 0;
288  sdcard_spi_request_bytes(&sdcard1, 1);
289  sdcard1.status = SDCard_ReadingCMD16Resp;
290  break;
291 
292  /* Continue polling bytes until there is a response to CMD16 */
293  case SDCard_ReadingCMD16Resp:
294  if (sdcard1.input_buf[0] == 0x00) {
295  sdcard1.status = SDCard_Idle;
296  } else if (sdcard1.response_counter >= 9) {
297  sdcard1.status = SDCard_Error;
298  sdcard1.error_status = SDCardError_SetBlockSizeNoResponse;
299  } else {
300  sdcard_spi_request_bytes(&sdcard1, 1);
301  }
302  break;
303 
304  /* Ready sending CMD24, request first response byte */
305  case SDCard_SendingCMD24:
306  sdcard1.response_counter = 0;
307  sdcard_spi_request_bytes(&sdcard1, 1);
308  sdcard1.status = SDCard_ReadingCMD24Resp;
309  break;
310 
311  /* Request additional bytes until response or timeout */
312  case SDCard_ReadingCMD24Resp:
313  if (sdcard1.input_buf[0] == 0x00) {
314  sdcard_spi_request_bytes(&sdcard1, 1);
315  sdcard1.status = SDCard_BeforeSendingDataBlock;
316  } else if (sdcard1.response_counter >= 9) {
317  sdcard1.status = SDCard_Error;
318  sdcard1.error_status = SDCardError_WriteBlockNoResponse;
319  } else {
320  sdcard_spi_request_bytes(&sdcard1, 1);
321  }
322  break;
323 
324  /* Send the data block */
325  case SDCard_BeforeSendingDataBlock:
326  sdcard1.spi_t.input_length = SD_BLOCK_SIZE + 4;
327  sdcard1.spi_t.output_length = SD_BLOCK_SIZE + 4;
328  sdcard1.spi_t.output_buf = &sdcard1.output_buf[5];
329  sdcard1.spi_t.output_buf[0] = 0xFE; // data token
330  sdcard1.spi_t.output_buf[SD_BLOCK_SIZE + 1] = 0xFF; // Fake CRC byte 1
331  sdcard1.spi_t.output_buf[SD_BLOCK_SIZE + 2] = 0xFF; // Fake CRC byte 2
332  sdcard1.spi_t.output_buf[SD_BLOCK_SIZE + 3] = 0xFF; // to request data response
333  sdcard1.spi_t.after_cb = &sdcard_spi_spicallback;
334  if (spi_submit(sdcard1.spi_p, &sdcard1.spi_t)) {
335  sdcard1.status = SDCard_SendingDataBlock;
336  } else {
337  sdcard1.status = SDCard_Error;
338  sdcard1.error_status = SDCardError_SpiDriverError;
339  }
340  break;
341 
342  /* Finished sending the data block */
343  case SDCard_SendingDataBlock:
344  if ((sdcard1.input_buf[SD_BLOCK_SIZE + 3] & 0x0F) == 0x05) {
345  sdcard1.status = SDCard_Busy;
346  } else {
347  sdcard1.status = SDCard_Error;
348  sdcard1.error_status = SDCardError_BlockWriteError;
349  }
350  sdcard1.spi_t.output_buf = sdcard1.output_buf;
351  break;
352 
353  case SDCard_Busy:
354  if (sdcard1.input_buf[0] != 0x00) {
355  sdcard1.status = SDCard_Idle;
356  }
357  break;
358 
359  /* Ready sending CMD17, request first response byte */
360  case SDCard_SendingCMD17:
361  sdcard1.response_counter = 0;
362  sdcard_spi_request_bytes(&sdcard1, 1);
363  sdcard1.status = SDCard_ReadingCMD17Resp;
364  break;
365 
366  /* Read response to CMD17 until response or timeout/error */
367  case SDCard_ReadingCMD17Resp:
368  if (sdcard1.input_buf[0] == 0x00) {
369  sdcard_spi_request_bytes(&sdcard1, 1);
370  sdcard1.timeout_counter = 0;
371  sdcard1.status = SDCard_WaitingForDataToken;
372  } else if (sdcard1.response_counter >= 9) {
373  sdcard1.status = SDCard_Error;
374  sdcard1.error_status = SDCardError_ReadBlockNoResponse;
375  } else {
376  sdcard_spi_request_bytes(&sdcard1, 1);
377  }
378  break;
379 
380  /* Processing byte to see if it is a data token */
381  case SDCard_WaitingForDataToken:
382  if (sdcard1.input_buf[0] == 0xFE) { // Data token received
383  sdcard1.status = SDCard_Idle;
384  sdcard1.spi_t.input_length = SD_BLOCK_SIZE + 2; // 2 Fake CRC bytes
385  sdcard1.spi_t.output_length = SD_BLOCK_SIZE + 2;
386  sdcard1.spi_t.cdiv = SPIDiv8;
387  for (uint16_t i = 0; i < (SD_BLOCK_SIZE + 2); i++) {
388  sdcard1.output_buf[i] = 0xFF;
389  }
390  sdcard1.spi_t.after_cb = &sdcard_spi_spicallback;
391  spi_submit(sdcard1.spi_p, &sdcard1.spi_t);
392  sdcard1.status = SDCard_ReadingDataBlock;
393  } else if (sdcard1.timeout_counter > 498) {
394  sdcard1.status = SDCard_Error;
395  sdcard1.error_status = SDCardError_ReadBlockTimeout;
396  }
397  break;
398 
399  /* Data block received in buffer, process data */
400  case SDCard_ReadingDataBlock:
401  sdcard1.status = SDCard_Idle;
402  if (sdcard1.external_callback != NULL) {
403  sdcard1.external_callback();
404  }
405  break;
406 
407  /* Ready sending CMD25, request first response byte */
408  case SDCard_SendingCMD25:
409  sdcard1.response_counter = 0;
410  sdcard_spi_request_bytes(&sdcard1, 1);
411  sdcard1.status = SDCard_ReadingCMD25Resp;
412  break;
413 
414  /* Request additional bytes until response or timeout */
415  case SDCard_ReadingCMD25Resp:
416  if (sdcard1.input_buf[0] == 0x00) {
417  sdcard_spi_request_bytes(&sdcard1, 1);
418  sdcard1.status = SDCard_MultiWriteIdle;
419  } else if (sdcard1.response_counter >= 9) {
420  sdcard1.status = SDCard_Error;
421  sdcard1.error_status = SDCardError_MultiWriteNoResponse;
422  } else {
423  sdcard_spi_request_bytes(&sdcard1, 1);
424  }
425  break;
426 
427  /* Check the response after a block was written */
428  case SDCard_MultiWriteWriting:
429  if ((sdcard1.input_buf[SD_BLOCK_SIZE + 3] & 0x0F) == 0x05 /* Data accepted */) {
430  sdcard1.status = SDCard_MultiWriteBusy;
431  if(sdcard1.external_callback != NULL) {
432  sdcard1.external_callback();
433  }
434  } else {
435  sdcard1.status = SDCard_Error;
436  sdcard1.error_status = SDCardError_MultiWriteError;
437  }
438  break;
439 
440  /* Check if sd card is still busy after a packet during multiwrite */
441  case SDCard_MultiWriteBusy:
442  if (sdcard1.input_buf[0] != 0x00) {
443  sdcard1.status = SDCard_MultiWriteIdle;
444  }
445  break;
446 
447  /* Finished the stop command transaction, waiting for internal processes of the SD card */
448  case SDCard_MultiWriteStopping:
449  sdcard1.status = SDCard_Busy;
450  break;
451 
452  /* Should not reach this */
453  default:
454  break;
455  }
456 }

467 void sdcard_spi_send_cmd(struct SDCard *sdcard, uint8_t cmd, uint32_t arg)
468 {
469  (void) cmd; (void) arg;
470  sdcard->spi_t.input_length = 6;
471  sdcard->spi_t.output_length = 6;
472  sdcard->output_buf[0] = 0x40 | cmd;
473  sdcard->output_buf[1] = arg >> 24;
474  sdcard->output_buf[2] = arg >> 16;
475  sdcard->output_buf[3] = arg >> 8;
476  sdcard->output_buf[4] = arg;
477  switch (cmd) {
478  case 0:
479  sdcard->output_buf[5] = 0x95; /* CRC byte for CMD0 */
480  break;
481  case 8:
482  sdcard->output_buf[5] = 0x87; /* CRC byte for CMD8 */
483  break;
484  default:
485  sdcard->output_buf[5] = 0x01; /* Fake CRC byte, will be ignored by the SDCard */
486  }
487  sdcard->spi_t.after_cb = &sdcard_spi_spicallback;
488 
489  spi_submit(sdcard->spi_p, &sdcard->spi_t);
490 }

502 void sdcard_spi_send_app_cmd(struct SDCard *sdcard, uint8_t cmd, uint32_t arg)
503 {
504  sdcard->spi_t.output_length = 21;
505  sdcard->spi_t.input_length = 21;
506

521  sdcard->output_buf[14] = 0xFF;

525  sdcard->output_buf[17] = arg >> 16;
526  sdcard->output_buf[18] = arg >> 8;
527  sdcard->output_buf[19] = arg;
528  sdcard->output_buf[20] = 0x01; /* End of the ACMD */

639 {
