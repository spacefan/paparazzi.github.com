187 void stabilization_rate_run(bool in_flight)
188 {
189  /* compute feed-back command */
190  struct FloatRates _error;
191  struct FloatRates *body_rate = stateGetBodyRates_f();
192  RATES_DIFF(_error, stabilization_rate_sp, (*body_rate));
193  if (in_flight) {
194  /* update integrator */
195  //divide the sum_err_increment to make sure it doesn't accumulate to the max too fast
196  struct FloatRates sum_err_increment;
197  RATES_SDIV(sum_err_increment, _error, PERIODIC_FREQUENCY);
198  RATES_ADD(stabilization_rate_sum_err, sum_err_increment);
199  RATES_BOUND_CUBE(stabilization_rate_sum_err, -MAX_SUM_ERR, MAX_SUM_ERR);
200  } else {
201  FLOAT_RATES_ZERO(stabilization_rate_sum_err);
202  }
203 
204  /* PI */
205  stabilization_rate_fb_cmd.p = stabilization_rate_gain.p * _error.p +
206  stabilization_rate_igain.p * stabilization_rate_sum_err.p;
207 
208  stabilization_rate_fb_cmd.q = stabilization_rate_gain.q * _error.q +
209  stabilization_rate_igain.q * stabilization_rate_sum_err.q;
210 
211  stabilization_rate_fb_cmd.r = stabilization_rate_gain.r * _error.r +
212  stabilization_rate_igain.r * stabilization_rate_sum_err.r;
213 
214  stabilization_cmd[COMMAND_ROLL] = stabilization_rate_fb_cmd.p;
215  stabilization_cmd[COMMAND_PITCH] = stabilization_rate_fb_cmd.q;
216  stabilization_cmd[COMMAND_YAW] = stabilization_rate_fb_cmd.r;
217 
218  /* bound the result */
219  BoundAbs(stabilization_cmd[COMMAND_ROLL], MAX_PPRZ);
220  BoundAbs(stabilization_cmd[COMMAND_PITCH], MAX_PPRZ);
221  BoundAbs(stabilization_cmd[COMMAND_YAW], MAX_PPRZ);
222 
223 }
