169 void gps_ubx_i2c_periodic(void)
170 {
171  if (gps_i2c.trans.status == I2CTransDone)
172  {
173  switch(gps_i2c.write_state)
174  {
175  case gps_i2c_write_standby:
176  if (!gps_i2c_char_available(&gps_i2c))
177  {
178  if (gps_ubx_i2c_bytes_to_read > GPS_I2C_BUF_SIZE || gps_ubx_i2c_ucenter_done)
179  {
180  gps_i2c.write_state = gps_i2c_write_request_size;
181  } else {
182  gps_i2c.tx_rdy = TRUE;
183  }
184  }
185  break;
186 
187  case gps_i2c_write_request_size:
188  gps_i2c.trans.buf[0] = GPS_I2C_ADDR_NB_AVAIL_BYTES;
189  i2c_transceive(&GPS_UBX_I2C_DEV, &gps_i2c.trans, GPS_I2C_SLAVE_ADDR, 1, 2);
190  gps_i2c.write_state = gps_i2c_write_standby;
191  gps_i2c.read_state = gps_i2c_read_sizeof;
192  break;
193 
194  case gps_i2c_write_cfg:
195  memcpy((uint8_t *) & (gps_i2c.trans.buf), gps_i2c.tx_buf, gps_i2c.tx_buf_idx);
196  i2c_transmit(&GPS_UBX_I2C_DEV, &gps_i2c.trans, GPS_I2C_SLAVE_ADDR, gps_i2c.tx_buf_idx);
197  gps_i2c.tx_buf_idx = 0;
198  gps_i2c.read_state = gps_i2c_read_standby;
199  gps_i2c.write_state = gps_i2c_write_request_size;
200  break;
201 
202  default:
203  break;
204  }
205  }
206 }

208 void gps_ubx_i2c_read_event(void)
209 {
210  switch(gps_i2c.read_state)
211  {
212  case gps_i2c_read_standby:
213  break;
214  break;
215 
216  // how many bytes are available
217  case gps_i2c_read_sizeof:
218  gps_ubx_i2c_bytes_to_read = ((uint16_t)(gps_i2c.trans.buf[0]) << 7) | (uint16_t)(gps_i2c.trans.buf[1]);
219  gps_i2c.trans.status = I2CTransDone;
220  if (gps_ubx_i2c_bytes_to_read == 0xFFFF || gps_ubx_i2c_bytes_to_read == 0x0000)
221  {
222  gps_i2c.write_state = gps_i2c_write_request_size;
223  return;
224  } else if (gps_ubx_i2c_bytes_to_read > GPS_I2C_BUF_SIZE)
225  {
226  gps_i2c.trans.buf[0] = GPS_I2C_ADDR_DATA;
227  i2c_transceive(&GPS_UBX_I2C_DEV, &gps_i2c.trans, GPS_I2C_SLAVE_ADDR, 1, GPS_I2C_BUF_SIZE);
228  gps_i2c.read_state = gps_i2c_read_data;
229  } else
230  {
231  gps_i2c.trans.buf[0] = GPS_I2C_ADDR_DATA;
232  i2c_transceive(&GPS_UBX_I2C_DEV, &gps_i2c.trans, GPS_I2C_SLAVE_ADDR, 1, gps_ubx_i2c_bytes_to_read);
233  gps_i2c.read_state = gps_i2c_read_data;
234  return;
235  }
236  break;
237  case gps_i2c_read_data:
238  if (gps_ubx_i2c_bytes_to_read > GPS_I2C_BUF_SIZE)
239  {
240  memcpy(&gps_i2c.rx_buf, (uint8_t *) & (gps_i2c.trans.buf), GPS_I2C_BUF_SIZE);
241  gps_i2c.rx_buf_idx = 0;
242  gps_i2c.rx_buf_avail = GPS_I2C_BUF_SIZE;
243  } else
244  {
245  memcpy(&gps_i2c.rx_buf, (uint8_t *) & (gps_i2c.trans.buf), gps_ubx_i2c_bytes_to_read);
246  gps_i2c.rx_buf_idx = 0;
247  gps_i2c.rx_buf_avail = gps_ubx_i2c_bytes_to_read;
248  }
249 
250  gps_i2c.write_state = gps_i2c_write_standby;
251  gps_i2c.read_state = gps_i2c_read_standby;
252  break;
253 
254  default:
255  break;
256  }
257 
258  // Transaction has been read
259  gps_i2c.trans.status = I2CTransDone;
260 }
