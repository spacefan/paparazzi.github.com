93 void nps_autopilot_run_step(double time)
94 {
95 
96  nps_electrical_run_step(time);
97 
98 #if RADIO_CONTROL && !RADIO_CONTROL_TYPE_DATALINK
99  if (nps_radio_control_available(time)) {
100  radio_control_feed();
101  main_event();
102  }
103 #endif
104 
105  if (nps_sensors_gyro_available()) {
106  imu_feed_gyro_accel();
107  main_event();
108  }
109 
110  if (nps_sensors_mag_available()) {
111  imu_feed_mag();
112  main_event();
113  }
114 
115  if (nps_sensors_baro_available()) {
116  float pressure = (float) sensors.baro.value;
117  AbiSendMsgBARO_ABS(BARO_SIM_SENDER_ID, pressure);
118  main_event();
119  }
120 
121  if (nps_sensors_temperature_available()) {
122  AbiSendMsgTEMPERATURE(BARO_SIM_SENDER_ID, (float)sensors.temp.value);
123  }
124 
125 #if USE_AIRSPEED
126  if (nps_sensors_airspeed_available()) {
127  stateSetAirspeed_f((float)sensors.airspeed.value);
128  }
129 #endif
130 
131 #if USE_SONAR
132  if (nps_sensors_sonar_available()) {
133  float dist = (float) sensors.sonar.value;
134  AbiSendMsgAGL(AGL_SONAR_NPS_ID, dist);
135 
136  uint16_t foo = 0;
137  DOWNLINK_SEND_SONAR(DefaultChannel, DefaultDevice, &foo, &dist);
138 
139  main_event();
140  }
141 #endif
142 
143 #if USE_GPS
144  if (nps_sensors_gps_available()) {
145  gps_feed_value();
146  main_event();
147  }
148 #endif
149 
150  if (nps_bypass_ahrs) {
151  sim_overwrite_ahrs();
152  }
153 
154  if (nps_bypass_ins) {
155  sim_overwrite_ins();
156  }
157 
158  handle_periodic_tasks();
159 
160  /* scale final motor commands to 0-1 for feeding the fdm */
161  for (uint8_t i = 0; i < NPS_COMMANDS_NB; i++) {
162 #if NPS_NO_MOTOR_MIXING
163  actuators_pprz[i] = autopilot_get_motors_on() ? actuators_pprz[i] : 0;
164  nps_autopilot.commands[i] = (double)actuators_pprz[i] / MAX_PPRZ;
165 #else
166  nps_autopilot.commands[i] = (double)motor_mixing.commands[i] / MAX_PPRZ;
167 #endif
168  }
169 }
