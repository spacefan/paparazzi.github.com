126 static void gyro_cb(uint8_t sender_id __attribute__((unused)),
127  uint32_t stamp, struct Int32Rates *gyro)
128 {
129  struct FloatRates gyro_f;
130  RATES_FLOAT_OF_BFP(gyro_f, *gyro);
131 
132 #if USE_AUTO_AHRS_FREQ || !defined(AHRS_PROPAGATE_FREQUENCY)
133  PRINT_CONFIG_MSG("Calculating dt for AHRS float_invariant propagation.")
134  /* timestamp in usec when last callback was received */
135  static uint32_t last_stamp = 0;
136 
137  if (last_stamp > 0 && ahrs_float_inv.is_aligned) {
138  float dt = (float)(stamp - last_stamp) * 1e-6;
139  ahrs_float_invariant_propagate(&gyro_f, dt);
140  compute_body_orientation_and_rates();
141  }
142  last_stamp = stamp;
143 #else
144  PRINT_CONFIG_MSG("Using fixed AHRS_PROPAGATE_FREQUENCY for AHRS float_invariant propagation.")
145  PRINT_CONFIG_VAR(AHRS_PROPAGATE_FREQUENCY)
146  const float dt = 1. / (AHRS_PROPAGATE_FREQUENCY);
147  if (ahrs_float_inv.is_aligned) {
148  ahrs_float_invariant_propagate(&gyro_f, dt);
149  compute_body_orientation_and_rates();
150  }
151 #endif
152 
153  ahrs_finv_last_stamp = stamp;
154 }

216 static void compute_body_orientation_and_rates(void)
217 {
218  if (ahrs_finv_output_enabled) {
219  /* Compute LTP to BODY quaternion */
220  struct FloatQuat ltp_to_body_quat;
221  struct FloatQuat *body_to_imu_quat = orientationGetQuat_f(&ahrs_float_inv.body_to_imu);
222  float_quat_comp_inv(&ltp_to_body_quat, &ahrs_float_inv.state.quat, body_to_imu_quat);
223  /* Set state */
224  stateSetNedToBodyQuat_f(&ltp_to_body_quat);
225 
226  /* compute body rates */
227  struct FloatRates body_rate;
228  RATES_DIFF(body_rate, ahrs_float_inv.cmd.rates, ahrs_float_inv.state.bias);
229  struct FloatRMat *body_to_imu_rmat = orientationGetRMat_f(&ahrs_float_inv.body_to_imu);
230  float_rmat_transp_ratemult(&body_rate, body_to_imu_rmat, &body_rate);
231  stateSetBodyRates_f(&body_rate);
232 
233  }
