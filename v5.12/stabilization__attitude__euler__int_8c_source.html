180 void stabilization_attitude_set_earth_cmd_i(struct Int32Vect2 *cmd, int32_t heading)
181 {
182  /* Rotate horizontal commands to body frame by psi */
183  int32_t psi = stateGetNedToBodyEulers_i()->psi;
184  int32_t s_psi, c_psi;
185  PPRZ_ITRIG_SIN(s_psi, psi);
186  PPRZ_ITRIG_COS(c_psi, psi);
187  stab_att_sp_euler.phi = (-s_psi * cmd->x + c_psi * cmd->y) >> INT32_TRIG_FRAC;
188  stab_att_sp_euler.theta = -(c_psi * cmd->x + s_psi * cmd->y) >> INT32_TRIG_FRAC;
189  stab_att_sp_euler.psi = heading;
190 }

197 void stabilization_attitude_run(bool in_flight)
198 {
199 
200  /* update reference */
201 #if USE_ATTITUDE_REF
202  attitude_ref_euler_int_update(&att_ref_euler_i, &stab_att_sp_euler);
203 #else
204  INT32_EULERS_LSHIFT(att_ref_euler_i.euler, stab_att_sp_euler, (REF_ANGLE_FRAC - INT32_ANGLE_FRAC));
205  INT_RATES_ZERO(att_ref_euler_i.rate);
206  INT_RATES_ZERO(att_ref_euler_i.accel);
207 #endif
208 
209  /* compute feedforward command */
210  stabilization_att_ff_cmd[COMMAND_ROLL] =
211  OFFSET_AND_ROUND(stabilization_gains.dd.x * att_ref_euler_i.accel.p, 5);
212  stabilization_att_ff_cmd[COMMAND_PITCH] =
213  OFFSET_AND_ROUND(stabilization_gains.dd.y * att_ref_euler_i.accel.q, 5);
214  stabilization_att_ff_cmd[COMMAND_YAW] =
215  OFFSET_AND_ROUND(stabilization_gains.dd.z * att_ref_euler_i.accel.r, 5);
216 
217  /* compute feedback command */
218  /* attitude error */
219  const struct Int32Eulers att_ref_scaled = {
220  OFFSET_AND_ROUND(att_ref_euler_i.euler.phi, (REF_ANGLE_FRAC - INT32_ANGLE_FRAC)),
221  OFFSET_AND_ROUND(att_ref_euler_i.euler.theta, (REF_ANGLE_FRAC - INT32_ANGLE_FRAC)),
222  OFFSET_AND_ROUND(att_ref_euler_i.euler.psi, (REF_ANGLE_FRAC - INT32_ANGLE_FRAC))
223  };
224  struct Int32Eulers att_err;
225  struct Int32Eulers *ltp_to_body_euler = stateGetNedToBodyEulers_i();
226  EULERS_DIFF(att_err, att_ref_scaled, (*ltp_to_body_euler));
227  INT32_ANGLE_NORMALIZE(att_err.psi);
228 
229  if (in_flight) {
230  /* update integrator */
231  EULERS_ADD(stabilization_att_sum_err, att_err);
232  EULERS_BOUND_CUBE(stabilization_att_sum_err, -MAX_SUM_ERR, MAX_SUM_ERR);
233  } else {
234  INT_EULERS_ZERO(stabilization_att_sum_err);
235  }
236 
237  /* rate error */
238  const struct Int32Rates rate_ref_scaled = {
239  OFFSET_AND_ROUND(att_ref_euler_i.rate.p, (REF_RATE_FRAC - INT32_RATE_FRAC)),
240  OFFSET_AND_ROUND(att_ref_euler_i.rate.q, (REF_RATE_FRAC - INT32_RATE_FRAC)),
241  OFFSET_AND_ROUND(att_ref_euler_i.rate.r, (REF_RATE_FRAC - INT32_RATE_FRAC))
242  };
243  struct Int32Rates rate_err;
244  struct Int32Rates *body_rate = stateGetBodyRates_i();
245  RATES_DIFF(rate_err, rate_ref_scaled, (*body_rate));
246 
247  /* PID */
248  stabilization_att_fb_cmd[COMMAND_ROLL] =
249  stabilization_gains.p.x * att_err.phi +
250  stabilization_gains.d.x * rate_err.p +
251  OFFSET_AND_ROUND2((stabilization_gains.i.x * stabilization_att_sum_err.phi), 10);
252 
253  stabilization_att_fb_cmd[COMMAND_PITCH] =
254  stabilization_gains.p.y * att_err.theta +
255  stabilization_gains.d.y * rate_err.q +
256  OFFSET_AND_ROUND2((stabilization_gains.i.y * stabilization_att_sum_err.theta), 10);
257 
258  stabilization_att_fb_cmd[COMMAND_YAW] =
259  stabilization_gains.p.z * att_err.psi +
260  stabilization_gains.d.z * rate_err.r +
261  OFFSET_AND_ROUND2((stabilization_gains.i.z * stabilization_att_sum_err.psi), 10);
262 
263 
264  /* with P gain of 100, att_err of 180deg (3.14 rad)
265  * fb cmd: 100 * 3.14 * 2^12 / 2^CMD_SHIFT = 628
266  * max possible command is 9600
267  */
268 #define CMD_SHIFT 11
269 
270  /* sum feedforward and feedback */
271  stabilization_cmd[COMMAND_ROLL] =
272  OFFSET_AND_ROUND((stabilization_att_fb_cmd[COMMAND_ROLL] + stabilization_att_ff_cmd[COMMAND_ROLL]), CMD_SHIFT);
273 
274  stabilization_cmd[COMMAND_PITCH] =
275  OFFSET_AND_ROUND((stabilization_att_fb_cmd[COMMAND_PITCH] + stabilization_att_ff_cmd[COMMAND_PITCH]), CMD_SHIFT);
276 
277  stabilization_cmd[COMMAND_YAW] =
278  OFFSET_AND_ROUND((stabilization_att_fb_cmd[COMMAND_YAW] + stabilization_att_ff_cmd[COMMAND_YAW]), CMD_SHIFT);
279 
280  /* bound the result */
281  BoundAbs(stabilization_cmd[COMMAND_ROLL], MAX_PPRZ);
282  BoundAbs(stabilization_cmd[COMMAND_PITCH], MAX_PPRZ);
283  BoundAbs(stabilization_cmd[COMMAND_YAW], MAX_PPRZ);
284
