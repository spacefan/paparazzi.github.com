124 void *sonar_bebop_read(void *data __attribute__((unused)))
125 {
126  while (true) {
127 #ifndef SITL
128  uint16_t i;
129 
130  /* Start ADC and send sonar output */
131  adc_enable(&adc0, 1);
132  sonar_bebop_spi_t.status = SPITransDone;
133  sonar_bebop_spi_t.output_buf = sonar_bebop_spi_d[mode];
134  spi_submit(&spi0, &sonar_bebop_spi_t);
135  while (sonar_bebop_spi_t.status != SPITransSuccess);
136  adc_read(&adc0, adc_buffer, SONAR_BEBOP_ADC_BUFFER_SIZE);
137  adc_enable(&adc0, 0);
138 
139  /* Find the peaks */
140  uint16_t start_send = 0;
141  uint16_t stop_send = 0;
142  uint16_t first_peak = 0;
143  uint16_t lowest_value = SONAR_BEBOP_ADC_MAX_VALUE;
144  uint16_t peak_value = 0;
145 
146  for (i = 0; i < SONAR_BEBOP_ADC_BUFFER_SIZE; i++) {
147  uint16_t adc_val = adc_buffer[i] >> 4;
148  if (start_send == 0 && adc_val == SONAR_BEBOP_ADC_MAX_VALUE) {
149  start_send = i;
150  } else if (start_send != 0 && stop_send == 0 && adc_val != SONAR_BEBOP_ADC_MAX_VALUE) {
151  stop_send = i - 1;
152  i += SONAR_BEBOP_PEAK_THRESHOLD;
153  continue;
154  }
155 
156  if (start_send != 0 && stop_send != 0 && first_peak == 0 && adc_val < lowest_value) {
157  lowest_value = adc_val; // find trough from initial broadcast signal
158  } else if (start_send != 0 && stop_send != 0 && adc_val > lowest_value + 100 && adc_val > peak_value) {
159  first_peak = i;
160  peak_value = adc_val;
161  }
162  }
163 
164  /* Calculate the distance from the peeks */
165  uint16_t diff = stop_send - start_send;
166  float peak_distance;
167  if (diff > 250) {
168  peak_distance = 0;
169  } else {
170  peak_distance = first_peak - (stop_send - diff / 2);
171  }
172 
173  sonar_bebop.distance = update_median_filter_f(&sonar_filt, peak_distance * SONAR_BEBOP_INX_DIFF_TO_DIST);
174 
175  // set sonar pulse mode for next pulse based on altitude
176  if (mode == 0 && sonar_bebop.distance > SONAR_BEBOP_TRANSITION_LOW_TO_HIGH) {
177  if (++pulse_transition_counter > SONAR_BEBOP_TRANSITION_COUNT) {
178  mode = 1;
179  pulse_transition_counter = 0;
180  }
181  } else if (mode == 1 && sonar_bebop.distance < SONAR_BEBOP_TRANSITION_HIGH_TO_LOW) {
182  if (++pulse_transition_counter > SONAR_BEBOP_TRANSITION_COUNT) {
183  mode = 0;
184  pulse_transition_counter = 0;
185  }
186  } else {
187  pulse_transition_counter = 0;
188  }
189 
190 #else // SITL
191  sonar_bebop.distance = stateGetPositionEnu_f()->z;
192  Bound(sonar_bebop.distance, 0.1f, 7.0f);
193  uint16_t peak_distance = 1;
194 #endif // SITL
195 
196  if (peak_distance > 0) {
197  // Send ABI message
198  AbiSendMsgAGL(AGL_SONAR_ADC_ID, sonar_bebop.distance);
199 #ifdef SENSOR_SYNC_SEND_SONAR
200  // Send Telemetry report
201  DOWNLINK_SEND_SONAR(DefaultChannel, DefaultDevice, &sonar_bebop.meas, &sonar_bebop.distance);
202 #endif
203  }
204  }
205  usleep(10000); //100Hz
206  return NULL;
207 }
