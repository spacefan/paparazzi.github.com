103 void imu_umarim_event(void)
104 {
105  uint32_t now_ts = get_sys_time_usec();
106 
107  // If the itg3200 I2C transaction has succeeded: convert the data
108  itg3200_event(&imu_umarim.itg);
109  if (imu_umarim.itg.data_available) {
110  RATES_COPY(imu.gyro_unscaled, imu_umarim.itg.data.rates);
111  imu_umarim.itg.data_available = false;
112  imu_scale_gyro(&imu);
113  AbiSendMsgIMU_GYRO_INT32(IMU_BOARD_ID, now_ts, &imu.gyro);
114  }
115 
116  // If the adxl345 I2C transaction has succeeded: convert the data
117  adxl345_i2c_event(&imu_umarim.adxl);
118  if (imu_umarim.adxl.data_available) {
119  VECT3_ASSIGN(imu.accel_unscaled, imu_umarim.adxl.data.vect.y, -imu_umarim.adxl.data.vect.x,
120  imu_umarim.adxl.data.vect.z);
121  imu_umarim.adxl.data_available = false;
122  imu_scale_accel(&imu);
123  AbiSendMsgIMU_ACCEL_INT32(IMU_BOARD_ID, now_ts, &imu.accel);
124  }
125 }
