44 void gps_sim_hitl_event(void)
45 {
46  if (SysTimeTimer(gps_sim_hitl_timer) > 100000) {
47  SysTimeTimerStart(gps_sim_hitl_timer);
48  gps.last_msg_ticks = sys_time.nb_sec_rem;
49  gps.last_msg_time = sys_time.nb_sec;
50  if (state.ned_initialized_i) {
51  if (!autopilot_in_flight()) {
52  struct Int32Vect2 zero_vector;
53  INT_VECT2_ZERO(zero_vector);
54  gh_set_ref(zero_vector, zero_vector, zero_vector);
55  INT_VECT2_ZERO(guidance_h.ref.pos);
56  INT_VECT2_ZERO(guidance_h.ref.speed);
57  INT_VECT2_ZERO(guidance_h.ref.accel);
58  gv_set_ref(0, 0, 0);
59  guidance_v_z_ref = 0;
60  guidance_v_zd_ref = 0;
61  guidance_v_zdd_ref = 0;
62  }
63  struct NedCoor_i ned_c;
64  ned_c.x = guidance_h.ref.pos.x * INT32_POS_OF_CM_DEN / INT32_POS_OF_CM_NUM;
65  ned_c.y = guidance_h.ref.pos.y * INT32_POS_OF_CM_DEN / INT32_POS_OF_CM_NUM;
66  ned_c.z = guidance_v_z_ref * INT32_POS_OF_CM_DEN / INT32_POS_OF_CM_NUM;
67  ecef_of_ned_point_i(&gps.ecef_pos, &state.ned_origin_i, &ned_c);
68  gps.lla_pos.alt = state.ned_origin_i.lla.alt - ned_c.z;
69  gps.hmsl = state.ned_origin_i.hmsl - ned_c.z;
70  ned_c.x = guidance_h.ref.speed.x * INT32_SPEED_OF_CM_S_DEN / INT32_SPEED_OF_CM_S_NUM;
71  ned_c.y = guidance_h.ref.speed.y * INT32_SPEED_OF_CM_S_DEN / INT32_SPEED_OF_CM_S_NUM;
72  ned_c.z = guidance_v_zd_ref * INT32_SPEED_OF_CM_S_DEN / INT32_SPEED_OF_CM_S_NUM;
73  ecef_of_ned_vect_i(&gps.ecef_vel, &state.ned_origin_i, &ned_c);
74  gps.fix = GPS_FIX_3D;
75  gps.last_3dfix_ticks = sys_time.nb_sec_rem;
76  gps.last_3dfix_time = sys_time.nb_sec;
77  gps_available = true;
78  }
79  else {
80  struct Int32Vect2 zero_vector;
81  INT_VECT2_ZERO(zero_vector);
82  gh_set_ref(zero_vector, zero_vector, zero_vector);
83  gv_set_ref(0, 0, 0);
84  }
85 
86  // publish gps data
87  uint32_t now_ts = get_sys_time_usec();
88  gps.last_msg_ticks = sys_time.nb_sec_rem;
89  gps.last_msg_time = sys_time.nb_sec;
90  if (gps.fix == GPS_FIX_3D) {
91  gps.last_3dfix_ticks = sys_time.nb_sec_rem;
92  gps.last_3dfix_time = sys_time.nb_sec;
93  }
94  AbiSendMsgGPS(GPS_SIM_ID, now_ts, &gps);
95  }
96 }
