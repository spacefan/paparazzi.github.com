62 uint32_t gec_bytes_to_counter(uint8_t *bytes)
63 {
64  // assume big endian
65  uint32_t x = (bytes[3] << 24) + (bytes[2] << 16) + (bytes[1] << 8) + bytes[0];
66  // now check the appropriate endiannes
67  /* ... for Linux */
68 #if defined(__linux__) || defined(__CYGWIN__)
69  return ntohl(x);
70  /* ... generic big-endian fallback code */
71 #elif defined(__BYTE_ORDER__) && __BYTE_ORDER__ == __ORDER_BIG_ENDIAN__
72  // the byte order is the same as for network
73  return x;
74  /* ... generic little-endian fallback code */
75 #elif defined(__BYTE_ORDER__) && __BYTE_ORDER__ == __ORDER_LITTLE_ENDIAN__
76  // do a byte swap
77  return htobe32(x);
78  /* ... couldn't determine endian-ness of the target platform */
79 #else
80 #pragma message "Please define __BYTE_ORDER__!"
81 #endif /* defined(__linux__) || ... */
82 }

87 void gec_counter_to_bytes(uint32_t n, uint8_t *bytes)
88 {
89  //first account for appropriate endiannes
90  /* ... for Linux */
91 #if defined(__linux__) || defined(__CYGWIN__)
92  n = htonl(n);
93  /* ... generic big-endian fallback code */
94 #elif defined(__BYTE_ORDER__) && __BYTE_ORDER__ == __ORDER_BIG_ENDIAN__
95  // the byte order is the same as for network
96  // do nothing
97  /* ... generic little-endian fallback code */
98 #elif defined(__BYTE_ORDER__) && __BYTE_ORDER__ == __ORDER_LITTLE_ENDIAN__
99  // do a byte swap
100  n = htobe32(n);
101  /* ... couldn't determine endian-ness of the target platform */
102 #else
103 #pragma message "Error: Please define __BYTE_ORDER__!"
104 #endif /* defined(__linux__) || ... */
105 
106  bytes[3] = (n >> 24) & 0xFF;
107  bytes[2] = (n >> 16) & 0xFF;
108  bytes[1] = (n >> 8) & 0xFF;
109  bytes[0] = n & 0xFF;
110 }

323 bool gec_encrypt_message(uint8_t *buf, uint8_t *payload_len)
324 {
325  // check payload length
326  if (*payload_len <= PPRZ_V2_MSG_ID) {
327  return false;
328  }
329  // check if the key is ready
330  if (!gec_tp.sts.tx_sym_key.ready) {
331  return false;
332  }
333 
334  // increment counter
335  uint32_t counter = gec_tp.sts.tx_sym_key.counter + 1;
336  // convert to bytes
337  uint8_t counter_as_bytes[4];
338  gec_counter_to_bytes(counter, counter_as_bytes);
339  // update nonce
340  memcpy(gec_tp.sts.tx_sym_key.nonce, counter_as_bytes, sizeof(uint32_t));
341 
342  // update intermediate fields
343  uint8_t auth[PPRZ_AUTH_LEN];
344  memcpy(auth, buf, PPRZ_AUTH_LEN);
345  uint8_t tag[PPRZ_MAC_LEN];
346  uint8_t ciphertext[TRANSPORT_PAYLOAD_LEN - PPRZ_CRYPTO_OVERHEAD];
347  uint32_t mlen = *payload_len - PPRZ_AUTH_LEN; // at least CLASS_BYTE and MSG_ID
348 
349  uint32_t res = Hacl_Chacha20Poly1305_aead_encrypt(ciphertext, tag,
350  &buf[PPRZ_AUTH_LEN], mlen, auth,
351  PPRZ_AUTH_LEN, gec_tp.sts.tx_sym_key.key, gec_tp.sts.tx_sym_key.nonce);
352 
353  if (res == 0) {
354  memset(buf, 0, TRANSPORT_PAYLOAD_LEN); // reset whole buffer
355  *payload_len = 0;
356  buf[PPRZ_GEC_IDX] = PPRZ_MSG_TYPE_ENCRYPTED; // CRYPTO_BYTE
357  *payload_len += 1;
358  memcpy(&buf[PPRZ_CNTR_IDX], counter_as_bytes, sizeof(uint32_t)); // counter
359  *payload_len += sizeof(uint32_t);
360  memcpy(&buf[PPRZ_AUTH_IDX], auth, PPRZ_AUTH_LEN); // auth data
361  *payload_len += PPRZ_AUTH_LEN;
362  memcpy(&buf[PPRZ_CIPH_IDX], ciphertext, mlen); // ciphertext
363  *payload_len += mlen;
364  memcpy(&buf[PPRZ_CIPH_IDX + mlen], tag, PPRZ_MAC_LEN); // tag
365  *payload_len += PPRZ_MAC_LEN;
366  gec_tp.sts.tx_sym_key.counter = counter; // update counter
367  return true;
368  } else {
369  gec_tp.sts.encrypt_err++;
370  return false;
371  }
372 
373 }

383 bool gec_decrypt_message(uint8_t *buf, volatile uint8_t *payload_len)
384 {
385  switch (buf[PPRZ_GEC_IDX]) {
386  case PPRZ_MSG_TYPE_PLAINTEXT:
387  // handle plaintext message
388  if (*payload_len < PPRZ_PLAINTEXT_MSG_MIN_LEN) {
389  return false;
390  }
391  // check if the message is in the whitelist
392  if (gec_is_in_the_whitelist(&gec_tp.whitelist,
393  buf[PPRZ_PLAINTEXT_MSG_ID_IDX])) {
394  // return the buffer minus the crypto byte
395  memmove(buf, &buf[1], *payload_len - 1);
396  return true;
397  }
398  return false;
399  break;
400  case PPRZ_MSG_TYPE_ENCRYPTED:
401  // attempt decryption
402  if (*payload_len < PPRZ_ENCRYPTED_MSG_MIN_LEN) {
403  return false;
404  }
405  if (!gec_tp.sts.rx_sym_key.ready) {
406  // the rx key is not ready yet
407  return false;
408  }
409  // first check the message counter
410  uint32_t counter = gec_bytes_to_counter(&buf[PPRZ_CNTR_IDX]);
411  // check against the saved counter
412  if (counter <= gec_tp.sts.rx_sym_key.counter) {
413  gec_tp.sts.rx_counter_err++;
414  return false;
415  }
416  // update nonce with 4 counter bytes
417  memcpy(gec_tp.sts.rx_sym_key.nonce, &buf[PPRZ_CNTR_IDX],
418  sizeof(uint32_t));
419  // update intermediate fields
420  uint8_t auth[PPRZ_AUTH_LEN];
421  memcpy(auth, &buf[PPRZ_AUTH_IDX], PPRZ_AUTH_LEN);
422  uint8_t tag[PPRZ_MAC_LEN];
423  memcpy(tag, &buf[*payload_len - PPRZ_MAC_LEN], PPRZ_MAC_LEN);
424  uint8_t ciphertext[TRANSPORT_PAYLOAD_LEN - PPRZ_CRYPTO_OVERHEAD];
425  // payload - AUTH_DATA(2) - TAG(16) - CRYPTO_BYTE(1) - COUNTER(4) = payload_len - 23
426  uint32_t mlen = *payload_len - PPRZ_CRYPTO_OVERHEAD - 1 - PPRZ_AUTH_LEN; // don't forget to count for CRYPTO_BYTE
427  memcpy(ciphertext, &buf[PPRZ_CIPH_IDX], mlen);
428  uint8_t plaintext[TRANSPORT_PAYLOAD_LEN - PPRZ_CRYPTO_OVERHEAD];
429 
430  // try decryption
431  uint32_t res = Hacl_Chacha20Poly1305_aead_decrypt(plaintext, ciphertext,
432  mlen, tag, auth,
433  PPRZ_AUTH_LEN, gec_tp.sts.rx_sym_key.key,
434  gec_tp.sts.rx_sym_key.nonce);
435 
436  // res == 0 means all good
437  if (res == 0) {
438  memset(buf, 0, TRANSPORT_PAYLOAD_LEN); // reset whole buffer
439  *payload_len = 0;
440  memcpy(buf, auth, PPRZ_AUTH_LEN); // copy auth data
441  *payload_len += PPRZ_AUTH_LEN;
442  memcpy(&buf[PPRZ_AUTH_LEN], plaintext, mlen); // copy plaintext
443  *payload_len += mlen;
444  gec_tp.sts.rx_sym_key.counter = counter; // update counter
445  return true;
446  } else {
447  gec_tp.sts.decrypt_err++;
448  return false;
449  }
450  break;
451  default:
452  break;
453  }
454  return false;
455 }

479 void gec_dl_event(void)
480 {
481  // strip PPRZ_STX, MSG_LEN, and CHEKSUM, return (CRYPTO_BYTE..MSG_PAYLOAD)
482  pprz_check_and_parse(&DOWNLINK_DEVICE.device, &gec_tp.pprz_tp,
483  gec_tp.pprz_tp.trans_rx.payload, (bool *) &gec_tp.trans_rx.msg_received);
484  // we have (CRYPTO_BYTE .. MSG_PAYLOAD) in
485  if (gec_tp.trans_rx.msg_received) { // self.rx.parse_byte(b)
486 
487  switch (gec_tp.sts.protocol_stage) {
488  case CRYPTO_OK:
489  // decrypt message
490  // if successfull return the message (sender_ID .. MSG_payload)
491  if (gec_decrypt_message(gec_tp.pprz_tp.trans_rx.payload,
492  &gec_tp.pprz_tp.trans_rx.payload_len)) {
493  // copy the buffer over
494  // NOTE/TODO: the real payload_len is at least one byte shorter
495  // but we can copy whole buffer anyway since we don't overflow
496  DatalinkFillDlBuffer(gec_tp.pprz_tp.trans_rx.payload,
497  gec_tp.pprz_tp.trans_rx.payload_len);
498  // pass to datalink
499  DlCheckAndParse(&DOWNLINK_DEVICE.device, &gec_tp.trans_tx, dl_buffer,
500  &dl_msg_available);
501  }
502  break;
503  case WAIT_MSG1:
504  // decrypt message (checks against the whitelist and removes the crypto byte)
505  // returns (SENDER_ID .. MSG_PAYLOAD
506  if (gec_decrypt_message(gec_tp.pprz_tp.trans_rx.payload,
507  &gec_tp.pprz_tp.trans_rx.payload_len)) {
508  gec_process_msg1(gec_tp.pprz_tp.trans_rx.payload);
509  } else {
510  }
511  break;
512  case WAIT_MSG3:
513  // decrypt message (checks against the whitelist and removes the crypto byte)
514  // returns (SENDER_ID .. MSG_PAYLOAD
515  if (gec_decrypt_message(gec_tp.pprz_tp.trans_rx.payload,
516  &gec_tp.pprz_tp.trans_rx.payload_len)) {
517  if (gec_process_msg3(gec_tp.pprz_tp.trans_rx.payload)) {
518  // if OK, we are ready for an ongoing communication
519  gec_tp.sts.protocol_stage = CRYPTO_OK;
520 #ifdef GEC_STATUS_LED
521  LED_ON(GEC_STATUS_LED);
522 #endif
523  }
524  }
525  break;
526  default:
527  // shouldn't be here unless something is really wrong
528  break;
529  }
530  // reset flag
531  gec_tp.trans_rx.msg_received = false;
532  }

544 void gec_process_msg1(uint8_t *buf)
545 {
546  if (SenderIdOfPprzMsg(buf) != 0) {
547  // process only messages from GCS
548  // log an error
549  return;
550  }
551 
552  // check message ID
553  if (IdOfPprzMsg(buf) != DL_KEY_EXCHANGE_GCS) {
554  gec_tp.sts.last_error = UNEXPECTED_MSG_ERROR;
555  return;
556  }
557 
558  // check message type
559  if (DL_KEY_EXCHANGE_GCS_msg_type(buf) != P_AE) {
560  gec_tp.sts.last_error = UNEXPECTED_MSG_TYPE_ERROR;
561  return;
562  }
563 
564  // check data length
565  if (DL_KEY_EXCHANGE_GCS_msg_data_length(buf) != PPRZ_KEY_LEN) {
566  gec_tp.sts.last_error = UNEXPECTED_MSG_DATA_ERROR;
567  return;
568  }
569 
570  // copy P_ae over
571  memcpy(&gec_tp.sts.their_public_ephemeral.pub,
572  DL_KEY_EXCHANGE_GCS_msg_data(buf), sizeof(struct gec_pubkey));
573  gec_tp.sts.their_public_ephemeral.ready = true;
574 
575  // 2. B generates ephemeral curve25519 key pair (Pbe, Qbe).
576  if (!gec_tp.sts.my_private_ephemeral.ready) {
577  gec_generate_ephemeral_keys(&gec_tp.sts.my_private_ephemeral);
578  }
579 
580  // 3. B computes the shared secret: z = scalar_multiplication(Qbe, Pae)
581  uint8_t z[32] = { 0 };
582  Hacl_Curve25519_crypto_scalarmult(z, gec_tp.sts.my_private_ephemeral.priv,
583  gec_tp.sts.their_public_ephemeral.pub);
584 
585  // 4. B uses the key derivation function kdf(z,1) to compute Kb || Sb,
586  // kdf(z,0) to compute Ka || Sa, and kdf(z,2) to compute Kclient || Sclient.
587  gec_derive_key_material(&gec_tp.sts, z);
588 
589  // 5. B computes the ed25519 signature: sig = signQb(Pbe || Pae)
590  uint8_t sig[PPRZ_SIGN_LEN] = { 0 };
591  uint8_t pbe_concat_p_ae[PPRZ_KEY_LEN * 2] = { 0 };
592  memcpy(pbe_concat_p_ae, &gec_tp.sts.my_private_ephemeral.pub, PPRZ_KEY_LEN);
593  memcpy(&pbe_concat_p_ae[PPRZ_KEY_LEN], &gec_tp.sts.their_public_ephemeral.pub,
594  PPRZ_KEY_LEN);
595  Hacl_Ed25519_sign(sig, gec_tp.sts.my_private_key.priv, pbe_concat_p_ae,
596  PPRZ_KEY_LEN * 2);
597 
598  // 6. B computes and sends the message Pbe || Ekey=Kb,IV=Sb||zero(sig)
599  uint8_t msg_data[PPRZ_KEY_LEN + PPRZ_SIGN_LEN + PPRZ_MAC_LEN] = { 0 };
600  memcpy(msg_data, &gec_tp.sts.my_private_ephemeral.pub, PPRZ_KEY_LEN);
601 
602  if (Hacl_Chacha20Poly1305_aead_encrypt(&msg_data[PPRZ_KEY_LEN],
603  &msg_data[PPRZ_KEY_LEN + PPRZ_SIGN_LEN], sig, PPRZ_SIGN_LEN, NULL, 0,
604  gec_tp.sts.tx_sym_key.key, gec_tp.sts.tx_sym_key.nonce)) {
605  gec_tp.sts.last_error = MSG1_ENCRYPT_ERROR;
606  return;
607  }
608 
609  // now we have to manually construct the message
610  // CRYPTO BYTE
611  // source_id
612  // sender_id
613  // class id
614  // msg id
615  // msg_type
616  // msg_data
617  gec_tp.tx_msg[PPRZ_GEC_IDX] = PPRZ_MSG_TYPE_PLAINTEXT;
618  gec_tp.tx_msg_idx = 1;
619  gec_tp.tx_msg[gec_tp.tx_msg_idx] = AC_ID;
620  gec_tp.tx_msg_idx++;
621  gec_tp.tx_msg[gec_tp.tx_msg_idx] = 0;
622  gec_tp.tx_msg_idx++;
623  gec_tp.tx_msg[gec_tp.tx_msg_idx] = 1; // telemetry
624  gec_tp.tx_msg_idx++;
625  gec_tp.tx_msg[gec_tp.tx_msg_idx] = 239;
626  gec_tp.tx_msg_idx++;
627  gec_tp.tx_msg[gec_tp.tx_msg_idx] = P_BE;
628  gec_tp.tx_msg_idx++;
