298 bool calc_fast9_lukas_kanade(struct opticflow_t *opticflow, struct image_t *img,
299  struct opticflow_result_t *result)
300 {
301  if (opticflow->just_switched_method) {
302  // Create the image buffers
303  image_create(&opticflow->img_gray, img->w, img->h, IMAGE_GRAYSCALE);
304  image_create(&opticflow->prev_img_gray, img->w, img->h, IMAGE_GRAYSCALE);
305 
306  // Set the previous values
307  opticflow->got_first_img = false;
308 
309  // Init median filters with zeros
310  InitMedianFilterVect3Float(vel_filt, MEDIAN_DEFAULT_SIZE);
311  }
312 
313  // Convert image to grayscale
314  image_to_grayscale(img, &opticflow->img_gray);
315  opticflow->img_gray.ts = img->ts;
316  opticflow->img_gray.pprz_ts = img->pprz_ts;
317  opticflow->img_gray.eulers = img->eulers;
318 
319  // Copy to previous image if not set
320  if (!opticflow->got_first_img) {
321  image_copy(&opticflow->img_gray, &opticflow->prev_img_gray);
322  opticflow->got_first_img = true;
323  return false;
324  }
325 
326  // variables for linear flow fit:
327  float error_threshold;
328  int n_iterations_RANSAC, n_samples_RANSAC, success_fit;
329  struct linear_flow_fit_info fit_info;
330 
331  // Update FPS for information
332  float dt = timeval_diff(&(opticflow->prev_img_gray.ts), &(img->ts)) / 1000.f;
333  if (dt > 1e-5) {
334  result->fps = 1 / dt;
335  } else {
336  return false;
337  }
338 
339  // *************************************************************************************
340  // Corner detection
341  // *************************************************************************************
342 
343  // if feature_management is selected and tracked corners drop below a threshold, redetect
344  if ((opticflow->feature_management) && (result->corner_cnt < opticflow->max_track_corners / 2)) {
345  manage_flow_features(img, opticflow, result);
346  } else if (!opticflow->feature_management) {
347  // needs to be set to 0 because result is now static
348  result->corner_cnt = 0;
349 
350  if (opticflow->corner_method == EXHAUSTIVE_FAST) {
351  // FAST corner detection
352  // TODO: There is something wrong with fast9_detect destabilizing FPS. This problem is reduced with putting min_distance
353  // to 0 (see defines), however a more permanent solution should be considered
354  fast9_detect(&opticflow->prev_img_gray, opticflow->fast9_threshold, opticflow->fast9_min_distance,
355  opticflow->fast9_padding, opticflow->fast9_padding, &result->corner_cnt,
356  &opticflow->fast9_rsize,
357  &opticflow->fast9_ret_corners,
358  NULL);
359 
360  } else if (opticflow->corner_method == ACT_FAST) {
361  // ACT-FAST corner detection:
362  act_fast(&opticflow->prev_img_gray, opticflow->fast9_threshold, &result->corner_cnt,
363  &opticflow->fast9_ret_corners, n_agents, n_time_steps,
364  opticflow->actfast_long_step, opticflow->actfast_short_step, opticflow->actfast_min_gradient,
365  opticflow->actfast_gradient_method);
366  }
367 
368  // Adaptive threshold
369  if (opticflow->fast9_adaptive) {
370 
371  // This works well for exhaustive FAST, but drives the threshold to the minimum for ACT-FAST:
372  // Decrease and increase the threshold based on previous values
373  if (result->corner_cnt < 40) { // TODO: Replace 40 with OPTICFLOW_MAX_TRACK_CORNERS / 2
374  // make detections easier:
375  if (opticflow->fast9_threshold > FAST9_LOW_THRESHOLD) {
376  opticflow->fast9_threshold--;
377  }
378 
379  if (opticflow->corner_method == ACT_FAST) {
380  n_time_steps++;
381  n_agents++;
382  }
383 
384  } else if (result->corner_cnt > OPTICFLOW_MAX_TRACK_CORNERS * 2 && opticflow->fast9_threshold < FAST9_HIGH_THRESHOLD) {
385  opticflow->fast9_threshold++;
386  if (opticflow->corner_method == ACT_FAST && n_time_steps > 5 && n_agents > 10) {
387  n_time_steps--;
388  n_agents--;
389  }
390  }
391  }
392  }
393 
394 #if OPTICFLOW_SHOW_CORNERS
395  image_show_points(img, opticflow->fast9_ret_corners, result->corner_cnt);
396 #endif
397 
398  // Check if we found some corners to track
399  if (result->corner_cnt < 1) {
400  // Clear the result otherwise the previous values will be returned for this frame too
401  VECT3_ASSIGN(result->vel_cam, 0, 0, 0);
402  VECT3_ASSIGN(result->vel_body, 0, 0, 0);
403  result->div_size = 0; result->divergence = 0;
404  result->noise_measurement = 5.0;
405 
406  image_switch(&opticflow->img_gray, &opticflow->prev_img_gray);
407  return false;
408  }
409 
410  // *************************************************************************************
411  // Corner Tracking
412  // *************************************************************************************
413 
414  // Execute a Lucas Kanade optical flow
415  result->tracked_cnt = result->corner_cnt;
416  struct flow_t *vectors = opticFlowLK(&opticflow->img_gray, &opticflow->prev_img_gray, opticflow->fast9_ret_corners,
417  &result->tracked_cnt,
418  opticflow->window_size / 2, opticflow->subpixel_factor, opticflow->max_iterations,
419  opticflow->threshold_vec, opticflow->max_track_corners, opticflow->pyramid_level);
420 
421 #if OPTICFLOW_SHOW_FLOW
422  image_show_flow(img, vectors, result->tracked_cnt, opticflow->subpixel_factor);
423 #endif
424 
425  static int n_samples = 100;
426  // Estimate size divergence:
427  if (SIZE_DIV) {
428  result->div_size = get_size_divergence(vectors, result->tracked_cnt, n_samples);// * result->fps;
429  } else {
430  result->div_size = 0.0f;
431  }
432 
433  if (LINEAR_FIT) {
434  // Linear flow fit (normally derotation should be performed before):
435  error_threshold = 10.0f;
436  n_iterations_RANSAC = 20;
437  n_samples_RANSAC = 5;
438  success_fit = analyze_linear_flow_field(vectors, result->tracked_cnt, error_threshold, n_iterations_RANSAC,
439  n_samples_RANSAC, img->w, img->h, &fit_info);
440 
441  if (!success_fit) {
442  fit_info.divergence = 0.0f;
443  fit_info.surface_roughness = 0.0f;
444  }
445 
446  result->divergence = fit_info.divergence;
447  result->surface_roughness = fit_info.surface_roughness;
448  } else {
449  result->divergence = 0.0f;
450  result->surface_roughness = 0.0f;
451  }
452 
453  // Get the median flow
454  qsort(vectors, result->tracked_cnt, sizeof(struct flow_t), cmp_flow);
455  if (result->tracked_cnt == 0) {
456  // We got no flow
457  result->flow_x = 0;
458  result->flow_y = 0;
459 
460  free(vectors);
461  image_switch(&opticflow->img_gray, &opticflow->prev_img_gray);
462  return false;
463  } else if (result->tracked_cnt % 2) {
464  // Take the median point
465  result->flow_x = vectors[result->tracked_cnt / 2].flow_x;
466  result->flow_y = vectors[result->tracked_cnt / 2].flow_y;
467  } else {
468  // Take the average of the 2 median points
469  result->flow_x = (vectors[result->tracked_cnt / 2 - 1].flow_x + vectors[result->tracked_cnt / 2].flow_x) / 2.f;
470  result->flow_y = (vectors[result->tracked_cnt / 2 - 1].flow_y + vectors[result->tracked_cnt / 2].flow_y) / 2.f;
471  }
472 
473  // TODO scale flow to rad/s here
474 
475  // Flow Derotation
476  float diff_flow_x = 0.f;
477  float diff_flow_y = 0.f;
478 
479  if (opticflow->derotation && result->tracked_cnt > 5) {
480  diff_flow_x = (opticflow->img_gray.eulers.phi - opticflow->prev_img_gray.eulers.phi) * OPTICFLOW_FX;
481  diff_flow_y = (opticflow->img_gray.eulers.theta - opticflow->prev_img_gray.eulers.theta) * OPTICFLOW_FY;
482  /*diff_flow_x = (cam_state->rates.p) / result->fps * img->w /
483  OPTICFLOW_FOV_W;// * img->w / OPTICFLOW_FOV_W;
484  diff_flow_y = (cam_state->rates.q) / result->fps * img->h /
485  OPTICFLOW_FOV_H;// * img->h / OPTICFLOW_FOV_H;*/
486  }
487 
488  float rotation_threshold = M_PI / 180.0f;
489  if (fabs(opticflow->img_gray.eulers.phi - opticflow->prev_img_gray.eulers.phi) > rotation_threshold) {
490  result->flow_der_x = 0.0f;
491  result->flow_der_y = 0.0f;
492  } else {
493  result->flow_der_x = result->flow_x - diff_flow_x * opticflow->subpixel_factor *
494  opticflow->derotation_correction_factor_x;
495  result->flow_der_y = result->flow_y - diff_flow_y * opticflow->subpixel_factor *
496  opticflow->derotation_correction_factor_y;
497  }
498 
499  // Velocity calculation
500  // Right now this formula is under assumption that the flow only exist in the center axis of the camera.
501  // TODO Calculate the velocity more sophisticated, taking into account the drone's angle and the slope of the ground plane.
502  result->vel_cam.x = (float)result->flow_der_x * result->fps * agl_dist_value_filtered /
503  (opticflow->subpixel_factor * OPTICFLOW_FX);
504  result->vel_cam.y = (float)result->flow_der_y * result->fps * agl_dist_value_filtered /
505  (opticflow->subpixel_factor * OPTICFLOW_FY);
506  result->vel_cam.z = result->divergence * result->fps * agl_dist_value_filtered;
507 
508  //Apply a median filter to the velocity if wanted
509  if (opticflow->median_filter == true) {
510  UpdateMedianFilterVect3Float(vel_filt, result->vel_cam);
511  }
512 
513  // Determine quality of noise measurement for state filter
514  //TODO develop a noise model based on groundtruth
515  //result->noise_measurement = 1 - (float)result->tracked_cnt / ((float)opticflow->max_track_corners * 1.25f);
516  result->noise_measurement = 0.25;
517 
518  // *************************************************************************************
519  // Next Loop Preparation
520  // *************************************************************************************
521  if (opticflow->feature_management) {
522  result->corner_cnt = result->tracked_cnt;
523  //get the new positions of the corners and the "residual" subpixel positions
524  for (uint16_t i = 0; i < result->tracked_cnt; i++) {
525  opticflow->fast9_ret_corners[i].x = (uint32_t)((vectors[i].pos.x + (float)vectors[i].flow_x) /
526  opticflow->subpixel_factor);
527  opticflow->fast9_ret_corners[i].y = (uint32_t)((vectors[i].pos.y + (float)vectors[i].flow_y) /
528  opticflow->subpixel_factor);
529  opticflow->fast9_ret_corners[i].x_sub = (uint16_t)((vectors[i].pos.x + vectors[i].flow_x) % opticflow->subpixel_factor);
530  opticflow->fast9_ret_corners[i].y_sub = (uint16_t)((vectors[i].pos.y + vectors[i].flow_y) % opticflow->subpixel_factor);
531  opticflow->fast9_ret_corners[i].count = vectors[i].pos.count;
532  }
533  }
534  free(vectors);
535  image_switch(&opticflow->img_gray, &opticflow->prev_img_gray);
536 
537  return true;
538 }

557  opticflow->fast9_ret_corners[c1].x = opticflow->fast9_ret_corners[result->corner_cnt - 1].x;

565  exists = true;

571  if (!exists) { c1++; }
572  }

605  roi[0] = (region_count[i][1] % root_regions) * (img->w / root_regions);
606  roi[1] = (region_count[i][1] / root_regions) * (img->h / root_regions);
607  roi[2] = roi[0] + (img->w / root_regions);
608  roi[3] = roi[1] + (img->h / root_regions);

611  uint16_t new_count = 0;
612 
613  fast9_detect(&opticflow->prev_img_gray, opticflow->fast9_threshold, opticflow->fast9_min_distance,
614  opticflow->fast9_padding, opticflow->fast9_padding, &new_count,
615  &opticflow->fast9_rsize, &new_corners, roi);
616 
617  // check that no identified points already exist in list
618  for (uint16_t j = 0; j < new_count; j++) {
619  bool exists = false;
620  for (uint16_t k = 0; k < result->corner_cnt; k++) {
621  if (abs((int16_t)new_corners[j].x - (int16_t)opticflow->fast9_ret_corners[k].x) < (int16_t)opticflow->fast9_min_distance
622  && abs((int16_t)new_corners[j].y - (int16_t)opticflow->fast9_ret_corners[k].y) < (int16_t)
623  opticflow->fast9_min_distance) {
624  exists = true;
625  break;
626  }
627  }
628  if (!exists) {
629  opticflow->fast9_ret_corners[result->corner_cnt].x = new_corners[j].x;
630  opticflow->fast9_ret_corners[result->corner_cnt].y = new_corners[j].y;
631  opticflow->fast9_ret_corners[result->corner_cnt].count = 0;
632  opticflow->fast9_ret_corners[result->corner_cnt].x_sub = 0;
633  opticflow->fast9_ret_corners[result->corner_cnt].y_sub = 0;
634  result->corner_cnt++;
635 
636  if (result->corner_cnt >= opticflow->fast9_rsize) {
637  break;
638  }
639  }
640  }
641 
642  free(new_corners);
643  }
644  for (uint16_t i = 0; i < opticflow->fast9_num_regions; i++) {
645  free(region_count[i]);
646  }
647  free(region_count);
648  }

796  result->vel_cam.y = edgeflow.flow_y * fps_y * agl_dist_value_filtered * OPTICFLOW_FY / RES;
797  result->vel_cam.z = result->divergence * fps_x * agl_dist_value_filtered;

803 
804  result->noise_measurement = 0.2;

811 
812  // Free alloc'd variables

814  free(displacement.y);
