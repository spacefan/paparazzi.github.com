222 void nps_fdm_init(double dt)
223 {
224  fdm.init_dt = dt; // JSBsim specific
225  fdm.curr_dt = dt; // JSBsim specific
226  fdm.nan_count = 0; // JSBsim specific
227 
228 #ifdef NPS_ACTUATOR_TIME_CONSTANTS
229  // Set up low-pass filter to simulate delayed actuator response
230  const float tau[NPS_COMMANDS_NB] = NPS_ACTUATOR_TIME_CONSTANTS;
231  for(int i=0; i<NPS_COMMANDS_NB; i++) {
232  init_first_order_low_pass(&gazebo_actuators.lowpass[i], tau[i], dt, 0.f);
233  }
234 #ifdef NPS_ACTUATOR_MAX_ANGULAR_MOMENTUM
235  // Set up high-pass filter to simulate spinup torque
236  const float Iwmax[NPS_COMMANDS_NB] = NPS_ACTUATOR_MAX_ANGULAR_MOMENTUM;
237  for(int i=0; i<NPS_COMMANDS_NB; i++) {
238  init_first_order_high_pass(&gazebo_actuators.highpass[i], tau[i], dt, 0.f);
239  gazebo_actuators.max_ang_momentum[i] = Iwmax[i];
240  }
241 #endif
242 #endif
243 }

255 {
256  // Initialize Gazebo if req'd.

263  init_gazebo();

271  gazebo_initialized = true;
272  }
273 
274  // Update the simulation for a single timestep.
275  gazebo::runWorld(model->GetWorld(), 1);
276  gazebo::sensors::run_once();
277  gazebo_write(act_commands, commands_nb);
278  gazebo_read();
279 #if NPS_SIMULATE_VIDEO
280  gazebo_read_video();
281 #endif
282 #if NPS_SIMULATE_LASER_RANGE_ARRAY

328 static void init_gazebo(void)
329 {
330  string gazebo_home = "/conf/simulator/gazebo/";
331  string pprz_home(getenv("PAPARAZZI_HOME"));
332  string gazebodir = pprz_home + gazebo_home;
333  cout << "Gazebo directory: " << gazebodir << endl;
334 
335  if (!gazebo::setupServer(0, NULL)) {
336  cout << "Failed to start Gazebo, exiting." << endl;
337  std::exit(-1);
338  }
339 
340  cout << "Add Paparazzi model path: " << gazebodir + "models/" << endl;
341  gazebo::common::SystemPaths::Instance()->AddModelPaths(
342  gazebodir + "models/");
343 
344  // get vehicles
345  cout << "Load vehicle: " << gazebodir + "models/" + NPS_GAZEBO_AC_NAME + "/" + NPS_GAZEBO_AC_NAME + ".sdf" << endl;
346  sdf::SDFPtr vehicle_sdf(new sdf::SDF());
347  sdf::init(vehicle_sdf);
348  sdf::readFile(gazebodir + "models/" + NPS_GAZEBO_AC_NAME + "/" + NPS_GAZEBO_AC_NAME + ".sdf", vehicle_sdf);
349 
350  // add or set up sensors before the vehicle gets loaded
351  // laser range array
352 #if NPS_SIMULATE_LASER_RANGE_ARRAY
353  vehicle_sdf->Root()->GetFirstElement()->AddElement("include")->GetElement("uri")->Set("model://range_sensors");
354  sdf::ElementPtr range_joint = vehicle_sdf->Root()->GetFirstElement()->AddElement("joint");
355  range_joint->GetAttribute("name")->Set("range_sensor_joint");
356  range_joint->GetAttribute("type")->Set("fixed");
357  range_joint->GetElement("parent")->Set("chassis");
358  range_joint->GetElement("child")->Set("range_sensors::base");
359 #endif
360  // bebop front camera
361 #ifdef NPS_SIMULATE_MT9F002
362  sdf::ElementPtr link = vehicle_sdf->Root()->GetFirstElement()->GetElement("link");
363  while(link) {
364  if(link->Get<string>("name") == "front_camera") {
365  int w = link->GetElement("sensor")->GetElement("camera")->GetElement("image")->GetElement("width")->Get<int>();
366  link->GetElement("sensor")->GetElement("camera")->GetElement("image")->GetElement("width")->Set(w * MT9F002_OUTPUT_SCALER);
367  int h = link->GetElement("sensor")->GetElement("camera")->GetElement("image")->GetElement("height")->Get<int>();
368  link->GetElement("sensor")->GetElement("camera")->GetElement("image")->GetElement("height")->Set(h * MT9F002_OUTPUT_SCALER);
369  int env = link->GetElement("sensor")->GetElement("camera")->GetElement("lens")->GetElement("env_texture_size")->Get<int>();
370  link->GetElement("sensor")->GetElement("camera")->GetElement("lens")->GetElement("env_texture_size")->Set(env * MT9F002_OUTPUT_SCALER);
371  cout << "Applied MT9F002_OUTPUT_SCALER (=" << MT9F002_OUTPUT_SCALER << ") to " << link->Get<string>("name") << endl;
372  link->GetElement("sensor")->GetElement("update_rate")->Set(MT9F002_TARGET_FPS);
373  cout << "Applied MT9F002_TARGET_FPS (=" << MT9F002_TARGET_FPS << ") to " << link->Get<string>("name") << endl;
374  }
375  link = link->GetNextElement("link");
376  }
377 #endif // NPS_SIMULATE_MT9F002
378 
379 
380  // get world
381  cout << "Load world: " << gazebodir + "world/" + NPS_GAZEBO_WORLD << endl;
382  sdf::SDFPtr world_sdf(new sdf::SDF());
383  sdf::init(world_sdf);
384  sdf::readFile(gazebodir + "world/" + NPS_GAZEBO_WORLD, world_sdf);
385 
386  // add vehicles
387  world_sdf->Root()->GetFirstElement()->InsertElement(vehicle_sdf->Root()->GetFirstElement());
388 
389  world_sdf->Write(pprz_home + "/var/gazebo.world");
390 
391  gazebo::physics::WorldPtr world = gazebo::loadWorld(pprz_home + "/var/gazebo.world");
392  if (!world) {
393  cout << "Failed to open world, exiting." << endl;
394  std::exit(-1);
395  }
396 
397  cout << "Get pointer to aircraft: " << NPS_GAZEBO_AC_NAME << endl;
398  model = world->GetModel(NPS_GAZEBO_AC_NAME);
399  if (!model) {
400  cout << "Failed to find '" << NPS_GAZEBO_AC_NAME << "', exiting."
401  << endl;
402  std::exit(-1);
403  }
404 
405  // Initialize sensors
406  gazebo::sensors::run_once(true);
407  gazebo::sensors::run_threads();
408  gazebo::runWorld(world, 1);
409  cout << "Sensors initialized..." << endl;
410 
411  // activate collision sensor
412  gazebo::sensors::SensorManager *mgr = gazebo::sensors::SensorManager::Instance();
413  ct = static_pointer_cast < gazebo::sensors::ContactSensor > (mgr->GetSensor("contactsensor"));
414  ct->SetActive(true);
415 
416  // Overwrite motor directions as defined by motor_mixing
417 #ifdef MOTOR_MIXING_YAW_COEF
418  const double yaw_coef[] = MOTOR_MIXING_YAW_COEF;
419 
420  for (uint8_t i = 0; i < NPS_COMMANDS_NB; i++) {
421  gazebo_actuators.torques[i] = -fabs(gazebo_actuators.torques[i]) * yaw_coef[i] / fabs(yaw_coef[i]);
422  gazebo_actuators.max_ang_momentum[i] = -fabs(gazebo_actuators.max_ang_momentum[i]) * yaw_coef[i] / fabs(yaw_coef[i]);
423  }
424 #endif
425  cout << "Gazebo initialized successfully!" << endl;
426 }

436 static void gazebo_read(void)
437 {
438  static gazebo::math::Vector3 vel_prev;
439  static double time_prev;
440 
441  gazebo::physics::WorldPtr world = model->GetWorld();
442  gazebo::math::Pose pose = model->GetWorldPose(); // In LOCAL xyz frame
443  gazebo::math::Vector3 vel = model->GetWorldLinearVel();
444  gazebo::math::Vector3 ang_vel = model->GetWorldAngularVel();
445  gazebo::common::SphericalCoordinatesPtr sphere =
446  world->GetSphericalCoordinates();
447  gazebo::math::Quaternion local_to_global_quat(0, 0,
448  -sphere->HeadingOffset().Radian());
449 
450  /* Fill FDM struct */
451  fdm.time = world->GetSimTime().Double();
452 
453  // Find world acceleration by differentiating velocity
454  // model->GetWorldLinearAccel() does not seem to take the velocity_decay into account!
455  // Derivation of the velocity also follows the IMU implementation of Gazebo itself:
456  // https://bitbucket.org/osrf/gazebo/src/e26144434b932b4b6a760ddaa19cfcf9f1734748/gazebo/sensors/ImuSensor.cc?at=default&fileviewer=file-view-default#ImuSensor.cc-370
457  double dt = fdm.time - time_prev;
458  gazebo::math::Vector3 accel = (vel - vel_prev) / dt;
459  vel_prev = vel;
460  time_prev = fdm.time;
461 
462  // init_dt: unused
463  // curr_dt: unused
464  // on_ground: unused
465  // nan_count: unused
466 
467  /* position */
468  fdm.ecef_pos = to_pprz_ecef(
469  sphere->PositionTransform(pose.pos.Ign(),
470  gazebo::common::SphericalCoordinates::LOCAL,
471  gazebo::common::SphericalCoordinates::ECEF));
472  fdm.ltpprz_pos = to_pprz_ned(
473  sphere->PositionTransform(pose.pos.Ign(),
474  gazebo::common::SphericalCoordinates::LOCAL,
475  gazebo::common::SphericalCoordinates::GLOBAL));
476  fdm.lla_pos = to_pprz_lla(
477  sphere->PositionTransform(pose.pos.Ign(),
478  gazebo::common::SphericalCoordinates::LOCAL,
479  gazebo::common::SphericalCoordinates::SPHERICAL));
480  fdm.hmsl = pose.pos.z;
481  /* debug positions */
482  fdm.lla_pos_pprz = fdm.lla_pos; // Don't really care...
483  fdm.lla_pos_geod = fdm.lla_pos;
484  fdm.lla_pos_geoc = fdm.lla_pos;
485  fdm.agl = pose.pos.z; // TODO Measure with sensor
486 
487  /* velocity */
488  fdm.ecef_ecef_vel = to_pprz_ecef(
489  sphere->VelocityTransform(vel.Ign(),
490  gazebo::common::SphericalCoordinates::LOCAL,
491  gazebo::common::SphericalCoordinates::ECEF));
492  fdm.body_ecef_vel = to_pprz_body(pose.rot.RotateVectorReverse(vel)); // Note: unused
493  fdm.ltp_ecef_vel = to_pprz_ned(
494  sphere->VelocityTransform(vel.Ign(),
495  gazebo::common::SphericalCoordinates::LOCAL,
496  gazebo::common::SphericalCoordinates::GLOBAL));
497  fdm.ltpprz_ecef_vel = fdm.ltp_ecef_vel; // ???
498 
499  /* acceleration */
500  fdm.ecef_ecef_accel = to_pprz_ecef(
501  sphere->VelocityTransform(accel.Ign(),
502  gazebo::common::SphericalCoordinates::LOCAL,
503  gazebo::common::SphericalCoordinates::ECEF)); // Note: unused
504  fdm.body_ecef_accel = to_pprz_body(pose.rot.RotateVectorReverse(accel));
505  fdm.ltp_ecef_accel = to_pprz_ned(
506  sphere->VelocityTransform(accel.Ign(),
507  gazebo::common::SphericalCoordinates::LOCAL,
508  gazebo::common::SphericalCoordinates::GLOBAL)); // Note: unused
509  fdm.ltpprz_ecef_accel = fdm.ltp_ecef_accel; // ???
510  fdm.body_inertial_accel = fdm.body_ecef_accel; // Approximate, unused.
511 
512  fdm.body_accel = to_pprz_body(
513  pose.rot.RotateVectorReverse(accel.Ign() - world->Gravity()));
514 
515  /* attitude */
516  // ecef_to_body_quat: unused
517  fdm.ltp_to_body_eulers = to_global_pprz_eulers(local_to_global_quat * pose.rot);
518  double_quat_of_eulers(&(fdm.ltp_to_body_quat), &(fdm.ltp_to_body_eulers));
519  fdm.ltpprz_to_body_quat = fdm.ltp_to_body_quat; // unused
520  fdm.ltpprz_to_body_eulers = fdm.ltp_to_body_eulers; // unused
521 
522  /* angular velocity */
523  fdm.body_ecef_rotvel = to_pprz_rates(pose.rot.RotateVectorReverse(ang_vel));
524  fdm.body_inertial_rotvel = fdm.body_ecef_rotvel; // Approximate
525 
526  /* angular acceleration */
527  // body_ecef_rotaccel: unused
528  // body_inertial_rotaccel: unused
529  /* misc */
530  fdm.ltp_g = to_pprz_ltp(
531  sphere->VelocityTransform(-1 * world->Gravity(),
532  gazebo::common::SphericalCoordinates::LOCAL,
533  gazebo::common::SphericalCoordinates::GLOBAL)); // unused
534  fdm.ltp_h = to_pprz_ltp(
535  sphere->VelocityTransform(world->MagneticField(),
536  gazebo::common::SphericalCoordinates::LOCAL,
537  gazebo::common::SphericalCoordinates::GLOBAL));
538 
539  /* atmosphere */
540 #if GAZEBO_MAJOR_VERSION >= 8 && 0 // TODO implement
541 
542 #else
543  // Gazebo versions < 8 do not have atmosphere or wind support
544  // Use placeholder values. Valid for low altitude, low speed flights.
545  fdm.wind = (struct DoubleVect3) {0, 0, 0};
546  fdm.pressure_sl = 101325; // Pa
547 
548  fdm.airspeed = 0;
549  fdm.pressure = pprz_isa_pressure_of_height(fdm.hmsl, fdm.pressure_sl);
550  fdm.dynamic_pressure = fdm.pressure_sl; // Pa
551  fdm.temperature = 20.0; // C
552  fdm.aoa = 0; // rad
553  fdm.sideslip = 0; // rad
554 #endif
555  /* flight controls: unused */
556  fdm.elevator = 0;
557  fdm.flap = 0;
558  fdm.left_aileron = 0;
559  fdm.right_aileron = 0;
560  fdm.rudder = 0;
561  /* engine: unused */
562  fdm.num_engines = 0;
563 }

582 static void gazebo_write(double act_commands[], int commands_nb)
583 {
584  for (int i = 0; i < commands_nb; ++i) {
585  // Thrust setpoint
586  double sp = autopilot.motors_on ? act_commands[i] : 0.0; // Normalized thrust setpoint
587 
588  // Actuator dynamics, forces and torques
589 #ifdef NPS_ACTUATOR_TIME_CONSTANTS
590  // Delayed response from actuator
591  double u = update_first_order_low_pass(&gazebo_actuators.lowpass[i], sp);// Normalized actual thrust
592 #else

734 static void read_image(
735  struct image_t *img,
736  gazebo::sensors::CameraSensorPtr cam)
737 {
738  int xstart = 0;
739  int ystart = 0;
740 #if NPS_SIMULATE_MT9F002
741  if(cam->Name() == "front_camera") {
742  image_create(img, MT9F002_OUTPUT_WIDTH, MT9F002_OUTPUT_HEIGHT, IMAGE_YUV422);
743  xstart = cam->ImageWidth() * (0.5 + MT9F002_INITIAL_OFFSET_X) - MT9F002_OUTPUT_WIDTH / 2;
744  ystart = cam->ImageHeight() * (0.5 + MT9F002_INITIAL_OFFSET_Y) - MT9F002_OUTPUT_HEIGHT / 2;
745  } else {
746  image_create(img, cam->ImageWidth(), cam->ImageHeight(), IMAGE_YUV422);
747  }
748 #else
749  image_create(img, cam->ImageWidth(), cam->ImageHeight(), IMAGE_YUV422);
750 #endif
751  // Convert Gazebo's *RGB888* image to Paparazzi's YUV422
752  const uint8_t *data_rgb = cam->ImageData();
753  uint8_t *data_yuv = (uint8_t *)(img->buf);
754  for (int x = 0; x < img->w; ++x) {
755  for (int y = 0; y < img->h; ++y) {
756  int idx_rgb = 3 * (cam->ImageWidth() * (y + ystart) + (x + xstart));
757  int idx_yuv = 2 * (img->w * y + x);
758  int idx_px = img->w * y + x;
759  if (idx_px % 2 == 0) { // Pick U or V
760  data_yuv[idx_yuv] = -0.148 * data_rgb[idx_rgb]
761  - 0.291 * data_rgb[idx_rgb + 1]
762  + 0.439 * data_rgb[idx_rgb + 2] + 128; // U
763  } else {
764  data_yuv[idx_yuv] = 0.439 * data_rgb[idx_rgb]
765  - 0.368 * data_rgb[idx_rgb + 1]
766  - 0.071 * data_rgb[idx_rgb + 2] + 128; // V
767  }
768  data_yuv[idx_yuv + 1] = 0.257 * data_rgb[idx_rgb]
769  + 0.504 * data_rgb[idx_rgb + 1]
770  + 0.098 * data_rgb[idx_rgb + 2] + 16; // Y
771  }
772  }
773  // Fill miscellaneous fields
774  gazebo::common::Time ts = cam->LastMeasurementTime();
775  img->ts.tv_sec = ts.sec;
776  img->ts.tv_usec = ts.nsec / 1000.0;
777  img->pprz_ts = ts.Double() * 1e6;
778  img->buf_idx = 0; // unused
779 }

783 /*
784  * Simulate range sensors:
785  *
786  * In the airframe file, set NPS_SIMULATE_LASER_RANGE_ARRAY if you want to make use of the integrated Ray sensors.
787  * These are defined in their own model which is added to the ardrone.sdf (called range_sensors). Here you can add
788  * single ray point sensors with a specified position and orientation. It is also possible add noise.
789  *
790  * Within the airframe file (see e.g ardrone2_rangesensors_gazebo.xml), the amount of sensors
791  * (LASER_RANGE_ARRAY_NUM_SENSORS) and their orientations
792  * (LASER_RANGE_ARRAY_ORIENTATIONS={phi_1,theta_1,psi_1...phi_n,theta_n,psi_n} n = number of sensors) need to be
793  * specified as well. This is to keep it generic since this need to be done on the real platform with an external
794  * ray sensor. The function will compare the orientations from the ray sensors of gazebo, with the ones specified
795  * in the airframe, and will fill up an array to send and abi message to be used by other modules.
796  *
797  * NPS_GAZEBO_RANGE_SEND_AGL defines if the sensor that is defined as down should be used to send an AGL message.
798  * to send an OBSTACLE_DETECTION message.
799  *
800  * Functions:
801  * gazebo_init_range_sensors() -> Finds and initializes all ray sensors in gazebo
802  * gazebo_read_range_sensors() -> Reads and evaluates the ray sensors values, and sending it to other pprz modules
803  */

811 static float range_orientation[] = LASER_RANGE_ARRAY_ORIENTATIONS;
812 static uint8_t ray_sensor_agl_index = 255;

816 static void gazebo_init_range_sensors(void)
817 {
818  gazebo::sensors::SensorManager *mgr = gazebo::sensors::SensorManager::Instance();
819  gazebo::sensors::Sensor_V sensors = mgr->GetSensors(); // list of all sensors
820 
821  uint16_t sensor_count = model->GetSensorCount();
822 
823  gazebo::sensors::RaySensorPtr ray_sensor;
824  uint8_t ray_sensor_count_selected = 0;
825 
826  // Loop though all sensors and only select ray sensors, which are saved within a struct
827  for (uint16_t i = 0; i < sensor_count; i++) {
828  if (ray_sensor_count_selected > LASER_RANGE_ARRAY_NUM_SENSORS) {
829  break;
830  }
831 
832  if (sensors.at(i)->Type() == "ray") {
833  ray_sensor = static_pointer_cast<gazebo::sensors::RaySensor>(sensors.at(i));
834 
835  if (!ray_sensor) {
836  cout << "ERROR: Could not get pointer to raysensor " << i << "!" << endl;
837  continue;
838  }
839 
840  // Read out the pose from per ray sensors in gazebo relative to body
841  struct DoubleEulers pose_sensor = to_pprz_eulers(ray_sensor->Pose().Rot());
842 
843  struct DoubleQuat q_ray, q_def;
844  double_quat_of_eulers(&q_ray, &pose_sensor);
845 
846  /* Check the orientations of the ray sensors found in gazebo, if they are similar (within 5 deg) to the orientations
847  * given in the airframe file in LASER_RANGE_ARRAY_RANGE_ORIENTATION
848  */
849  for (int j = 0; j < LASER_RANGE_ARRAY_NUM_SENSORS; j++) {
850  struct DoubleEulers def = {0, range_orientation[j*2], range_orientation[j*2 + 1]};
851  double_quat_of_eulers(&q_def, &def);
852  // get angle between required angle and ray angle
853  double angle = acos(QUAT_DOT_PRODUCT(q_ray, q_def));
854 
855  if (fabs(angle) < RadOfDeg(5)) {
856  ray_sensor_count_selected++;
857  rays[j].sensor = ray_sensor;
858  rays[j].sensor->SetActive(true);
859 
860 #if LASER_RANGE_ARRAY_SEND_AGL
861  // find the sensor pointing down
862  if (fabs(range_orientation[j*2] + M_PI_2) < RadOfDeg(5)) {
863  ray_sensor_agl_index = j;
864  }
865 #endif
866  break;
867  }
868  }
869  }
870  }
871 
872  if (ray_sensor_count_selected != LASER_RANGE_ARRAY_NUM_SENSORS) {
873  cout << "ERROR: you have defined " << LASER_RANGE_ARRAY_NUM_SENSORS << " sensors in your airframe file, but only "
874  << ray_sensor_count_selected << " sensors have been found in the gazebo simulator, "
875  "with the same orientation as in the airframe file " << endl;
876  exit(0);
877  }
878  cout << "Initialized laser range array" << endl;
879 }
