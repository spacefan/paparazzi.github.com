69 void gec_generate_ephemeral_keys(struct gec_privkey *sk)
70 {
71  for (uint16_t i = 0; i < PPRZ_KEY_LEN; i += sizeof(uint32_t)) {
72  uint32_t tmp = rng_wait_and_get();
73  sk->priv[i] = (uint8_t) tmp;
74  sk->priv[i + 1] = (uint8_t)(tmp >> 8);
75  sk->priv[i + 2] = (uint8_t)(tmp >> 16);
76  sk->priv[i + 3] = (uint8_t)(tmp >> 24);
77  }
78  uint8_t basepoint[32] = {0};
79  basepoint[0] = 9; // default basepoint
80  Hacl_Curve25519_crypto_scalarmult(sk->pub, sk->priv, basepoint);
81  sk->ready = true;
82 }
