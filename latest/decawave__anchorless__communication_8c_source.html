111 static void decodeHighBytes(uint8_t bytes_received, uint8_t *received_message)
112 {
113  static uint8_t receive_buffer[4];
114  float tempfloat;
115  uint8_t data_received_count = 0;
116  uint8_t this_address = received_message[1];
117  uint8_t msg_from = received_message[2];
118  uint8_t msg_type = received_message[3];
119  uint8_t nodeIndex = msg_from - 1 - (uint8_t)(this_address < msg_from);
120  for (uint8_t i = 4; i < bytes_received - 1; i++) {
121  // Skip the begin marker (0), this address (1), remote address (2), message type (3), and end marker (bytes_received-1)
122  uint8_t var_byte = received_message[i];
123  if (var_byte == UWB_SERIAL_COMM_SPECIAL_BYTE) {
124  i++;
125  var_byte = var_byte + received_message[i];
126  }
127  if (data_received_count <= 4) {
128  receive_buffer[data_received_count] = var_byte;
129  }
130  data_received_count++;
131  }
132  if (data_received_count == 4) {
133  // Move memory from integer buffer to float variable
134  memcpy(&tempfloat, &receive_buffer, 4);
135  // Set the variable to the appropriate type and store it in state
136  handleNewStateValue(nodeIndex, msg_type, tempfloat);
137  }
138 }

145 static void encodeHighBytes(uint8_t *send_data, uint8_t msg_size, uint8_t *data_send_buffer, uint8_t *data_total_send)
146 {
147  uint8_t data_send_count = msg_size;
148  *data_total_send = 0;
149  for (uint8_t i = 0; i < data_send_count; i++) {
150  if (send_data[i] >= UWB_SERIAL_COMM_SPECIAL_BYTE) {
151  data_send_buffer[*data_total_send] = UWB_SERIAL_COMM_SPECIAL_BYTE;
152  (*data_total_send)++;
153  data_send_buffer[*data_total_send] = send_data[i] - UWB_SERIAL_COMM_SPECIAL_BYTE;
154  } else {
155  data_send_buffer[*data_total_send] = send_data[i];
156  }
157  (*data_total_send)++;
158  }
159 }

165 static void sendFloat(uint8_t msg_type, float data)
166 {
167  static uint8_t data_send_buffer[UWB_SERIAL_COMM_MAX_MESSAGE];
168  static uint8_t data_total_send = 0;
169 
170  // Make bytes of the float
171  uint8_t floatbyte[4];
172  memcpy(floatbyte, &data, 4);
173  encodeHighBytes(floatbyte, 4, data_send_buffer, &data_total_send);
174 
175  UWB_SERIAL_PORT->put_byte(UWB_SERIAL_PORT->periph, 0, UWB_SERIAL_COMM_START_MARKER);
176  UWB_SERIAL_PORT->put_byte(UWB_SERIAL_PORT->periph, 0, msg_type);
177 
178  for (uint8_t i = 0; i < data_total_send; i++) {
179  UWB_SERIAL_PORT->put_byte(UWB_SERIAL_PORT->periph, 0, data_send_buffer[i]);
180  }
181 
182  UWB_SERIAL_PORT->put_byte(UWB_SERIAL_PORT->periph, 0, UWB_SERIAL_COMM_END_MARKER);
183 }

220 static void getSerialData(uint8_t *bytes_received)
221 {
222  static bool in_progress = false;
223  static uint8_t var_byte;
224  static uint8_t received_message[UWB_SERIAL_COMM_MAX_MESSAGE];
225 
226  while (external_device->char_available(external_device->periph)) {
227  var_byte = UWB_SERIAL_PORT->get_byte(UWB_SERIAL_PORT->periph);
228 
229  if (var_byte == UWB_SERIAL_COMM_START_MARKER) {
230  (*bytes_received) = 0;
231  in_progress = true;
232  }
233 
234  if (in_progress) {
235  received_message[*bytes_received] = var_byte;
236  (*bytes_received)++;
237  }
238 
239  if (var_byte == UWB_SERIAL_COMM_END_MARKER) {
240  in_progress = false;
241  decodeHighBytes(*bytes_received, received_message);
242  }
243  }
244 }
