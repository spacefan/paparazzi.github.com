103 // raw message parsing function
104 static void jevois_parse(struct jevois_t *jv, char c)
105 {
106  switch (jv->state) {
107  case JV_SYNC:
108  // wait for sync (newline character)
109  if (c == '\n') {
110  jv->state = JV_TYPE;
111  jv->idx = 0;
112  jv->n = 0;
113  }
114  break;
115  case JV_TYPE:
116  jv->buf[jv->idx++] = c; // fill buffer
117  // parse type
118  if (jv->idx > 2) { // msg type + white space
119  if (jv->buf[0] == 'T' && jv->buf[1] == '1') {
120  jv->state = JV_COORD;
121  jv->msg.type = JEVOIS_MSG_T1;
122  jv->msg.nb = 1;
123  } else if (jv->buf[0] == 'N' && jv->buf[1] == '1') {
124  jv->state = JV_ID;
125  jv->msg.type = JEVOIS_MSG_N1;
126  jv->msg.nb = 1;
127  } else if (jv->buf[0] == 'D' && jv->buf[1] == '1') {
128  jv->state = JV_ID;
129  jv->msg.type = JEVOIS_MSG_D1;
130  jv->msg.nb = 2;
131  } else if (jv->buf[0] == 'T' && jv->buf[1] == '2') {
132  jv->state = JV_COORD;
133  jv->msg.type = JEVOIS_MSG_T2;
134  jv->msg.nb = 2;
135  } else if (jv->buf[0] == 'N' && jv->buf[1] == '2') {
136  jv->state = JV_ID;
137  jv->msg.type = JEVOIS_MSG_N2;
138  jv->msg.nb = 2;
139  } else if (jv->buf[0] == 'D' && jv->buf[1] == '2') {
140  jv->state = JV_ID;
141  jv->msg.type = JEVOIS_MSG_D2;
142  jv->msg.nb = 8;
143  } else if (jv->buf[0] == 'F' && jv->buf[1] == '2') {
144  jv->state = JV_ID;
145  jv->msg.type = JEVOIS_MSG_F2;
146  jv->msg.nb = 0;
147  } else if (jv->buf[0] == 'T' && jv->buf[1] == '3') {
148  jv->state = JV_COORD;
149  jv->msg.type = JEVOIS_MSG_T3;
150  jv->msg.nb = 3;
151  } else if (jv->buf[0] == 'N' && jv->buf[1] == '3') {
152  jv->state = JV_ID;
153  jv->msg.type = JEVOIS_MSG_N3;
154  jv->msg.nb = 3;
155  } else if (jv->buf[0] == 'D' && jv->buf[1] == '3') {
156  jv->state = JV_ID;
157  jv->msg.type = JEVOIS_MSG_D3;
158  jv->msg.nb = 3;
159  } else if (jv->buf[0] == 'F' && jv->buf[1] == '3') {
160  jv->state = JV_ID;
161  jv->msg.type = JEVOIS_MSG_F3;
162  jv->msg.nb = 0;
163  } else {
164  jv->state = JV_SYNC; // error
165  }
166  jv->idx = 0;
167  }
168  break;
169  case JV_ID:
170  if (JEVOIS_CHECK_DELIM(c)) {
171  jv->buf[jv->idx] = '\0'; // end string
172  if (jv->msg.type == JEVOIS_MSG_F2 ||
173  jv->msg.type == JEVOIS_MSG_F3) {
174  jv->state = JV_SIZE; // parse n before coordinates
175  } else {
176  jv->state = JV_COORD; // parse directly coordinates
177  }
178  break;
179  }
180  else {
181  jv->msg.id[jv->idx++] = c;
182  if (jv->idx > JEVOIS_MAX_LEN - 1) {
183  jv->state = JV_SYNC; // too long, return to sync
184  }
185  }
186  break;
187  case JV_SIZE:
188  if (JEVOIS_CHECK_DELIM(c)) {
189  jv->buf[jv->idx] = '\0'; // end string
190  jv->msg.nb = (uint8_t)atoi(jv->buf); // store size
191  jv->state = JV_COORD;
192  jv->idx = 0;
193  }
194  else {
195  jv->buf[jv->idx++] = c; // fill buffer
196  }
197  break;
198  case JV_COORD:
199  if (JEVOIS_CHECK_DELIM(c)) {
200  jv->buf[jv->idx] = '\0'; // end string
201  jv->msg.coord[jv->n++] = (int16_t)atoi(jv->buf); // store value
202  if (jv->n == jv->msg.nb) {
203  // got all coordinates, go to next state
204  jv->n = 0; // reset number of received elements
205  jv->idx = 0; // reset index
206  switch (jv->msg.type) {
207  case JEVOIS_MSG_T1:
208  case JEVOIS_MSG_T2:
209  case JEVOIS_MSG_T3:
210  jv->state = JV_SEND_MSG;
211  break;
212  case JEVOIS_MSG_N1:
213  case JEVOIS_MSG_N2:
214  case JEVOIS_MSG_N3:
215  case JEVOIS_MSG_D3:
216  jv->state = JV_DIM;
217  break;
218  case JEVOIS_MSG_D1:
219  case JEVOIS_MSG_D2:
220  case JEVOIS_MSG_F2:
221  case JEVOIS_MSG_F3:
222  jv->state = JV_EXTRA;
223  break;
224  default:
225  jv->state = JV_SYNC; // error
226  break;
227  }
228  }
229  jv->idx = 0; // reset index
230  }
231  else {
232  jv->buf[jv->idx++] = c; // fill buffer
233  }
234  break;
235  case JV_DIM:
236  if (JEVOIS_CHECK_DELIM(c)) {
237  jv->buf[jv->idx] = '\0'; // end string
238  jv->msg.dim[jv->n++] = (uint16_t)atoi(jv->buf); // store dimension
239  if (jv->n == jv->msg.nb) {
240  // got all dimensions, go to next state
241  jv->n = 0; // reset number of received elements
242  jv->idx = 0; // reset index
243  if (jv->msg.type == JEVOIS_MSG_D3) {
244  jv->state = JV_QUAT;
245  } else {
246  jv->state = JV_SEND_MSG;
247  }
248  break;
249  }
250  jv->idx = 0; // reset index
251  }
252  else {
253  jv->buf[jv->idx++] = c; // fill buffer
254  }
255  break;
256  case JV_QUAT:
257  if (JEVOIS_CHECK_DELIM(c)) {
258  jv->buf[jv->idx] = '\0';
259  float q = 0.f;//(float) atof(jv->buf);
260  switch (jv->n) {
261  case 0:
262  jv->msg.quat.qi = q; // TODO check quaternion order
263  break;
264  case 1:
265  jv->msg.quat.qx = q;
266  break;
267  case 2:
268  jv->msg.quat.qy = q;
269  break;
270  case 3:
271  jv->msg.quat.qz = q;
272  break;
273  case 4:
274  jv->state = JV_EXTRA;
275  break;
276  default:
277  jv->state = JV_SYNC; // error
278  break;
279  }
280  jv->n++;
281  jv->idx = 0; // reset index
282  }
283  else {
284  jv->buf[jv->idx++] = c; // fill buffer
285  }
286  break;
287  case JV_EXTRA:
288  if (JEVOIS_CHECK_DELIM(c)) {
289  jv->buf[jv->idx] = '\0'; // end string
290  jv->state = JV_SEND_MSG;
291  }
292  else {
293  jv->msg.id[jv->idx++] = c; // store extra string
294  if (jv->idx > JEVOIS_MAX_LEN - 1) {
295  jv->state = JV_SYNC; // too long, return to sync
296  }
297  }
298  break;
299  case JV_SEND_MSG:
300  // send ABI message
301  AbiSendMsgJEVOIS_MSG(CAM_JEVOIS_ID,
302  jv->msg.type,
303  jv->msg.id,
304  jv->msg.nb,
305  jv->msg.coord,
306  jv->msg.dim,
307  jv->msg.quat,
308  jv->msg.extra);
309  // also send specific messages if needed
310  jevois_send_message();
311  jv->state = JV_SYNC;
312  break;
313  default:
314  // error, back to SYNC
315  jv->state = JV_SYNC;
316  break;
317  }
318 }
